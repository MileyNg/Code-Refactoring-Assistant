#include<stdio.h>
#include<math.h>

int main(){
  int n,i,sum;
  int s[1000];
  double ave,sigm;
  while(1){
    scanf("%d",&n);
    if(n==0){
      break;
    }
    sum=0;
    for(i=0;i<n;i++){
      scanf("%d",&s[i]);
      sum+=s[i];
    }
    ave=sum/n;
    sigm=0;
    for(i=0;i<n;i++){
      sigm+=pow((s[i]-ave),2);
    }
    printf("%f\n",sqrt(sigm/n));
  }
  return 0;
}
