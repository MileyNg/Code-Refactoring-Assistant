#include <stdio.h>
main()
{
  int i,j,key,n,k;
  int A[100];
  
  scanf("%d",&n);
  for(i=0;i<n;i++)
    {
      scanf("%d",&A[i]);
    }
  for(i=0;i<n;i++)
    {
      printf("\n");
      key=A[i];
      j=i-1;
      while(j>=0&&A[j]>key)
	{
	  A[j+1]=A[j];
	  j--;
	}
      A[j+1]=key;
      for(k=0;k<n;k++)
	{
	  printf("%d",A[k]);
	}
    }
  printf("\n");
}
