#include <cstdio>

#include <cstdlib>

#include <cstring>

#include <string>

#include <map>

#include <vector>

#include <iostream>

#include <set>

#include <algorithm>

using namespace std;



char s[150];
int zhan[150];

bool solve(){
	int len,top,i;
	gets(s);
	len=strlen(s);
	if(len==1 && s[0]=='.')return false;
	top=0;
	for(i=0;i<len;++i){
		if(s[i]=='('){
			zhan[++top]=0;
		}else if(s[i]=='['){
			zhan[++top]=1;
		}else if(s[i]==')'){
			if(top==0 || zhan[top]==1){
				printf("no\n");
				return true;
			}
			--top;
		}else if(s[i]==']'){
			if(top==0 || zhan[top]==0){
				printf("no\n");
				return true;
			}
			--top;
		}
	}
	if(top==0)printf("yes\n");else printf("no\n");
	return true;
}


int main(){

	while(solve());

	return 0;

}
