{

	{
		int sum = test1 + test2;
		if(test1 == -1 || test2 == -1)
		{
			printf("F\n");
		}
		else if(sum >= 80)
		{
			printf("A\n");
		}
		else if(sum >= 65 && sum < 80)
		{
			printf("B\n");
		}
		else if(sum >= 50 && sum < 65)
		{
			printf("C\n");
		}
		else if(sum >= 30 && sum <50)
		{
			if(test3 >= 50)
			{
				printf("C\n");
			}
			else
			{
				printf("D\n");
			}
		}
		else if(sum < 30)
		{
			printf("F\n");
		}

		scanf("%d %d %d",&test1,&test2,&test3);
	}
	return 0;
}
