#include<stdio.h>
#include<string.h>

int main(){

  int n,m,i,j,k;
  char before[100000]={'\0'};
  char after[100000]={'\0'};
  char str[100000]={'\0'};

  while(1){

    scanf(" %d",&n);

    if(n==0){
      break;
    }

    for(i=0;i<n;i++){
      scanf(" %c %c",&before[i],&after[i]);
    }

    scanf("%d",&m);

    for(j=0;j<m;j++){

      scanf(" %c",&str[j]);

      for(k=0;k<n;k++){
	if(str[j]==before[k]){
	  str[j]=after[k];
	  break;
	}
      }

    }

    for(i=0;i<m;i++){
      printf("%c",str[i]);
    }
    printf("\n");
  }

  return 0;
}
