#include <iostream>
#include <algorithm>

using namespace std;

bool ok(int n, int a, int b){
    if(a<b) swap(a,b);
    for(int i=0; i<=n; i+=a){
        if((n-i)%b==0) return true;
    }
    return false;
}

int main(){
    int n,a,b;

    while(cin>>n>>a>>b, n|a|b){
        int ans=0;
        for(int i=1; i<=n; i++){
            if(ok(i,a,b)) continue;
            ans++;
        }
        cout << ans << endl;
    }
}
