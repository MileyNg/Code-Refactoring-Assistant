#include<stdio.h>
int main(){

  int n,i,j;
  int num[50100]={0};
 
  for(i=3;i<50100;i+=2){
    if(num[i]==1)
      continue;
    for(j=3*i;j<50100;j+=i*2)
      num[j]=1;
  }
  while(scanf("%d",&n)!=EOF){
	int i;
	int tmp=2;
	for(i=3;i<n;i+=2)
	  if(num[i]==0)tmp=i;
	printf("%d",tmp);
	if(n%2==0)
	  i=n+1;
	else
	  i=n+2;
	for(;i<50100;i+=2)
	  if(num[i]==0)break;
	printf(" %d\n",i);
  }
  return 0;
}
