#include <iostream>

#define MAX 100
#define rep(i,n) for(int i = 0 ; i < n ; i++)

int main(){
  int N, M, a[MAX][MAX], b[MAX], c[MAX];

  std::cin >> N >> M;
  rep(i,N){
    rep(j,M){
      std::cin >> a[i][j];
    }
    c[i] = 0;
  }
  rep(i,M){
    std::cin >> b[i];
  }

  rep(i,N){
    rep(j,N){
      c[i] += a[i][j]*b[j];
    }
  }
  rep(i,N){
    std::cout << c[i] << std::endl;
  }

  return 0;
}
