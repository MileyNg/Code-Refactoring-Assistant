#include <stdio.h>
#define N 2000000

int B[N], n;

void CountingSort( int A[], int B[], int k ){
  int i, j, C[k];
  for( i = 0; i <= k; i++ ){
    C[i] = 0;
  }
  for( j = 1; j < n; j++ ){
    C[A[j]]++;
  }
  for( i = 1; i <= k; i++ ){
    C[i] = C[i] + C[i-1];
  }
  for( j = n; j > 0; j-- ){
    B[C[A[j]]] = A[j];
    C[A[j]]--;
  }
}

main(){

  int i, A[N];
  scanf("%d", &n);

  for( i = 0; i < n; i++ ){
    scanf("%d", &A[i]);
  }
  int max = 0;
  for( i = 0; i < n; i++ ){
    if(max < A[i]){
      max = A[i];
    }
  }
  CountingSort(A, B, max);
  
  for( i = 0; i < n; i++ ){
    if(i > 0) printf(" ");
    printf("%d", B[i]);
  }
  printf("\n");
  return 0;
}
