#include <stdio.h>
int main()
{
	int n,q;
	int i,j;
	int x[101],y[101];
	int ans=0;
	
	scanf("%d",&n);
	for(i=0;i<n;i++)
		scanf("%d",&x[i]);
	
	scanf("%d",&q);
	for(i=0;i<q;i++)
		scanf("%d",&y[i]);
	
	for(i=0;i<n;i++){
		for(j=0;j<q;j+){
			if(x[i] == y[j])
				ans++;
		}
	}
	
	printf("%d\n",ans);
	
	return 0;
}
