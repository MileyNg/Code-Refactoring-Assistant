#include <stdio.h>

#define N 2000000
#define M 10000

int  A[N],B[N], C[M];

void CountingSort(int);
int n;
 


int main(){
  
  int i,j, max = 0;
  scanf("%d", &n);

  for(i = 1 ; i <=n ; i++ ){
    
    scanf("%d", &A[i]);

    if(A[i] > max) max = A[i];
     
 




  

  
 
  
 
  for(i = 0 ; i <= max ; i++){
    
    C[i] = 0;
  
  }
  
    for(j = 1 ; j <= n ; j++){
      
      C[A[j]] = C[A[j]]+1;
      
    }

  
  
  for(i = 1 ; i <= max ; i++){
    
    C[i] = C[i] + C[i - 1];
  }
  
  for(j = n ; j >= 1 ; j--){
    
    B[C[A[j]]] = A[j];
    C[A[j]] = C[A[j]]-1;
    
  }

for(i=1;i<n;i++){
    printf("%d ",B[i]);
  }
  printf("%d\n",B[n]);
   
  return 0;
  
 
}
