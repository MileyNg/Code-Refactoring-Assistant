#include<stdio.h>
#include<math.h>


int main(void)
{
  int r,theta,rad=0;
  double x,y;
  x=0;y=0;
  while(1){
    //printf("%d %d\n",r,theta);
    //rewind(stdin);
    scanf("%d,%d",&r,&theta);
    if(r==0 && theta==0)
      break;
    rad+=theta;
    x+=r*cos((double)(rad+90)*M_PI/180);
    y+=r*sin((double)(rad+90)*M_PI/180);
  }

  printf("%d\n%d\n",(int)x,(int)y);

  return 0;
}
