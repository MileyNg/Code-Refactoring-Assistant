#include <stdio.h>
#include <string.h>

void r_shuffle(int n);
void cut(int k, int n);

char c[220];

int main(void)
{
	int n, m;
	int k, i;
	
	scanf("%d %d", &n, &m);
	
	for (i = 0; i < n * 2; i++){
		c[i] = i + 1;
	}
	
	for (i = 0; i < m; i++){
		scanf("%d", &k);
		
		if (k == 0){
			r_shuffle(n);
		}
		else{
			cut(k, n);
		}
	}
	for (i = 0; i < n * 2; i++){
		printf("%d\n", c[i]);
	}
}

void r_shuffle(int n)
{
	char tmp[220];
	int i;
	
	for (i = 0; i < n; i++){
		tmp[i * 2] = c[i];
		tmp[i * 2 + 1] = c[n + i];
	}
	tmp[i * 2] = '\0';
	
	strcpy(c, tmp);
}

void cut(int k, int n)
{
	char tmp[220];
	int i, j;
	
	for (i = 0; i < 2 * n - k; i++){
		tmp[i] = c[k + i];
	}
	for (j = 0; j < k; j++){
		tmp[i + j] = c[j];
	}
	tmp[j + i] = '\0';
	
	strcpy(c, tmp);
}
