#include <stdio.h>

int func(int a ,int b,int c,int d,int e,int f){
	return (c*d-a*f)/(b*d-a*e);
}

int main(void){
	double a,b,c,d,e,f,x,y;
	while(scanf("%lf %lf %lf %lf %lf %lf",&a,&b,&c,&d,&e,&f) != EOF){
		x=func(a,b,c,d,e,f)+0.00001;
		y=((c-(a*x))/b)+0.00001;
		printf("%.3lf %.3lf\n",x,y);
	}
	return 0;
}
