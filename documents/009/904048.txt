int dp[401];


int main(){
	int n, m; cin >> n >> m;
	vector<vector<int> > edge(n, vector<int>(n, 0));
	FOR(i, m){
		int s, t, d; cin >> s >> t >> d;
		edge[s][t] = d;
	}

	vector<int> s;
	s.push_back(0);
	while (1){
		vector<int> c;
		FOR(i, s.size()){
			FOR(j, n){
				if (edge[s[i]][j]){
					dp[j] = max(dp[j], dp[s[i]] + edge[s[i]][j]);
					c.push_back(j);
				}
			}
		}
		
		s = c;

		if (!s.size()) break;
	}

	cout << dp[n - 1] << endl;
	return 0;
}
