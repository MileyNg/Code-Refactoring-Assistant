#include<stdio.h>
int main(){
	int a;
	int b;
	char op;
	int p;
	int m;
	int w;
	int k;
	scanf("%d %c %d",&a,&op,&b);
	if(op=='+'){
		p=a+b;
		printf("%d\n",p);
	}else if(op=='-'){
		m=a-b;
		printf("%d\n",m);
	}else if(op=='*'){
		k=a*b;
		printf("%d\n",k);
	}else if(op=='/'){
		w=a/b;
		printf("%d\n",w);
	}else if(op=='?'){
		return 0;
	}
}
