#include <iostream>
#include <string>

using namespace std;

int main() {
	string abc = "XYZABCDEFGHIJKLMNOPQRSTUVWXYZ";
	string a;
	cin >> a;
	for (int i = 0; i < a.size(); i++) {
		for (int j = 28; j >= 0 ; j--) {
			if (a[i] == abc[j]) {
				cout << abc[j-3];
			}
		}
	}
	cout << endl;
	return 0;
}
