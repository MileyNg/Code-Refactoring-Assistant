#include <iostream>
#include <fstream>
#include <string>

using namespace std;

int main(){
	string s, t;

	while (getline(cin, s, '\n'), s != "END OF INPUT"){
		for (int i = 0; i < s.length(); i++){
			if (isupper(s[i]))
				t += tolower(s[i]);
			else
				t += toupper(s[i]);
		}
		cout << t << endl;
	}
	return 0;
}
