#include <iostream>
#include <string>
#include <queue>
using namespace std;

string vl = "khtps";
string vowels = "aiueo";

int main(){
	string s;
	while( cin >> s, s != "#" ){
		queue<int> check;
		bool is_continued = false, former_is = false;
		int former = 0;
		string w = "";
		for( int i = 0; i < s.size(); i++ ){
			int iuao= 0;
			w += s[i];
			//cout << "w:" << w << "--------" << endl;
			//cout << "is_continued:" << is_continued << endl;
			//cout << "former_is:" << former_is << endl;
			//cout << "former:" << former << endl;
			//for( int j = 0; j < vowels.size(); j++ ){
				if( s[i] == 'i' || s[i] == 'u' ) iuao = 1;
				else if( s[i] == 'a' || s[i] == 'o' ) iuao = 2;
				else if( s[i] == 'e' || s[i] == '\'' ) iuao = 3;
				//if( iuao ) break;
			//}
			bool ok = false, f = false;
			if( !iuao ) continue;	// 母音でなければ次へ
			// 母音なら以下
			
			// 無声子音+母音かチェック
			for( int j = 0; j < vl.size(); j++ ){
				if( w[0] == vl[j] ){
					f = true;
					break;
				}
			}

			//cout << "f: " << ok << endl;
			if( !f ){	// 無声子音+母音でないとき
				former_is = false;
				goto end;
			}
			//if( iuao == 1 && s.size() == i+1 ) f = true;
			if( !former_is ) ok = false;
			else{
				if( s[former] == 'i' || s[former] == 'u' ) ok = true;
				if( iuao == 2 && s[former] == s[i] &&  w[0] != w[1] ) ok = true;
			}

			if( !f ){
				former_is = false;
			}else{
				if( former == 0 && i+1 != s.size() ) ok = false;
				former_is = true;
			}
			end:
			w = "";
			if( ok && !is_continued ){
				check.push(former);
				is_continued = true;
			}else{
				is_continued = false;
			}
			former = i;
			if( !is_continued && f && iuao == 1 && i == s.size()-1 ) check.push(i);
		}
		for( int i = 0; i < s.size(); i++ ){
			if( !check.empty() && i == check.front() ){
				cout << "(" << s[i] << ")";
				check.pop();
			}else{
				cout << s[i];
			}
		}
		cout << endl;
	}
}
