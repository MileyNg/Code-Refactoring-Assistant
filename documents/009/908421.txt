#include <cstdio>
#include <cstring>
#include <cstdlib>
#include <cmath>
#include <vector>
#include <string>
#include <map>
#include <queue>
#include <stack>
#include <algorithm>

#define rep(i, j) FOR(i, 0, j)
#define FOR(i, j, k) for(int i = j; i < k; ++i)

using namespace std;

int main(){
  int n, L[2], D[2], M[2], hh, mm, MM;
  while(scanf("%d", &n) && n){
    L[0] = L[1] = D[0] = D[1] = M[0] = M[1] = 0;
    rep(i, n){
      scanf("%d:%d%d", &hh, &mm, &MM);
      if(hh >= 15 && hh < 18) continue;
      if(hh < 15){
	L[0]++;
	if((MM < mm?MM+60:MM) - mm <= 8) L[1]++;
      }else if(hh < 21){
	D[0]++;
	if((MM < mm?MM+60:MM) - mm <= 8) D[1]++;
      }else{
	M[0]++;
	if((MM < mm?MM+60:MM) - mm <= 8) M[1]++;
      }
    }
    printf("lunch ");
    if(L[0]) printf("%d\n", L[1] * 100 / L[0]);
    else puts("no guest");
    printf("dinnuer ");
    if(D[0]) printf("%d\n", D[1] * 100 / D[0]);
    else puts("no guest");
    printf("midnight ");
    if(M[0]) printf("%d\n", M[1] * 100 / M[0]);
    else puts("no guest");
  }
  return 0;
}
