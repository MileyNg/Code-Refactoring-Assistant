#include <iostream>
using namespace std;

int main(){
	int apart[4][3][10];
	for (int i = 0; i < 4; i++){
		for (int j = 0; j < 3; j++){
			for (int k = 0; k < 10; k++){
				apart[i][j][k] = 0;
			}
		}
	}
	int n;
	cin >> n;
	int i, j, k, p;
	for (int c = 0; c < n; c++){
		cin >> i >> j >> k >> p;
		apart[i - 1][j - 1][k - 1] += p;
	}
	for (int i = 0; i < 4; i++){
		for (int j = 0; j < 3; j++){
			for (int k = 0; k < 10; k++){
				cout <<" "<< apart[i][j][k];
			}
			cout << endl;
		}
		if (i!=3)
			cout << "####################" << endl;
	}


	return 0;
}
