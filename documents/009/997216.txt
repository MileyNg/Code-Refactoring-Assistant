#include <stdio.h>
#include <math.h>
#include <limits.h>

int r[100], c[100];
int M[100][100];

int matrix(int left, int right)
{
  int res, i;

  if (M[left][right] >= 0){
    return M[left][right];
  }
  if (left == right){
    return 0;
  }
  res = INT_MAX;
  for (i = left; i < right; i++){
    res = fmin(res, matrix(left, i) + matrix(i + 1, right) + r[left] * c[i] * c[right]);
  }
  M[left][right] = res;
  return M[left][right];
}

int main()
{
  int N;
  int i, j;
  
  scanf("%d", &N);
  for (i = 0; i < N; i++){
    scanf("%d %d", &r[i], &c[i]);
  }
  for(i = 0; i < 100; i++){
    for(j = 0; j < 100; j++){
      M[i][j] = -1;
    }
  }
  printf("%d\n", matrix(0, N - 1));
  return 0;
}
