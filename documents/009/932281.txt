#include <stdio.h>

int n,a[20];

int solve(int p,int key){
  if(!key){
    return 1;
  }

  if(key < 0 || p >= n){
    return 0;
  }

  if(solve(p+1,key) || solve(p+1,key-a[p])){
    return 1;
  }

  return 0;
}

int main(){
  int i,q,key;
  scanf("%d" ,&n);
  for(i = 0 ; i < n ; i++){
    scanf("%d" ,a + i);
  }

  scanf("%d" ,&q);
  while(q--){
    scanf("%d" ,&key);
    if(solve(0,key)){
      printf("yes\n");
    }else{
      printf("no\n");
    }
  }

  return 0;
}
