#include<stdio.h>

main(){
  int S[10000],T[500],n,q,i,j,count=0;

  scanf("%d",&n);
  for(i=0 ; i<n ; i++) scanf("%d",&S[i]);

  scanf("%d",&q);
  for(i=0 ; i<q ; i++) scanf("%d",&T[i]);

  for(i=0 ; i<n ; i++){
    for(j=0 ; j<q ; j++){
      if(S[i] == T[i]){
	count++;
	break;
      }
    }
  }

  printf("%d\n",count);
  return 0;
}
