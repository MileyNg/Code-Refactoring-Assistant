#include <stdio.h>
#include <algorithm>

using namespace std;

char data[1000000];

int main(void)
{
	double a1, a2, b1, b2, c1, c2, d1, d2;
	double m, m2;
	
	while (scanf("%lf %lf %lf %lf %lf %lf %lf %lf", &a1, &a2, &b1, &b2, &c1, &c2, &d1, &d2) != EOF){
		
		if (b1 - a1 == 0) m = 1000000000000;
		else m = (b2 - a2) / (b1 - a1);
		
		if (d1 - c1 == 0) m2 = 1000000000000;  
		else m2 = (d2 - c2) / (d1 - c1);
		
		if (m * m2 == -1 || (m == 1000000000000 && m2 == 0) || (m2 == 1000000000000 && m == 0)){
			printf("YES\n");
		}
		else{
			printf("NO\n");
		}
	}
	
	return (0);
}
