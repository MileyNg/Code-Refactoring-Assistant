#include <stdio.h>

int IsPrime(int);

int main(){
  int n,A[10000],i,x=0,count=0;
  scanf("%d",&n);
  for(i=0;i<n;i++){
    scanf("%d",&A[i]);
  }

 
for(i=0;i<n;i++){
  x=IsPrime(A[i]);
  if(x==1)count++;
  }

 printf("%d\n",count);

 return 0;
}



int IsPrime(int k){
 int i;
 
if(k<2)
  return 0;
 else if(k==2)
   return 1;

if(k%2==0)
  return 0;

 for(i=3;i<=k/i;i+=2){
  if(k%i==0)
    return 0;
 }
return 1;
}
