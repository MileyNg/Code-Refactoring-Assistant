#include <iostream>
#include <vector>
using namespace std;
 
typedef vector<int> vi;
#define loop(i,a,b) for(int i=(a); i<int(b);i++)
#define rep(i,b) loop(i,0,b)
 
int main(){
    int n,m;
    while(cin >>n>>m){
        if(n==0&&m==0)break;
        vector<bool> in(n,true);
        int num=-1;
        for(int i=0;i<n-1;i++){
            rep(j,m){
                num++;
                if(num>=n)num=0;
                if(!in[num]){j--;continue;}
            }
            // cout<<num<<endl;
            in[num]=false;
        }
 
        // cout << "end" << endl;
        rep(i,n){
            if(in[i]){
                cout << i+1 << endl;
            }  
        }
    }
}
