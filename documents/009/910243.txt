include<stdio.h>

int main(){
  int a,b,buf;
  
  scanf("%d%d", &a, &b);
  
  if(a >= b){
    while((buf = a % b) != 0){
      a = b;
      b = buf;
    }
    printf("%d\n", b);
  }
  else
    {
      while((buf = b % a) != 0){
	b = a;
	a = buf;
      }
      printf("%d\n", a);
    }
  
  return 0;
}
