#include <stdio.h>

int main(){
	int count[4][3][10] = {0}, b, f, r, v, n, i;
	scanf("%d", &n);
	for(i = 0; i < n; i++){
		scanf("%d %d %d %d", &b, &f, &r, &v);
		count[b - 1][f - 1][r - 1] += v;
	}
	for(b = 0; b < 4; b++){
		for(f = 0; f < 4; f++){
			for(r = 0; r < 10; r++){
				if(b == 3 && f == 3) break;
				if(b != 3 && f == 3){
					printf("##");
				} else {
					printf(" %d", count[b][f][r]);
				}
			}
		if(b == 3 && f == 3) break;
			printf("\n");
		}
	}
	return 0;
}
