#include<iostream>
#include<cstring>
#include<string>
using namespace std;
const int maxs=201;
char map[201][201];
int h,w,len;
void init()
{
	for(int i=0;i<=102;i++)
		for(int j=0;j<=102;j++)
			map[i][j]='X';
}
void find()
{
	int x,y;
	int flag=0;
	for(int i=1;i<=h;i++)
		for(int j=1;j<=w;j++)
		{
			if(map[i][j]=='E'||map[i][j]=='W'||map[i][j]=='S'||map[i][j]=='N')
			{
				x=i;
				y=j;
				flag=1;
				break;
			}
			if(flag)
				break;
		}
		//cout<<x<<y<<endl;
		int tag;
		if(map[x][y]=='E')tag=0;
		else if(map[x][y]=='W')tag=1;
		else if(map[x][y]=='S')tag=2;
		else if(map[x][y]=='N')tag=3;
		int count=0;
	while(len)
	{
		if(tag==0)
		{
			if(map[x][y+1]=='X'||map[x][y+1]=='#')
			{
				tag=2;
			}
			else
			{
				y+=1;
				len-=1;
			}
			//	cout<<count++<<' '<<x<<' '<<y<<' '<<tag<<endl;
		}
		else if(tag==1)
		{
			if(map[x][y-1]=='X'||map[x][y-1]=='#')
			{
				tag=3;
			}
			else
			{
				y-=1;
				len-=1;
			}
			//	cout<<count++<<' '<<x<<' '<<y<<' '<<tag<<endl;
		}
		else if(tag==2)
		{
			if(map[x+1][y]=='X'||map[x+1][y]=='#')
			{
				tag=1;
			}
			else
			{
				x+=1;
				len-=1;
			}
			//cout<<count++<<' '<<x<<' '<<y<<' '<<tag<<endl;
		}
		else if(tag==3)
		{
			if(map[x-1][y]=='X'||map[x-1][y]=='#')
			{
				tag=0;
			}
			else 
			{
				x-=1;
				len-=1;
			}
		}
			//cout<<count++<<' '<<x<<' '<<y<<' '<<tag<<endl;
	}
	/*if(map[x][y]=='E')tag=0;
		else if(map[x][y]=='W')tag=1;
		else if(map[x][y]=='S')tag=2;
		else if(map[x][y]=='N')tag=3;*/
	char c;
	if(tag==0)
		c='E';
	else if(tag==1)
		c='W';
	else if(tag==2)
		c='S';
	else if(tag==3)
		c='N';
	cout<<x<<' '<<y<<' '<<c<<endl;
}
int main()
{
	while(cin>>h>>w>>len)
	{
		if(h+w+len==0)break;
		init();
		for(int i=1;i<=h;i++)
			for(int j=1;j<=w;j++)
				cin>>map[i][j];
		find();
	}
	//system("pause");
}
