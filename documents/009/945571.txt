#include<stdio.h>
#include<string.h>

#define M 17000000/* your task*/
#define L 14

char H[M][L]; /* Hash Table */

int getChar(char ch){
  if ( ch == 'A') return 1;
  else if ( ch == 'C') return 2;
  else if ( ch == 'G') return 3;
  else if ( ch == 'T') return 4;
}

/* convert a string into an integer value */
long long getKey(char str[]){
  long long sum = 0, p = 1, i;
  for ( i = 0; i < strlen(str); i++ ){
    sum += p*(getChar(str[i]));
    p *= 5;
  }
  return sum;
}

int h1(int key){ return key%M; }
int h2(int key){ return (key+1)%(M-1); }

int find(char str[]){
  /* your task */ 
  int ans1,ans2,clu,ansC,i=0;
  long long key;
  
  key=getKey(str);
  ans1=h1(key);
  ans2=h2(key);
  
  while(1){
    clu=(ans2*i+ans1)%M;
    ansC=strcmp(str,H[clu]);
    if(ansC==0) return 1;
    if(H[clu][0]=='\0'||i==M) return 0;
    i++;
  }
}

int insert(char str[]){
  /* your task */  
  int ans1,ans2,clu,i=0;
  long long key;
  
  key=getKey(str);
  ans1=h1(key);
  ans2=h2(key);
  
  while(1){
    clu=(ans2*i+ans1)%M;
    if(H[clu][0]=='\0'){
      break;
    }
    i++;
  }
  strcpy(H[clu],str);
}
