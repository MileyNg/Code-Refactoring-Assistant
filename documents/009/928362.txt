#include<stdio.h>
#include<string.h>
#define LEN 100005

typedef struct pp{
  char name[100];
  int t;
}P;

P Q[LEN+1];
int head=1, tail, n;

void enqueue(P x){
  Q[tail]=x;
  if(tail+1==LEN){
    tail=0;
  }
  else{
    tail++;
  }
}

P dequeue(){
  P a=Q[head];
  if (head+1>= LEN) {
    head=0;
  }
  else{
    head++;
  }
  return a;
}


int main(){
  int elaps = 0;
  int count=0;
  int i,q;
  
  scanf("%d %d", &n, &q);
  
  for ( i = 1; i <= n; i++){
    scanf("%s", Q[i].name);
    scanf("%d", &Q[i].t);
  }
  
  tail=n+1;
  
  while(head<tail){
    if(Q[head].t<=q){
      printf("%s",Q[head].name);
      elaps=elaps+Q[head].t;
      printf(" %d\n",elaps);
      dequeue();
    }
    else {
      count++;
      elaps=elaps+q;
      Q[head].t=Q[head].t-q;
      enqueue(Q[head]);
      dequeue();
    }
  } 
  return 0;
}
