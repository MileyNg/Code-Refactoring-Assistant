int main(int argc, const char * argv[])
{
    int h,w;
    int i,j;
    
    do{
        scanf("%d %d",&h, &w);
        for (i=0; i < h; i++) {
            for (j=0; j < w; j++) {
                printf("#");
            }
            printf("\n\n");
        }
        
    }while (h!=0 && w!=0);
    
    return 0;
}
