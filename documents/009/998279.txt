#include<iostream>
using namespace std;

int main(){
  int n,x,i,j,k,data,c=0;


  while(1){
  
    cin >> n >> x;
    
    if ( n == 0 && x == 0)break;
    else{
      for ( i = 1 ; i <= n ; i++){
	for(j = i+1 ; j <= n ; j++){
	  for( k = j+1; k <= n ; k++){
	    data = i + j + k;
	  if ( data == x){
	    c++;
	  }
	  }
	}
      }
    }
 
  }
			      
        cout << c<<endl;
   



  return 0;
}
