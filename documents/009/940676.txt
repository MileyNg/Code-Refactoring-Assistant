#include<stdio.h>
#include<stdlib.h>
#include<string.h>
int main(){
	long int S[10001],T[501],U[501];
    int i,j,lengh1,lengh2,count=0;
	
	scanf("%d",&lengh1);
	for(i=1;i<=lengh1;i++){
		scanf("%d",&S[i]);}
	
	scanf("%d",&lengh2);
	for(i=1;i<=lengh2;i++){
		scanf("%d",&T[i]);}
	
	for(i=1;i<=lengh1;i++){
		for(j=1;j<=lengh2;j++){
			if(S[i]==T[j]){count++; break;}
			
		}
		
		memcpy(U,T,sizeof(int)*lengh2);
		for(i=1;i<=lengh2;i++){
			for(j=1;j<=lengh2;j++){
				if(T[i]==U[j]) count--;
			}}
			
	}	
	printf("%d\n",count);
	
	return 0;
}
