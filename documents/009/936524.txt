#include<cstdio>
#include<algorithm>
using namespace std;

int getSameNum(int[],int[],int ,int );

int main(){
  int n,q;
  scanf("%d",&n);
  int s[n];
  for(int i=0;i<n;i++) 
    scanf("%d",&s[i]);
  scanf("%d",&q);
  int t[q];
  for(int i=0;i<q;i++)
    scanf("%d",&t[i]);
  sort(t,t+n);
  printf("%d\n",getSameNum(s,t,n,q));
  return 0;
}

int getSameNum(int s[],int t[],int n,int q){
  int cnt=0;
  for(int i=0;i<q;i++)
    for(int j=0;j<n;j++)
      if(t[i]==s[j]){
	cnt++;
	break;
      }
  return cnt;
}
