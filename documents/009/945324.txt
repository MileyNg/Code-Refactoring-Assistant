#include <stdio.h>
#define N 10000
#define Q 500

main(){  
  int n,q,i,j,count=0;
  int S[N],T[Q];
  
  scanf("%d",&n);

  for(i=0;i<=n-1;i++){
    scanf("%d",&S[i]);
  }
  
  scanf("%d",&q);
  
  for(j=0;j<=q-1;j++){
    scanf("%d",&T[j]);
  }
  
  for(i = 0 ; i <= n-1 ; i++){
    for(j = 0 ; j <= q-1 ; j++){
      if(S[i]==T[j]){
	count++;
      }
    }
  }
  printf("%d\n",count);
  return 0;
}
