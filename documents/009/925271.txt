#include <stdio.h>
#include <string.h>

int main(){
    char number[1001];
    int num,sum;
    while (1) {
        scanf("%s",number);
        sum=0;
        if (number[0]== '0') {
            break;
        }
        for (int i=0; i < strlen(number); i++) {
            num = number[i] - '0';
            printf("%c = %d\n",number[i], num);
            sum+=num;
        }
        printf("%d\n",sum);
    }
    return 0;
}
