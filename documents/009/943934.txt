#include<stdio.h>
main(){
  int m,f,r;
  m,f<=50;
  r<=100;
  while(1){
    scanf("%d %d %d",&m,&f,&r);
    if(m == -1 && f == -1 && r == -1)break;
    if(m+f >= 80){
      printf("A\n");
    }
    if(80 > m+f && m+f>= 65){
      printf("B/n");
    }
    if(65 > m+f && m+f >= 50){
      printf("C\n");
    }
    if(50 > m+f && m+f >= 35){
      printf("D\n");
    }
    if(50 > m+f && m+f >= 35 && r >= 50){
      printf("C\n");
    }
    else{
      printf("F\n");
    }
  }
  return 0;
}
