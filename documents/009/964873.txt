#include <stdio.h>
#include <math.h>

int main(void) {
	int n,i;
	scanf("%d", &n);
	
	int a[n];

	for(i=0; i<n; i++) {
		int j, check;
		double circ[6];
		double dist, center;
		
		for(j=0; j<6; j++)
			scanf("%lf", &circ[j]);
		
		center = circ[2] + circ[5];
		dist = sqrt( pow(circ[0]-circ[3], 2.0) + pow(circ[1]-circ[4], 2.0) );
		
		if(center<dist) check = 0;
		else if(fmax(circ[2],circ[5])-dist<=0) check = 1;
		else{
			if(circ[2] > circ[5]) check = 2;
			else check = -2;
		}
		
		a[i] = check;
	}
			
	for(i=0; i<n; i++)
		printf("%d\n", a[i]);
			
	return 0;
}
