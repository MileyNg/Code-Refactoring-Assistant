#include <stdio.h>
#include <string.h>

int main(){
    
    char ch;
    int num=0,i=0,counter[26]={0};
        
    while (scanf("%c",&ch) != EOF) {
        if (ch >=97) {
            num = ch - 'a';
        }else {
            num = ch - 'A';
        }
        counter[num]++;
    }
    
    char a = 'a';
    for (i=0; i <26; i++) {
        printf("%c : %d\n",a,counter[i]);
        a++;
    }
    return 0;
}
