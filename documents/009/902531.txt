int get_weight(int x){
  int res=0;
  for(int i=x;i<n;i++)res+=w[i];
  return res;
}
 
void swap_data(int x){
   
  tmp=w[x];
  w[x]=w[x+1];
  w[x+1]=tmp;
 
  tmp=s[x];
  s[x]=s[x+1];
  s[x+1]=tmp;
 
  temp=f[x];
  f[x]=f[x+1];
  f[x+1]=temp;
}
 
void sort_data(){
  for(int i=n-1;i>0;i--){
    for(int j=0;j<i;j++){
      if(s[j]<s[j+1])swap_data(j);
      else if(s[j]==s[j+1]&&w[j]<w[j+1])swap_data(j);
    }
  }
}
