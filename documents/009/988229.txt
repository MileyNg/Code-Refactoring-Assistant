#include <stdio.h>
#include <math.h>

int main(){

	double x[2] = {}, y[2] = {}, a = 0, b = 0;
	scanf("%lf %lf %lf %lf",&x[0],&y[0],&x[1],&y[1]);

	if (x[0] < 0){ x[0] = x[0] * -1; }
	if (x[1] < 0){ x[1] = x[1] * -1; }
	if (y[0] < 0){ y[0] = y[0] * -1; }
	if (y[1] < 0){ y[1] = y[1] * -1; }

	
	if (x[0] == 0 && x[1] == 0){ a = 0; }
	else if (x[0] < x[1]){
		a = x[1] - x[0]; 
	}
	else{ a = x[0] - x[1];  }


	if (y[0] == 0 && y[1] == 0){ b = 0; }
	else if (y[0] < y[1]){
		b = y[1] - y[0];
	}
	else{ a = y[0] - y[1]; }

	printf("%.8f\n", sqrt(a*a + b*b));
	return 0;
}
