#include <stdio.h>

unsigned int dp[2][2001][3001];

int main() {
    while (1) {
        int i;
        int n, m, s;
        unsigned int sum;
    
        scanf("%d %d %d", &n, &m, &s);
    
        if (n == 0 && m == 0 && s ==0 ) return 0;

        for (i=0; i<2; i++) {
            int j;
    
            for (j=1; j<=m; j++) {
                int k;
    
                for (k=1; k<=s; k++) {
                    if (i == 0 && j == k) {
                        dp[i][j][k] = 1;
                    } else {
                        dp[i][j][k] = 0;
                    }
                }
            }
        }
    
        for (i=1; i<n*n; i++) {
            int j;
    
            for (j=1; j<=m; j++) {
                int k;
    
                for (k=1; k<=s; k++) {
                    int t;
    
                    for (t=1; t*(i+1)<k; t++) {
                        if (j > t) {
                            dp[1][j][k] += dp[0][j - t][k - t*(i+1)];
                            dp[1][j][k] %= 100000;
                        } else {
                            break;
                        }
                    }
                }
            }
    
            for (j=1; j<=m; j++) {
                int k;
    
                for (k=1; k<=s; k++) {
                    dp[0][j][k] = dp[1][j][k];
                    dp[1][j][k] = 0;
                }
            }
        }
    
        sum = 0;
        for (i=0; i<=m; i++) {
            sum += dp[0][i][s];
            sum %= 100000;
        }
    
        printf("%u\n", sum);
    }
}
