#include<stdio.h>
int main(){
	int i,a,b;
	int c[12]={31,29,31,30,31,30,31,31,30,31,30,31};
	int count=0,judge;
	while(1){
		scanf("%d",&a);
		scanf("%d",&b);
		if(a==0||b==0)
			break;
		for(i=0;i<a-1;i++)
			count=count+c[i];
		count= count+b;
		judge=count%7;
		switch(judge){
			case 0:
				printf("Wednesday\n");
				break;
			case 1:
				printf("Thursday\n");
				break;
			case 2:
				printf("Friday\n");
				break;
			case 3:
				printf("Saturday\n");
				break;
			case 4:
				printf("Sunday\n");
				break;
			case 5:
				printf("Monday\n");
				break;
			case 6:
				printf("Tuesday\n");
				break;
		}
	}
	return 0;
}
