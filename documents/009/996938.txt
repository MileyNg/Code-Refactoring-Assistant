int res[64], N, c[64], used[64];

int dfs(int n, int sum){
  if(n < 0){
    memcpy(res, used, sizeof(used));
    return 1;
  }

  int res = 0;

  rep(i, N){
    if(!c[i]) continue;
    if(sum - c[i] == 0 ||  (sum-c[i])%c[i] == 0){
      used[n] = c[i];
      c[i] = 0;
      res |= dfs(n-1, sum-used[n]);
      if(res) return res;
      c[i] = used[n];
      used[n] = 0;
    }
  }

  return res;
}

int main(){
  while(scanf("%d", &N) && N){
    rep(i, N) scanf("%d", c+i);
    int sum = 0;
    rep(i, N) sum += c[i];
    if(dfs(N-1, sum)){
      rep(i, N) printf(i?" %d":"%d", res[i]); puts("");
    }else{
      puts("NO");
    }
  }
  return 0;
}
