#include<stdio.h>

#define N 20
#define Q 200

int A[N],m[Q];
int n,q,i;
int flag;

void solve(int i,int x){
   
  if(x == 0) flag = 1;
  else if(i != n){
    solve(i+1, x-A[i]);
    solve(i+1,x);
  }
}

int main(){
  scanf("%d",&n);
  for(i=0; i<n; i++) scanf("%d",&A[i]);
  scanf("%d",&q);
  for(i=0; i<q; i++) scanf("%d",&m[i]);
  for(i=0; i<q; i++){
    flag = 0;
    solve(0, m[i]);
    if(flag==1) printf("yes\n");
    else printf("no\n");
  }
  return 0;
}
