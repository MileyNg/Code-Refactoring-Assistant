#include <stdio.h>

void swapArr(int a[],int i ,int j){
    int tmp;
    tmp=a[i];
    a[i]=a[j];
    a[j]=tmp;
}

void bsort(int a[],int n){
    int i,j;
    for(i=0;i<n;i++){
        for(j=n-1;j>i;j--){
            if( a[j-1] > a[j] ){
                swapArr(a,j-1,j);
            }
        }
    }
}

int main()
{
    int n,i;
    int a[1000];
    
    scanf("%d",&n);
    for (i=0; i<n; i++) {
        scanf("%d",&a[i]);
    }
    bsort(a, n);
    
    for (i=0; i<n; i++) {
        if (i)printf(" ");
        printf("%d",a[i]);
    }
    printf("\n");
    return 0;
}
