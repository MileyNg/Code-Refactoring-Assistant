#include <stdio.h>
int main(void){
  int i,bt[4]={0},t;
  char l[5],z;
  while(scanf("%d %c %s",&t,&z,l)!=EOF){
    switch(l[0]-'A'){
    case 0:
      if(l[1]=='B')bt[2]++;
      else bt[0]++;
      break;
    case 1:bt[1]++;break;
    case 14:bt[3]++;break;
    }
  }
  for(i=0;i<3;i++)
    printf("%d\n",bt[i]);
  printf("%d\n",bt[3]);
  return 0;
}
