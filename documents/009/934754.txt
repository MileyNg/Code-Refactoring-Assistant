#include<stdio.h>

int main(void){
    int n,a[5000],hoge,i,j,MAX=0,wa;
    while(scanf("%d",&n),n){
        for(i=0;i<n;i++){
            scanf("%d",&a[i]);
        }
        MAX=0;
        for(i=0;i<n;i++){
            wa=a[i];
            for(j=i+1;j<n;j++){
                wa+=a[j];
            }
            if(MAX<wa){
                MAX=wa;
            }
        }
        printf("%d\n",MAX);
    }
    return 0;
}
