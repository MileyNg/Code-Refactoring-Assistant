#include <stdio.h>
#include <stdlib.h>

int main(void) {
	int i, j, n, *a, max = 0;

	scanf("%d", &n);
	a = (int *)malloc(sizeof(int) * n);

	for (i = 0; i < n; i++) {
		scanf("%d", &a[i]);
	}

	for (j = n - 1; j > 1; j--) {
		for (i = j - 1; i >= 0; i--) {
			if (max < (a[j] - a[i])) {
				max = a[j] - a[i];
			}
		}
	}

	printf("%d", max);

	return 0;
}
