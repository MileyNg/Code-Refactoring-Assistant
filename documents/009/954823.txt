#include<stdio.h>
#include<string.h>

int search(char []);

int main(void){
	int i,j,flag;

	char string[90];

	fgets(string,80,stdin);

	for(i = 0 ; i < 25; i ++){
		for(j = 0 ;string[j] != '.' ; j ++){
			if(string[j] != ' '){
				if(string[j] == 'z'){
					string[j] -= 26;
				}
				string[j] += 1;
			}
		}
		flag = search(string);
		if(flag == 1){
			printf("%s\n",string);
			break;
		}
	}

	return  0;
}
