#include <iostream>
#include <vector>
#include <queue>
using namespace std;
const int INF = 1 << 29;
int dx[4] = {0,1,0,-1};
int dy[4] = {1,0,-1,0};
int w,h;
int gx,gy;
int ans = INF;
void print(vector<vector<int> > map){
  for(int i = 0;i < map.size();i++){
    for(int j =0;j < map[0].size();j++){
      if(map[i][j] == INF)cout << ";" << " ";
      else if(map[i][j] == -1)cout << "%" << " ";
      else cout << map[i][j] << " ";
    }
    cout << endl;
  }
}
int dfs(vector<vector<int> > &map,pair<int,int> c,int cnt){
  int x,y;
  if(cnt >= 10)return -1;
  for(int i = 0;i < 4;i++){
    x = c.first;y = c.second;
    while(1){
      if(x+dx[i] < 0 || x+dx[i] >= w || y+dy[i] < 0 || y+dy[i] >= h )break;
      if(x+dx[i] == gx && y+dy[i] == gy){
	ans = min(ans,cnt+1);
	break;
      }
      if(map[x+dx[i]][y+dy[i]] == 1){
	if(x == c.first && y == c.second)break;
	/*vector<vector<int> > dp(map);
	dp[x+dx[i]][y+dy[i]] = 0;
	pair<int,int> tmp(x,y);
	dfs(dp,tmp,cnt+1);*/
	pair<int,int> tmp(x,y);
	map[x+dx[i]][y+dy[i]] = 0;
	dfs(map,tmp,cnt+1);
	map[x+dx[i]][y+dy[i]] = 1;
	break;
      }
      x += dx[i];y += dy[i];
    }
  }
  return 0;
}

int main(){

  while(cin >> w >> h,w || h){
    vector<vector<int> > map(w,vector<int>(h,0));
    pair<int,int> s;
    pair<int,int> g;
    for(int i = 0;i < w;i++){
      for(int j = 0;j < h;j++){
	cin >> map[i][j];
	if(map[i][j] == 2){
	  s.first = i;
	  s.second = j;
	  map[i][j]  = 0;
	}else if(map[i][j] == 3){
	  gx = i;
	  gy = j;
	  map[i][j]  = 0;
	}
      }
    }
    ans = INF;
    dfs(map,s,0);
    if(ans > 10)cout << "-1" << endl;
    else cout << ans << endl;
  }
  return 0;
 
}
