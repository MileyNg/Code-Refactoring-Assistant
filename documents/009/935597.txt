#include<stdio.h>
#include<string.h>
#define LEN 100005

typedef struct pp{
  char name[100];
  int t;
}P;

P Q[LEN+1];
int head, tail, n;

void enqueue(P);
P dequeue(void);


int main(){
  int elaps = 0;
  int i, q;
  P u;
  scanf("%d %d", &n, &q);
  
  for ( i = 1; i <= n; i++){
    scanf("%s", Q[i].name);
    scanf("%d", &Q[i].t);
  }
  head = 1;
  tail = n+1;
  
  while(head != tail){
    if(Q[head].t<=q){
      elaps += Q[head].t;
      u=dequeue();
      printf("%s %d\n",u.name,elaps);
    }else{
      elaps += q;
      Q[head].t -= q;
      enqueue(Q[head]);
      dequeue();
    }
  }  

 
  return 0;
  
}

void enqueue(P x){ 
  Q[tail]=x;
  if(tail==LEN){
    tail = 1;
  }else{
    tail++;
  }
}

P dequeue(){
  P x;
  x = Q[head];
  if(head == LEN){
    head = 1;
  }else{
    head++;
  }
  return x; 
}
