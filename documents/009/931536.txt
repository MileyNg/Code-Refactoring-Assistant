#include<stdio.h>

int main(){
int a,b ;
char op ;
int rst ;

scnaf("%d %c %d",&a,&op,&b) ;

if( op != ( '+' || '-'|| '*' || '/' ){
printf("error\n") ;
}

else{
    switch(op){
    Case '+' :
 	rst = a+b ;
	print("%d",rst) ;
	break ;


    Case '-' :
 	rst = a-b ;
	print("%d",rst) ;
	break ;
	

    Case '*' :
 	rst = a*b ;
	print("%d",rst) ;
	break ;


    Case '/' :
	if(b != 0){
 	rst = a/b ;
	print("%d",rst) ;
	break ;}

	else{
	printf("error\n") ;
	}

    Case '?' :
	break ;
}

return 0 ;

}
