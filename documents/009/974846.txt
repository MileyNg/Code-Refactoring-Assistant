#include <iostream>
#include <string>
#include <vector>
#include <algorithm>

using namespace std;

struct P {
	string a;
	int b;
};

int main() {
	int n;
	cin >> n;
	vector<P> p;
	string name;
	int number;
	cin >> name >> number; 
	P ab;
	ab.a = name;
	ab.b = number;
	p.push_back(ab);
	for (int i = 1; i < n; i++) {
		cin >> name >> number;
		bool hantei = true;
		for (int j = 0; j < p.size(); j++) {
			if (p[j].a == name) {
				p[j].b += number;
				hantei = false;
			}
		}
		if (hantei) {
			ab.a = name;
			ab.b = number;
			p.push_back(ab);
		}
	}
	for (int i = p.size()-1; i > 0; i--) {
		for (int j = 0; j < i; j++) {
			if ((p[j].a).size() > (p[j+1].a).size()) {
				swap(p[j], p[j+1]);
			} else if ((p[j].a).size() == (p[j+1].a).size()) {
				for (int k = 0; k < (p[j].a).size(); k++) {
					if ((p[j].a)[k] > (p[j+1].a)[k]) {
						swap(p[j], p[j+1]);
						break;
					}
				}
			}
		}
	}
	for (int i = 0; i < p.size(); i++) {
		cout << p[i].a << " " << p[i].b << endl;
	}
	return 0;
}
