#include<iostream>

using namespace std;

int main()
{
    int n;

    cin >> n;

    int house[5][4][11];

    for(int i = 1; i < 5; i++)
        for(int j = 1; j < 4; j++)
            for(int k = 1; k < 11; k++)
                house[i][j][k] = 0;

    int b, f, r, v;

    for(int l = 0; l < n; l++)
    {
        cin >> b >> f >> r >> v;
        house[b][f][r] += v;
    }

    for(int m = 1; m < 5; m++)
    {
        for(int n = 1; n < 4; n++)
        {
            for(int o = 1; o < 11; o++)
            {
                cout << " " << house[m][n][o];
            }
            cout << endl;
        }
        if(!(m == 4))
            cout << "####################" << endl;
    }

    return 0;
}
