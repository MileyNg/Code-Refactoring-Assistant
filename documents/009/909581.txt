#include<stdio.h>
#include<string.h>
#include<stdlib.h>


int main(void)
{
  int n,i,j,k;
  long tmp,max;
  long *a;
  while(1){
    scanf("%d",&n);
    if(n==0)
      break;
    a=malloc(sizeof(long)*n);
    for(i=0;i<n;i++)
      scanf("%ld",&a[i]);
    // for(i=0;i<n;i++)
    //printf("%ld\n",a[i]);
    max=-9999;
    tmp=0;
    for(i=0;i<n;i++){
      for(j=i;j<n;j++){  
	  tmp+=a[i];
	  
	  // printf("tmp=%ld\n",tmp);
	  if(tmp>max)
	    max=tmp;
      }
      tmp=0;
    } 
    
    printf("%ld\n",max);
    free(a);
  }

  return 0;
}
