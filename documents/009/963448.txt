#include<iostream>
#include<cctype>
using namespace std;

int main(){
        char c;
        int a[26];
        for(int i=0;i<26;i++){
                a[i]=0;
        }
        while(1){
                scanf("%c", &c);
                if(c=='\n'){
                        break;
                }else if(c!='.' and c!=' '){
                        if(islower(c)){
                            a[(int)c-97]++;
                        } else{
                            a[(int)c-65]++;
                        }
                }
        }
        for(int i=0;i<26;i++){
                printf("%c : %d\n",(char)(i+97),a[i]);
        }
        return 0;
}
