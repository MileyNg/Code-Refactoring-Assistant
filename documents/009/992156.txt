#include<iostream>
#include<algorithm>
using namespace std;
int main(){
	int n,m;
	int count=0;
	cin>>n;
	int a[n];
	for(int i=0;i<n;i++)cin>>a[i];
	
	cin>>m;
	int b[m];
	for(int i=0;i<m;i++)cin>>b[i];
	
	for(int i=0;i<n;i++){
		for(int j=0;j<m;j++){
			if(a[i]==b[j])count++;
		}
	}
	cout<<count<<endl;
	
}
