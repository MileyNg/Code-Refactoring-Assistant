#include <stdio.h>

int main( void ){
    int nCount = 0;
    int n;
    while( true ){
    scanf( "%d",&n );
    if( n == NULL ) break;

    for( int i = 0;i < 10;i++ ){
        for( int j = 0;j < 10;j++ ){
            for( int k = 0;k < 10;k++ ){
                for( int l = 0;l < 10;l++ ){
                    if( i+j+k+l == n ) nCount++;
                }
            }
       }
    }

    printf( "%d\n",nCount );
    }
    return 0;
}
