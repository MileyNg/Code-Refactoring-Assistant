#include<iostream>
using namespace std;
int main(){
  
  int Q,c,a,n,counter;
  
  cin >>Q;
  
  for(int i=0 ; i < Q ; i++){
    
    cin >>c>>a>>n;
    counter=0;
    
    while(1){
      
      if( (c <= 2 && a == 0) || c == 0 || (c <= 1 && a <= 1 && n== 0 ))break;
      
      if( c >= 1 && a >= 1 && n >=1){
	counter++;
	c--;
	a--;
	n--;
      }else if(c >= 2 && a >= 1){
	counter++;
	c-=2;
	a--;
      }else if(c >= 3){
	counter++;
	c-=3;
      }
      
    }
    
    cout <<counter<<endl;
    
  }  

  return 0;
}
