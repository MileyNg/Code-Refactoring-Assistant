#include<iostream>
#include<string>
using namespace std;

int main(){
	string line;
	cin>>line;
	for(int i=0;i<line.size();i+=2){
		switch(line[i]-'0'){
			case 1:break;
			case 2:cout<<'k';break;
			case 3:cout<<'s';break;
			case 4:cout<<'t';break;
			case 5:cout<<'n';break;
			case 6:cout<<'h';break;
			case 7:cout<<'m';break;
			case 8:cout<<'y';break;
			case 9:cout<<'r';break;
			case 0:if(line[i+1]=='U')cout<<'n';else cout<<'w';break;
		}
		if(line[i+1]=='U')
			if(line[i]=='0')cout<<'n';
			else cout<<'u';
		else if(line[i+1]=='T')cout<<'a';
		else if(line[i+1]=='L')cout<<'i';
		else if(line[i+1]=='R')cout<<'e';
		else if(line[i+1]=='D')cout<<'o';
		
	
	}
	cout<<endl;
}
