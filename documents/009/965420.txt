#include <iostream>
using namespace std;

int N, T, H, L;
int coins[101][2];

int solve() {
    int cur = 0;
    while (1) {
//        cout << coins[cur][0] << endl;
        if (coins[cur][0]) {
            T++; coins[cur][0]--;
            if (T > L) break;
            int y = H * 10 + T - 9;
            if (y > 0) {
                T -= y % 10; coins[cur][0] += y % 10;
                H -= y / 10; coins[cur][1] += y / 10;
                if (T < 0 || H < 0) break;
            }
        }
        else if (coins[cur][1]) {
            H++; coins[cur][1]--;
            int y = H * 10 + T - 9;
            T -= y % 10; coins[cur][0] += y % 10;
            H -= y / 10; coins[cur][1] += y / 10;
            if (T < 0 || H < 0) break;
        }
        else {
            break;
        }
        cur = (cur + 1) % N;
    }
    return cur + 1;
}

int main() {
    while (cin >> N >> T >> H >> L) {
        for (int i = 0; i < N; i++) {
            cin >> coins[i][0] >> coins[i][1];
        }
        cout << solve() << endl;
    }
    return 0;
}
