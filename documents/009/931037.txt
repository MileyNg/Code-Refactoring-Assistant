#include <stdio.h>

#define N 100000
#define Q 50000

int main(){
  int S[N],T[Q],ans[Q];
  int n,m,i,j,k=0,count=0,flag=0;
  scanf("%d",&n);
  for(i=0; i<n ; i++){
    scanf("%d",&S[i]);
  }
  
  scanf("%d",&m);
  for(i=0; i<m ; i++){
    scanf("%d",&T[i]);
    flag=0;
    for(j=0; j<n ; j++){
      if(flag!=0 && flag!=1)
	continue;
      if(S[j]==T[i]){
	ans[k]=T[i];
	k++;
	count++;
	break;
      }
    }
  }
  for(i=0; i<k ; i++){
    printf("%d",ans[i]);
  }
  if(count==0) printf("0\n");
  else printf("%d\n",count);
  return 0;
}
