#include<stdio.h>
#define N 100

int main(){

  int len=0,a[len],i,j,key=0;

  scanf("%d",&len);
  if(len>=1 && len<=N){
    for(i=0;i<len;i++){  
      scanf("%d",&a[i]);
    }
  }
  for(i=1;i<len;i++){
    key=a[i];
    j=i-1;
    while(j>=0){
      if(a[j]>key){
	a[j+1]=a[j];
	j--;
      } else break;
    }
    a[j+1]=key;
    for(j=0;j<len;j++){
      printf("%d ",a[j]);
    }
    printf("\n"); 
  } 
  return 0;
}
