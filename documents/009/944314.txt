/** Problem0081 : A Symmetric Point **/
int main()
{
	double x1, y1, x2, y2, xq, yq; char c;
	while (cin >> x1 >> c >> y1 >> c >> x2 >> c >> y2 >> c >> xq >> c >> yq) {
		if (x1==x2)
			printf("%.6f %.6f\n", 2*x1-xq , yq);
		else if (y1==y2)
			printf("%.6f %.6f\n", xq , 2*y1-yq);
		else {
			double a = (y2-y1)/(x2-x1);
			double aa = -1/a;
			double b = y1 - a*x1;
			double bb = yq - aa*xq;
			double xm = (bb - b)/(a - aa);
			printf("%.6f %.6f\n", 2*xm-xq , aa*(2*xm-xq)+bb);
		}
	}
	return 0;
}
