#include<stdio.h>
#include<string.h>
#include<stdlib.h>


int main(void)
{
  int n,i,j,k;
  int tmp,max;
  int *a;
  while(1){
    scanf("%d",&n);
    if(n==0)
      break;
    a=malloc(sizeof(int)*n);
    for(i=0;i<n;i++)
      scanf("%d",&a[i]);
    //for(i=0;i<n;i++)
      //  printf("a[%d]=%ld\n",i,a[i]);
    max=a[0];
    tmp=0;
    for(i=0;i<n;i++){
      for(j=i;j<n;j++){  
	  tmp+=a[j];
  
	  //printf("tmp=%ld\n",tmp);
	  if(tmp>max)
	    max=tmp;
      }
      tmp=0;
    } 
   
    printf("\nmax:%d\n",max);
    free(a);
  }

  return 0;
}
