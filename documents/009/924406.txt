#include <stdio.h>

int main(void){
	
	int m, f, r, sum;
	
	
	while(1){
		
		scanf("%d %d %d", &m, &f, &r);
		
		if(m == -1 && f == -1 && r == -1){
			break;
		}
		
		sum = m + f;
 		
		if(m == -1 || f == -1){
			putchar('F');
            puts("");
		}
		else if(sum >= 80){
			putchar('A');
			puts("");
		}
		else if(65 <= sum && sum < 80){
			putchar('B');
			puts("");
		}
		else if(50 <= sum && sum < 65 || r >= 50){
			putchar('C');
			puts("");
		}
		else if(30 <= sum && sum < 50){
			putchar('D');
			puts("");
		}
        else{
            putchar('F');
            puts("");
 
        }
		
		
	}
	
	return 0;
    
}
