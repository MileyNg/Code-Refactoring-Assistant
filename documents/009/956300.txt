#include <iostream>
#include <vector>

using namespace std;
typedef vector<int> vi;
#define loop(i,a,b) for(int i=(a); i<int(b); i++)
#define rep(i,b) loop(i,0,b)

vi riffle(vi const& v){
    int n = v.size();
    vi res(n);
    rep(i,n){
        if(i&1) res[i]=v[i/2+n/2];
        else res[i]=v[i/2];
    }
    return res;
}

vi cut(vi const& v, int k){
    int n = v.size();
    vi res(n);
    rep(i,n) res[i] = v[(i+k)%n];
    return res;
}

int main(){
    int n; cin >> n;
    n*=2;
    vi v(n);
    rep(i,n) v[i] = i+1;
    int m; cin>>m;
    rep(i,m){
        int op; cin>>op;
        if(op) v = cut(v,op);
        else v = riffle(v);
    }
    rep(i,n) cout << v[i] << "\n";
}
