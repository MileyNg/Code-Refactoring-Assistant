int main(){
  int a=0,b=0;
  
  scanf("%d%d",&a,&b);
  
  printf("%d\n",gcd(a,b));

  return 0;

}

int gcd(int x, int y){
  int n;
  
  if(x < y){
    n = x;
    x = y;
    y = n;
  }
  
  while(y>0){
    n = x%y;
    x = y;
    y = n;
  }
  return x;
}
