#include <iostream>
#include <algorithm>
#include <map>
#include <set>
#include <cstring>

using namespace std;

#define loop(i,a,b) for(int i=(a); i<int(b); i++)
#define rep(i,b) loop(i,0,b)

int dp[2][5010] = {};

int main(){
    int n, m; cin >> n >> m;

    typedef pair<int, int> P;
    map<P, int> ps;

    rep(i, m){
        int x, y, c; cin >> x >> y >> c;
        x--, y--, c++;
        ps[P(x, y)] = c;
    }

    int ans = 0;

    int* cur = dp[0];
    int* prev = dp[1];
    loop(i, 0, n){
        memset(cur, 0, sizeof(cur));
        rep(j, i + 1){
            int t = ps[P(i, j)];
            if (j - 1 >= 0) t = max(t, prev[j - 1] - 1);
            if (j < i + 1) t = max(t, prev[j] - 1);
            ans += t != 0;
            cur[j] = t;
        }
        swap(cur, prev);
    }

    cout << ans << endl;
}
