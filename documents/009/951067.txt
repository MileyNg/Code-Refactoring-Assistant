#include<iostream>

using namespace std;

int main(){

	int n, m;
	int fi[1001];
	int dice[1001];
	int s;
	int cnt = 1;

	while (1){

		cin >> n >> m;
		if (n == 0 && m == 0) break;

		for (int i = 1; i <= n; i++){
			cin >> fi[i];
		}
		for (int i = 1; i <= m; i++){
			cin >> dice[i];
		}

		s = 1;
		while (1){
			s += dice[cnt];
			if (s >= n){
				cout << cnt << endl;
				break;
			}
			s += fi[s];
			if (s >= n){
				cout << cnt << endl;
				break;
			}
			cnt++;
		}
	}

	return 0;
}
