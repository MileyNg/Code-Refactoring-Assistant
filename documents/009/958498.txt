#include<stdio.h>
int main(){
	double a,b,c,d,e,f;
	double A,B[2][2];
	double x,y;
	while(scanf("%lf%lf%lf%lf%lf%lf",&a,&b,&c,&d,&e,&f)!=EOF){
		A=a*e-b*d;
		B[0][0]=c*(e/A);
		B[0][1]=f*(-b/A);
		B[1][0]=c*(-d/A);
		B[1][1]=f*(a/A);
		x=B[0][0]+B[0][1];
		y=B[1][0];+B[1][1];
		printf("%.3f %.3f\n",x,y);
	}
	return 0;
}
