#include<stdio.h>
#include<math.h>
int main(void){
	double x[4],y[4],a,b,c,cos[2],sin[2],t;
	while(scanf("%lf%lf%lf%lf%lf%lf%lf%lf",&x[0],&y[0],&x[1],&y[1],&x[2],&y[2],&x[3],&y[3])!=EOF){
		a=sqrt((x[0]-x[1])*(x[0]-x[1])+(y[0]-y[1])*(y[0]-y[1]));
		b=sqrt((x[0]-x[3])*(x[0]-x[3])+(y[0]-y[3])*(y[0]-y[3]));
		c=sqrt((x[1]-x[3])*(x[1]-x[3])+(y[1]-y[3])*(y[1]-y[3]));
		cos[0]=(b*b+c*c-a*a)/(2*b*c);
		sin[0]=sqrt(1-cos[0]*cos[0]);
		a=sqrt((x[0]-x[2])*(x[0]-x[2])+(y[0]-y[2])*(y[0]-y[2]));
		c=sqrt((x[2]-x[3])*(x[2]-x[3])+(y[2]-y[3])*(y[2]-y[3]));
		cos[1]=(b*b+c*c-a*a)/(2*b*c);
		sin[1]=sqrt(1-cos[1]*cos[1]);
		t=sin[0]*cos[0]+cos[1]*sin[1];
		if(t>0){
			printf("NO\n");
		}else{
			printf("YES\n");
		}
	}
	return 0;
}
