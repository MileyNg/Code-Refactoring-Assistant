#include<iostream>
#include<cctype>
using namespace std;

//a-zA-z -> テ篠ゥテδ?つサテδキ
int toInt(char a){
  if(islower(a))return a-'a';
  return a-'A'+26;
}

//テ篠ゥテδ?つサテδキ -> a-zA-Z
char toChar(int a){
  if(a<26)return 'a'+a;
  return 'A'+a-26;
}

//テつアテδソテδクテδ?
const int M = 52;

int main(){
  int n,k[100];
  string s;

  while(cin >> n,n){
    for(int i=0;i<n;i++)cin >> k[i];
    cin >> s;

    for(int i=0;i<(int)s.size();i++){
      //テ篠ゥテ「ツつャテつ・テ?津つァテδ堙「ツ?「テ「ツ按妥「ツ?「テッツャツづ「ツ?「テδづ「ツ?「テδ湘つーテつコテ「ツ?「テ「ツ按妥「ツ?「テδ?「ツ?「テδ崚つァテ渉?つァテδ偲つァテδ?つァテ?杜odテつァテδ堙つェテつサテつァテつョテつァテ「ツ?禿?津「ツ?ヲテつァテつァ
      int tmp = toInt(s[i]) - k[i%n];
      //テ「ツ?ヲテδ暗δクテδ凖つァテ「ツ?堙つァテつ「テつァテδ催「ツ按?つソテつァテδ偲δヲテδ湘渉?δ?つァテ?津「ツ按榲つャテつ。テつ・テつァテつ?odテつァテ?津つコテδ凝つァテδ催つ?仰敕つァテ??(x%mod) + mod)%mod
      tmp = ((tmp%M) + M) % M;
      cout << toChar(tmp);
    }
    cout << endl;
  }
}
