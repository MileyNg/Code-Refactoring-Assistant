#include<stdio.h>

int count_areas(int x);

int count_areas(int x){

   int y[100005],i;

   for(i=0;i<x;i++){
      if(i==0)y[0]=2;
      else{
         y[i]=y[i-1]+i+1;
      }
   }
      return(y[x-1]);
}

int main(void){

   int n,a;

   while((scanf("%d",&n))!=EOF){

      a=count_areas(n);
      printf("%d\n",a);
   }
   return 0;
}
