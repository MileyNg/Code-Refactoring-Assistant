int main() {
    int n;
    cin >> n;
    vector<int> ss(n);
    for (auto& s : ss) {
        cin >> s;
    }
    int q;
    cin >> q;
    int ans = 0;
    for (int i = 0; i < q; i++) {
        int t;
        cin >> t;
        if (binary_search(begin(ss), end(ss), t)) {
            ans++;
        }
    }
    cout << ans << endl;
}
