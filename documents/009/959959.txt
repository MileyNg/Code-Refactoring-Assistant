#include <bits/stdc++.h>
#include <unistd.h>

#define loop(i,a,b) for(int i=(a); i<int(b); i++)
#define rep(i,n) loop(i,0,n)

using namespace std;

inline void pon(int n){
    cout << n << endl;
    fflush(stdout);
}

inline int get(){
    int res;
    cin >> res;
    return res;
}

int main(){
    srand(time(0u)+getpid());

    int n; cin>>n;
    vector<string> win(n);
    rep(i,n)cin>>win[i];

    int me = rand()%n;
    rep(i,1000){
        pon(me+1);
        int ai = get()-1;
        if (win[me][ai] == 'x')me = rand()%n;
    }
}
