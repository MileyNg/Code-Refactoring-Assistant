#include <stdio.h>

int main(void){
	int a, b, c, d, e, f, t;
	unsigned double x, y;
	while(scanf("%d %d %d %d %d %d", &a, &b, &c, &d, &e, &f) != EOF){
		t = 1.0*a*e - 1.0*b*d;
		x = (1.0*e*c - 1.0*b*f)/t + 0.0001;
		y = (1.0*a*f - 1.0*d*c)/t + 0.0001;
		printf("%.3f %.3f\n", x, y);
	}
	
	return 0;
}
