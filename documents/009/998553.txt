#include <stdio.h>

int main(void)
{
    int n;
    scanf("%d", &n);
    int x[n];
    for (int i = 0; i < n; ++i) {
        scanf("%d", &x[i]);
    }
    for (int j = n - 1; j >= 0; --j) {
        if (j != n - 1) {
            printf(" ");
        }
        printf("%d", x[j]); 
    }
    return 0;
}
