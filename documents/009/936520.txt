#include <iostream>
#include <limits.h>
using namespace std;

int main(void){
	int n;
	int r;
	int max = -200000000;
	int min;

	cin >> n;

	cin >> r;
	min = r;
	for(int i=1; i<n; i++){
		cin >> r;
		if(r - min > max){
			max = r - min; 
		}
		if(min > r) min = r;
	}

	cout << max << endl;

	return 0;
}
