#include <stdio.h>

int main(void)
{
  int n;

  scanf("%d",&n);

  if(n==0)
    {
      printf("\n");
      return 0;
    }

  int input[n];

  for(int i=0;i<n;i++)
    {
      scanf("%d",&input[i]);
    }

  int min,max;
  int sum=0;

  min=max=input[0];

  for(int i=0;i<n;i++)
    {
      if(input[i]>max)
	{
	  max=input[i];
	}

      if(input[i]<min)
	{
	  min=input[i];
	}

      sum+=input[i];
    }

  printf("%d %d %d\n",min,max,sum);      

  return 0;

}
