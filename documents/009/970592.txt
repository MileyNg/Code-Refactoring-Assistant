char field[100][100];

void move(int h, int w) {
  char c = field[h][w];
  if(c == '.') cout << w << ' ' << h << endl;
  else if(c == '#') cout << "LOOP" << endl;
  else {
    field[h][w] = '#';
    if(c == '<') move(h, w-1);
    else if(c == '>') move(h, w+1);
    else if(c == 'v') move(h+1, w);
    else move(h-1, w);
  }
}
