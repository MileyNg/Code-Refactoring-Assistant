using System;

class Integral {
    static void Main(String[] args) {
        int input;
        int i;
        long sum = 0;

        while(true){
            input = int.Parse(Console.ReadLine());

            for (i = 1; i*input < 600; i++) {
                sum += Funct((long)(i * input)) * input;
            }

            Console.WriteLine("{0}",sum);
            sum = 0;
        }
    }

    static long Funct(long num) {
        return (long)(num * num);
    }
}
