#include<iostream>
#include<cctype>
#include<algorithm>
#include<vector>
using namespace std;

int main(){
        int a[26];
        for(int j=0;j<26;j++){
                a[j]=0;
        }
        char c;
        while(scanf("%c",&c)!=EOF){
                if(c!='\n' || c!='.' || c!=' '){
                    int n=c-'a';
                    a[n]++;
                }
        }
        for(int i=0;i<26;i++){
                printf("%c : %d\n",i+'a',a[i]);
        }
        return 0;
}
