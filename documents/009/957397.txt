#include<stdio.h>
#include<stdlib.h>
#include<math.h>
#include<string.h>
int main(void)
{
  int n;
  while(scanf("%d",&n)!=0){
    int count[10]={0};
    int i;
    int no;

    for(i=0;i<n;i++){
      scanf("%d",&no);
      count[no]++;
    }
    for(i=0;i<10;i++){
      int j;
      for(j=0;j<count[i];j++)
	putchar('*');
      if(count[i]==0)
	putchar('-');
      putchar('\n');
    }
  }


  return 0;
}
