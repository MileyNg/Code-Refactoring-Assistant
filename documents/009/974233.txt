#include<iostream>
#include<iomanip>
#include<cmath>

using namespace std;

int main(){
	double a,b,S,L,h,pi;
	int c;
	pi=3.1415926535;
	cin>>a>>b>>c;
	h=b*sin(c*pi/180);
	S=1.0/2*a*b*sin(c*pi/180);
	L=a+b+sqrt(a*a+b*b-2*a*b*cos(c*pi/180));
	cout<<setprecision(20);
	cout<<S<<endl;
	cout<<L<<endl;
	cout<<h<<endl;
	while(1);
	return 0;
}
