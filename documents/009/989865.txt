#include<iostream>
using namespace std;

int main(){
  int i,j;
  int n,m;
  
  while(1){
    cin >> n >> m;
    if(n==0 && m==0){
      break;
    }
    if(n<3 && m<3){ 
      for(i=0;i<n;i++){
	for(j=0;j<m;j++){
	  cout << "#";
	}
      }
    } 
    if(n>=3 && m>=3){
      for(i=0;i<m;i++){
	cout << "#";
      }
      cout << endl;
      for(i=0;i<n-2;i++){
	cout << "#";
	for(j=0;j<m-2;j++){
	  cout << ".";
	}
	cout << "#";
	cout << endl;
      }
      for(i=0;i<m;i++){
	cout << "#";
      }
      cout << endl;   
    } 
    cout << endl;  
  }
  return 0;
}
