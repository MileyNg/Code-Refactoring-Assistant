//a-zA-z -> テδィテδ。テδョテ「ツ??
int toInt(char a){
  if(islower(a))return a-'a';
  return a-'A'+26;
}

//テδィテδ。テδョテ「ツ??-> a-zA-Z
char toChar(int a){
  if(a<26)return 'a'+a;
  return 'A'+a-26;
}
