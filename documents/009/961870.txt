#include <stdio.h>

int main(void)
{
	int m, t, r;
	
	while(1){
		scanf("%d %d %d", &m, &t, &r);
		
		if(m == -1 && t == -1 && r == -1){
			break;
		}
		
		else {
			if (m == -1){
				m++;
			}
			else if (t == -1){
				t++;
			}
			if (m == -1 && t == -1 || m+t < 30){
				printf("F\n");
			}
			
			else if (m + t >= 80){
				printf("A\n");
			}
			else if (m + t >= 65){
				printf("B\n");
			}
			else if (m + t >= 50 || r >= 50){
				printf("C\n");
			}
			else {
				printf("D\n");
			}
		}
	}
	
	return (0);
}
