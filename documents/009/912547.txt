#include <iostream>
using namespace std;

int main() {
int tate,yoko;
while(cin>>tate>>yoko){
if(tate==0&&yoko==0){
	break;
}	
for(int l=1;l<=yoko;l++){
		cout<<"#";
	}
	cout<<endl;
for(int k=1;k<=tate-2;k++){
	cout<<"#";
	for(int l=1;l<=yoko-2;l++){
		cout<<".";
	}
	cout<<"#"<<endl;
}
for(int l=1;l<=yoko;l++){
		cout<<"#";
	}
	cout<<endl;
cout<<endl;
}
	return 0;
}
