#include<cstdio>
#include<iostream>
using namespace std;

int main(){
  
  int k,n,x,y,lim;
  cin>>k;
  cin>>n;
  if(k%2==0)
    lim=k/2;
  else
    lim=k/2+1;
  for(int i=0;i<n;i++){
    cin>>x>>y;
    if(x>lim && y>lim){
      x=k+1-x;
      y=k+1-y;
    }
    else if(x>lim && y<lim) x=k+1-x;
    else if(x<lim && y>lim) y=k+1-x;
    if(x<y) printf("%d\n",(x+2)%3+1);
    else printf("%d\n",(y+2)%3+1);
  }
  return 0;
}
