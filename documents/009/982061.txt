#include <iostream>
using namespace std;

int main()
{
	int n;
	cin >> n;
	for (int i = 0; i < n; i++){
		double xa, ya, ra, xb, yb, rb;
		cin >> xa >> ya >> ra >> xb >> yb >> rb;
		double d = (xb - xa) * (xb - xa) + (yb - ya) * (yb - ya);
		if (d <= (rb - ra)*(rb - ra)){
			cout << -2 << endl;
		}
		else if (d <= (ra - rb)*(ra - rb)){
			cout << 2 << endl;
		}
		else if (d >(ra + rb)*(ra + rb)){
			cout << 0 << endl;
		}
		else{
			cout << 1 << endl;
		}
	}
	return 0;
}
