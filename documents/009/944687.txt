#define M 1000000/* your task*/

char H[M][L]; /* Hash Table */

int getChar(char ch){
  if ( ch == 'A') return 1;
  else if ( ch == 'C') return 2;
  else if ( ch == 'G') return 3;
  else if ( ch == 'T') return 4;
  return 0;
}

/* convert a string into an integer value */
long long getKey(char str[]){
  long long sum = 0, p = 1, i;
  for ( i = 0; i < strlen(str); i++ ){
    sum += p*(getChar(str[i]));
    p *= 5;
  }
  return sum;
}

int find(char str[]){

  /* your task */
  int i,j,count=0;

  for(i=0; H[i][0]!='\0'; i++){
    count = 0;
    for(j=0; H[i][j]!='\0'; j++){
      if(str[j] != H[i][j]){
	break;
      }
      count++;
    }
    if(count == strlen(str)) return 1;
  }
  return 0;
}

int insert(char str[]){

  /* your task */
  static int i=0;
  strcpy(H[i],str);
  i++;
  return 1;
}
