#include <stdio.h>
#include <math.h>

int main(void)
{
	int n, i;
	int ans;
	float xa, ya, ra,
		  xb, yb, rb;
	float x, y, len;
	
	scanf("%d", &n);
	
	for (i = 0; i < n; i++){
		scanf("%f %f %f %f %f %f", &xa, &ya, &ra, &xb, &yb, &rb);
		
		x = xb - xa;
		y = yb - ya;
		
		len = sqrt((x * x) + (y * y));
		if (len > ra + rb){
			ans = 0;
		}
		else if (len == ra + rb){
			ans = 1;
		}
		else if (len < ra + rb){
			if (ra > rb){
				ans = 2;
			}
			else {
				ans = -2;
			}
		}
		printf("%d\n", ans);
	}
	
	return (0);
}
