#include<iostream>
#include<cctype>
using namespace std;

//a-zA-z -> テゥツ??ァツ閉ェ
int toInt(char a){
  if(islower(a))return a-'a';
  return a-'A'+26;
}

//テゥツ??ァツ閉ェ -> a-zA-Z
char toChar(int a){
  if(a<26)return 'a'+a;
  return 'A'+a-26;
}

//テゥツァツ?ヲツ閉ー
const int M = 52;

int main(){
  int n,k[100];
  string s;

  while(cin >> n,n){
    for(int i=0;i<n;i++)cin >> k[i];
    cin >> s;

    for(int i=0;i<(int)s.size();i++){
      //テ・ツセツェテァツ陳ーテ」ツつ津」ツつキテ」ツδ淌」ツδ・テ」ツδャテ」ツδシテ」ツつキテ」ツδァテ」ツδウテ」ツ?凖」ツつ凝」ツ?ォテ」ツ?ッmodテ」ツつ津、ツスツソテ」ツ?暗」ツ?ーティツ可ッテ」ツ??
      int tmp = toInt(s[i]) - k[i%n];
      //ティツイツ?ヲツ閉ーテ」ツつづ」ツ?づ」ツつ甘・ツセツ療」ツつ凝・ツ?エテ・ツ青暗」ツ?ョテ・ツョツ嘉・ツ?ィテ」ツ?ェmodテ」ツ?ョテ・ツ渉姪」ツつ甘ヲツ鳴ケテ」ツ?ッ((x%mod) + mod)%mod
      tmp = ((tmp%M) + M) % M;
      cout << toChar(tmp);
    }
    cout << endl;
  }
}
