typedef complex<double> P;
const double EPS = 1e-9;

//テ・ツ、ツ姪ァツゥツ?
double cross(P x,P y){return real(x)*imag(y) - imag(x)*real(y);}

//テ、ツクツ嘉ィツァツ津・ツスツ「テ」ツ?ョテゥツ敖「テァツゥツ?
double area(P a,P b,P c){ return abs(cross(b-a,c-a)/2); }

//テ、ツクツ嘉ィツァツ津・ツスツ「abcテ」ツ?ォテァツつケxテ」ツ?古・ツ青ォテ」ツ?セテ」ツつ古」ツ?ヲテ」ツ??」ツつ凝」ツ?凝」ツ?ゥテ」ツ??」ツ?凝」ツつ津・ツ按、テ・ツョツ?
bool inter(P a,P b,P c,P x){
  //テヲツオツョテ・ツ仰陛・ツーツ湘ヲツ閉ーテァツつケテヲツシツ氾ァツョツ療」ツ?ァテ」ツ?ッティツェツ、テ・ツキツョテ・ツ?ヲテァツ青?」ツつ津ィツ。ツ古」ツ??
  //テ」ツ?警 > y +EPSテ」ツ?催」ツ?ッテゥツ?堙・ツクツクテ」ツ?ョティツィツ暗ァツョツ療」ツ?ォテ」ツ?甘」ツ?妥」ツつ凝」ツ?警>yテ」ツ?催」ツ?ォテ・ツスツ禿」ツ?淌」ツつ?
  //テ」ツ?警 > y -EPSテ」ツ?催」ツ?づ」ツつ凝」ツ??」ツ?ッテ」ツ?警 +EPS > yテ」ツ?催」ツ?ッテゥツ?堙・ツクツクテ」ツ?ョティツィツ暗ァツョツ療」ツ?ォテ」ツ?甘」ツ?妥」ツつ凝」ツ?警>=yテ」ツ?催」ツ?ォテ・ツスツ禿」ツ?淌」ツつ?
  if( area(a,b,x) + area(b,c,x) + area(c,a,x) > area(a,b,c) + EPS)return false;

}
