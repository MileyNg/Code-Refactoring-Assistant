#include <iostream>
#include <string>
#include <algorithm>
using namespace std;

int main(){
   string azunyan = "AIDUNYAN";
   string rev = azunyan;
   sort(rev.begin(), rev.end());

   string d;
   cin >> d;

    for(int i=0; i < d.size(); i++){
        string dd = d.substr(i, azunyan.size());
        sort(dd.begin(), dd.end());
        if(rev == dd){
            d = d.substr(0,i) + azunyan + d.substr(i+azunyan.size());
            i += azunyan.size();
        }
    }
    cout << d << endl;
}
