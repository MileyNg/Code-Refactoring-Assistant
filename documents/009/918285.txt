#include<stdio.h>
#define SWITCH 1

main()
{
	int n = 0, x = 0;
	int counter = 0, i = 0, j = 0, k = 0;
	
	scanf("%d %d", &n, &x);
	
	while((n != 0) || (x != 0))
	{
		for(i = 1; i <= n; i++)
		{
			for(j = i + 1; j <= n; j++)
			{
				for(k = i + 2; k <= n; k++)
				{
					if((i == j) || (j == k) || (k == i) || (i >= j) || (j >= k) || (i >= k))
					{
						#if SWITCH == 0
						printf("i = %d j = %d k = %d", i, j, k);
						printf("counter = %d\n", counter); //debug
						#endif
						continue;
					}
					else if((i + j + k) == x)
					{
						counter++;
						#if SWITCH == 0
						printf("i = %d j = %d k = %d", i, j, k);
						printf("counter = %d\n", counter); //debug
						#endif
					}
				}
			}
		}
		printf("%d\n", counter);
		scanf("%d %d", &n, &x);
	}
	return 0;
}
