#define _CRT_SECURE_NO_WARNINGS
#include<stdio.h>

int main(void){
	int a[50];
	int count, x;

	for (count = 0; a[count] != 0; count++){
		scanf("%d", &a[count]);
		if (a[count] == 0)count--;
	}

	for (x = 0; x < count; x++){
		printf("Case %d: %d", x + 1, a[x]);
		if (x + 1 != count){
			printf("\n");
		}
	}
	return 0;
}
