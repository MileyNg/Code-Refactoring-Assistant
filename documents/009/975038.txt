#include<iostream>
#include<stdio.h>
using namespace std;

main() {

  int i, r, n, s[13], h[13], c[13], d[13];
  string f;
  cin >> n;

  for(i = 0; i < n; i++){
    cin >> f >> r;
    if(f == "S"){
      s[r-1] = r; 
    }else if(f == "H"){
      h[r-1] = r;
    }else if(f == "C"){
      c[r-1] = r;
    }else {
      d[r-1] = r;
    }
    
  }

  for(i = 0; i < 13; i++){
    if(i+1 != s[i]){
      printf("S %d\n",i+1);
    }
  }
  for(i = 0; i < 13; i++){
    if(i+1 != h[i]){
      printf("H %d\n",i+1);
    }
  }
  for(i = 0; i < 13; i++){
    if(i+1 != c[i]){
      printf("C %d\n",i+1);
    }
  }
  for(i = 0; i < 13; i++){
    if(i+1 != d[i]){
      printf("D %d\n",i+1);
    }
  }
