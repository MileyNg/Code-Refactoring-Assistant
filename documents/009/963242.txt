#include <stdio.h>

int main(void)
{
	int n;
	int i;
	int max = 0;
	int max_num = 0;
	
	scanf("%d", &n);
	for (i = 1; i <= n; i++){
		int a, v;
		scanf("%d %d", &a, &v);
		
		if (max < v || (max <= v && max_num > a)){
			max = v;
			max_num = a;
		}
	}
	printf("%d %d\n", max_num, max);
	
	return 0;
}
