#include<stdio.h>
#include<string.h>
int main(){

  char in[100]={'\0'};
  int c=0;
  while(scanf("%s",in)!=EOF){
    if(strcmp(in,"apple")==0)
      strcpy(in,"peach");
    else if(strcmp(in,"peach")==0)
      strcpy(in,"apple");
    if(strcmp(in,"apple.")==0)
      strcpy(in,"peach.");
    else if(strcmp(in,"peach.")==0)
      strcpy(in,"apple.");
    if(c==0)
       printf("%s",in);
    else{
      putchar(' ');
      printf("%s",in);
    }
    c++;
  }

}
