#include <stdio.h>

int main(void)
{
	int num[5];
	int i;
	int j;
	
	for( i = 0; i < 5; i++ )
	{
		scanf( "%d", &num[i] );
	}
	
	for( i = 0; i < 5; i++ )
	{
		for( j = i + 1; j < 5; j++ )
		{
			if( num[i] < num[j] )
			{
				int buf = num[i];
				num[i] = num[j];
				num[j] = buf;
			}
		}
	}
	
	for( i = 0; i < 5; i++ )
	{
		printf( "%d%c", num[i], i == 4 ? '\n' : ' ' );
	}
	
	return (0);
}
