#include <stdio.h>

int main(){

	int a[3] = {}, i, g, n;
	for (i = 0; i < 2; i++){ scanf("%d", &a[i]); }
	for (n = 0; n < 2; n++){
		for (i = 0; i < 2; i++){
			if (a[i] > a[i + 1]){
				g = a[i];
				a[i] = a[i + 1];
				a[i + 1] = g;
			}
		}
	}
	for (i = 0; i < 3; i++){ printf("%d ", a[i]); }
	printf("\n");
}
