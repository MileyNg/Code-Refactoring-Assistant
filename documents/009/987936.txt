#include <stdio.h>
#include <string.h>

const char *operation[] = {
    "North",
    "East",
    "South",
    "West",
    "Right",
    "Left"
};

const int param[][3] = {
    {1, 2, -1},
    {0, 2, -1},
    {1, 2, 1},
    {0, 2, 1},
    {0, 1, -1},
    {0, 1, 1}
};


void rotate(int v[], int a1, int a2, int direction) {
    int t = v[a1];
    v[a1] = - direction * v[a2];
    v[a2] = direction * t;
}


int main (void) {
    int n, score, i, j, k;
    char buf[6];

    while (scanf("%d\n", &n) != EOF && n) {
        int m[][3] = {
            {1, 0, 0},
            {0, 1, 0},
            {0, 0, 1}
        };
        for (i = 0, score = 1; i < n; i++) {
            scanf("%s", buf);
            for (j = 0; j < 6; j++) {
                if (strcmp(buf, operation[j]) == 0) break;
            }
            for (k = 0; k < 3; k++) {
                rotate(m[k], param[j][0], param[j][1], param[j][2]);
            }
            if (m[0][2] == 1) score += 3;
            else if (m[0][2] == -1) score += 4;
            else if (m[1][2] == 1) score += 5;
            else if (m[1][2] == -1) score += 2;
            else if (m[2][2] == 1) score += 1;
            else if (m[2][2] == -1) score += 6;
        }
        printf("%d\n", score);
    }
    return 0;
}
