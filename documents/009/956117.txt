#include<stdio.h>
int main(void)
{
  char moji;
  
  while(1){
      scanf("%c",&moji);
      
      if(moji >= 'A' && moji <= 'Z'){
	moji += 'a'-'A';
	printf("%c",moji);
      }
      
      else if(moji >= 'a' && moji <= 'z'){
	moji -= 'a'-'A';
	printf("%c",moji);
      }
      
      else if(moji != '\n'){
	printf("%c",moji);
      }
      else if(moji == '\n'){
	break;
      }
  }
  printf("\n");
  return 0;
}
