#include<iostream>
#include<string>

using namespace std;

#define MAX 10000000

int main(){

	string data[MAX+1];
	int yeda[MAX+1];
	string en;
	int N,Q,eby,wy;
	int qu;
	string tm;

	while(1){
		cin>>N>>Q;
		if(N==0&&Q==0) break;
		for(int i=1;i<=MAX;i++){
			data[i]="Unkown";
			yeda[i]=-1;
		}
		for(int i=0;i<N;i++){
			cin>>en>>eby>>wy;
			for(int i=0;i<10;i++){
				data[wy-eby+1+i]=en;
				yeda[wy-eby+1+i]=i+1;
			}
		}
		for(int i=0;i<Q;i++){
			cin>>qu;
			if(data[qu]!="Unkown") cout<<data[qu]<<" "<<yeda[qu]<<endl;
			else cout<<"Unkown"<<endl;
		}
	}

	return 0;
}
