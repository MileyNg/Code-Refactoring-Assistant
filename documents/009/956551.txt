int Euler_phi(int mynum){//オイラーのファイ関数
  
  int ret = mynum;
  for(int i = 2; i * i <= mynum;i++){
    if(mynum%i == 0){
      ret -= ret/i;
      while(mynum%i == 0){
        mynum /= i;
      }
    }
  }
  if(mynum > 1){
    ret -= ret/mynum;
  }
  return ret;
}
