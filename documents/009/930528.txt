#include <stdio.h>
#include <string.h>

int main()
{
	int n;
	int i;
	int taro=0,hana=0;
	
	char s[101],tmp[101];
	
	scanf("%d",&n);
	for(i=0;i<n;i++){
		scanf("%s %s",s,tmp);
		if(strcmp(s,tmp) == 0){
			taro++;
			hana++;
		}
		else if(strcmp(s,tmp) < 0)
			hana += 3;
		else
			taro += 3;
	}
	printf("%d %d\n",taro,hana);
}
