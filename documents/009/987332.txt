#include<stdio.h>

int main(){

int n,input,sum=0,max=0,min=0,i;

scanf("%d",&n);

if(n==0){
 printf("undefined\n");
 return 0;
}
for(i=0;i<n;i++){
 scanf("%d",&input);
 if(i==0){
  min = input;
  max = input;
  sum = input;
 }else{
  if(max < input) max=input;
  if(input < min) min=input;
  sum += input;
 }
}

printf("%d %d %d\n",min,max,sum);

return 0;

}
