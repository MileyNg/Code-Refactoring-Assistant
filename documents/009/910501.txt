#include<stdio.h>
#include<math.h>
#define N 10000

main(){
  int count=0, i, k, j;
  int number[N]; 
  scanf("%d",&k);
  for(i=0;i<k;i++){
    scanf("%d", &number[i]);
    if(number[i] == 1)
      break;
  }
  
  for(i=0;i<k;i++){
    for(j=2; j <= sqrt(number[i]); j++)
      if(number[i] % j == 0){
	count += 1;
        break;
       }
      }
  printf("%d\n", k-count);
  return 0;
}
