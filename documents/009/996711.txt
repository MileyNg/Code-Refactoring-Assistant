#include <stdio.h>
#include <string.h>

int main(void)
{
	char W[10];
	char Ti[100];
	char T[1000];
	int i;
	int n=0;
	
	scanf("%s", W);
	
	while (strcmp(T, "END_OF_TEXT") != 0){
		
		Ti[0] = '\0';
		
		scanf("%s", T);
		
		for (i=0; T[i] != '\0'; i++){
			Ti[i] = T[i];
			if (Ti[i] >= 'A' && Ti[i] <= 'Z'){
				Ti[i] += 32;
			}
		}
		
		Ti[i] = '\0';

		if (strcmp(W,Ti) == 0){
			n++;
		}
			
	}
	printf("%d\n", n);
	return (0);
}
