#include <iostream>
#include <string>
#include <cmath>
#include <cstdio>

using namespace std;

int main()
{
	int a, b, C;
	cin >> a >> b >> C;

	double S, L, h, rad;
	rad = C * M_PI / 180.0;
	S = a * b * sin(rad) / 2;
	L = a + b + sqrt(a * a + b * b + 2 * a * b * cos(rad));
	h = b * sin(rad);
	
	printf("%.8lf\n", S);
	printf("%.8lf\n", L);
	printf("%.8lf\n", h);

	return 0;
}
