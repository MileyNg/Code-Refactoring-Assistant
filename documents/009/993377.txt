#include <cstdio>
#include <iostream>
#include <string.h>

using namespace std;

void decode(int l, int k[108], char s[108])
{
	char alph[56];
	int num[108];
	
	for(int i = 0; i < 26; i++){
		alph[i] = 'a' + i;
	}
	for(int i = 0; i < 26; i++){
		alph[i + 26] = 'A' + i;
	}
	
	for(int i = 1; i <= l; i++){
		if(s[i - 1] >= 'a' && s[i - 1] <= 'z'){
			num[i] = s[i - 1] - 'a';
		}
		else{
			num[i] = s[i - 1] - 'A' + 26;
		}
		if(num[i] - k[i] < 0){
			while(num[i] - k[i] < 0){
				num[i] += 52 - k[i];
			}
		}
		else{
			num[i] -= k[i];
		}
		cout << alph[num[i]];
	}
	puts("");
}


int main()
{
	int n, m, l;
	int k[108];
	char s[108];
	
	while(scanf("%d", &n) != EOF){
		if(n == 0) break;
		for(int i = 1; i <= n; i++){
			cin >> k[i];
		}
		cin >> s;
		
		m = n;
		l = strlen(s);
		while(l > m){
			for(int i = 1; i <= n; i++){
				k[m + i] = k[i];
			}
			m += n;
		}
		decode(l, k, s);
	}
	return 0;
}
