#include<stdio.h>
#include<string.h>
#include<stdlib.h>


int main(void)
{
  double a[4][2];
  int i,n,j;
  scanf("%d",&n);
  for(j=0;j<n;j++){
    for(i=0;i<4;i++)
      scanf("%lf %lf",&a[i][0],&a[i][1]);
    
    if(a[0][0]==a[1][0] && a[2][0]==a[3][0])
      printf("YES\n");
    else if( (a[1][1]-a[0][1])/(a[1][0]-a[0][0])==(a[3][1]-a[2][1])/(a[3][0]-a[2][0])   )
      printf("YES\n");
    else
      printf("NO\n");
  }
  return 0;
}
