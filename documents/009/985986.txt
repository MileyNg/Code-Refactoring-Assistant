#include<iostream>
#include<string>
#include<cstdio>

using namespace std;

int main(){
	string s;
	getline(cin,s);
	while(getline(cin,s),s[0]!='\0'){
		for(int i=0;i<s.size()-5;i++){
			if(s[i]=='H'&&s[i+1]=='o'&&s[i+2]=='s'&&s[i+3]=='h'&&s[i+4]=='i'&&s[i+5]=='n'&&s[i+6]=='o'){
				s[i+1]='a';s[i+6]='o';
			}
		}
		
		cout<<s<<endl;
		
	}

}
