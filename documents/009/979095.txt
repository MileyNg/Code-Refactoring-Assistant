int T, m, n;

main(){
	int i,j;
	cin >> T;
	while(T--){
		cin >> m >> n;
		int ans = 0;
		for(int p=-150;p<150;p++)for(int q=-150;q<150;q++){
			int g = p*p+q*q;
			if(!g) continue;
			if((m*p+n*q)%g == 0 && (m*q-n*p)%g == 0) ans ++;
		}
		cout << (ans>8 ? "C":"P") << endl;
	}
	return 0;
