#include<stdio.h>

int main(void){
	int number,x;

	while( scanf("%d %d",&number ,&x) ){
		if( (number==0)&&(x==0) ){
			break;
		}

		int i,j,k;
		int count = 0;

		for(i=1; i<=number; i++){
			for(j=1; j<=number; j++){
				for(k=1; k<=number; k++){
					if( (i+j+k==x)&&(i!=j)&&(j!=k)&&(k!=i) ){
						count++;
					}
				}
			}
		}
		count = count / 6;
		printf("%d\n",count);

	}

	return 0;
}
