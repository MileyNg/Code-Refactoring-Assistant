#include<stdio.h>
#define LENGTH 1200
#define SWITCH 1
main()
{
	char string[LENGTH];
	int i = 0;
	
	gets(string);
	
	#if SWITCH == 0
	printf("%s\n", string);
	#endif
	
	while(string[i] != '\0')
	{
		if((97 <= string[i]) && (string[i] <= 122))
		{
			string[i] -= 32;
		}
		else if((65 <= string[i]) && (string[i] <= 90))
		{
			string[i] += 32;
		}
		i++;
	}
	printf("%s\n", string);
}
