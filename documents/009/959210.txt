#include <iostream>
#include <cstdio>
using namespace std;

int main(){
  int a, b, c;
  char op;

  while(1){
    cin >> a >> op >> b;
    if (op == '?'){
      break;
    }
    else if (op == '+'){
      c = a+b;
    }
    else if (op == '-'){
      c = a-b;
    }
    else if (op == '*'){
      c = a *b;
    }
    else if (op == '/'){
      c = a/b;
        }
    else{
      break;
    }
    printf("%d", c);
  }
  return 0;
}
