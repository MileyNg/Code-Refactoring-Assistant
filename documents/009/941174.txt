#include <stdio.h>

void drawRect(int H, int W) {
	for (int y = 0; y < H; y++) {
		for (int x = 0; x < W; x++) {
			if ((x == 0) || (x == (W - 1)) || (y == 0) || (y == (H - 1))) {
				printf("#");
			} else {
				printf(".");
			}
		}
		printf("\n");
	}
	printf("\n");
}

int main(void) {
	int H = 0, W = 0;
	for (;;) {
		scanf("%d %d", &H, &W);
		if ((H == 0) && (W == 0)) {
			break;
		}
		drawRect(H, W);
	}
	return 0;
}
