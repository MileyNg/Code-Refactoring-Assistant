#include <stdio.h>

int main(void)
{
	int m, n, c = 0;
	int a, b;
	int i, max;
	
	scanf("%d %d", &n, &m);
	max = m;
	
	for (i = 0; i < n; i++){
		
		scanf("%d %d", &a, &b);
		
		m += a - b;
		
		if (max < m){
			max = m;
		}
		
		if (m < 0){
			c = 1;
		}
	}
	if (c == 0){
		printf("%d\n", max);
	}
	else {
		printf("0\n");
	}
	
	return (0);
}
