#include <iostream>
#include<vector>
using namespace std;

int main() {
	int n,m;
	int in;
	vector<int> a,b,sum;
	
	cin>>n>>m;
	for(int i=0;i<n;i++){
		cin>>in;
		a.push_back(in);
		sum.push_back(0);
	}
	
	for(int i=0;i<m;i++){
		cin>>in;
		b.push_back(in);
	}
	
	for(int i=0;i<m;i++){
		for(int j=0;j<n;j++){
			if(a[j]<b[i]){
				sum[j]++;break;
			}	
		}
	}
	
	int max=0,num;
	
	for(int i=0;i<n;i++){
		if(max<sum[i]){
			max=sum[i];
			num=i;
		}
	}
	
	cout<<num+1<<endl;
	return 0;
}
