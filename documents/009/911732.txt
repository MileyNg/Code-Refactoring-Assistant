#include<stdio.h>
#define N 100

main(){

	int A[N];
	int key;
	int i,j,k,l;

	scanf("%d",&k);
		
	for(j =	0;j < k;j++){
			scanf("%d",&A[j]);
	}	
	
	for(i = 0;i < k;i++){
		key = A[i];
		j =	i - 1;
			
		while(j >= 0 &&	A[j] > key){
			A[j + 1] = A[j];
			j--;	
			A[j + 1] = key;
		}	
			
	for(l =	0;l < k;l++){
		printf("%d",A[l]);	
	 }			
　　　　　printf("\n");
	}	
	
	return 0;
}
