#include <bits/stdc++.h>
using namespace std;
int a[1000010], p[1000010];
int main()
{
    int n, r, l;
    scanf("%d%d%d", &n, &r, &l);
    map<int, vector<pair<int, int>>> history;
    for(int i=0;i<r;++i) {
        int d, t, x;
        scanf("%d%d%d", &d, &t, &x);
        history[t].push_back(make_pair(d, x));
    }

    set<pair<int, int>, greater<pair<int, int>>> s;
    s.insert(make_pair(0, -1));
    int pt=0;
    for(auto& i: history) {
        int id=-s.begin()->second;
        int t=i.first;
        a[id]+=t-pt;
        for(auto& j: i.second) {
            int d=j.first;
            int x=j.second;
            auto k=s.find(make_pair(p[d], -d));
            if (k!=s.end()) s.erase(k);
            p[d]+=x;
            s.insert(make_pair(p[d], -d));
        }
        pt=t;
    }
    a[-s.begin()->second]+=l-(history.rbegin()->first);
    //for(int i=1;i<=n;++i) printf("%d ", a[i]); puts("");
    printf("%d\n", max_element(a+1, a+n+1)-a);
}
