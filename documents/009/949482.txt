#include <stdio.h>

int main(){

	
	int a, b,op;
	for (op = 1; op > 0;op++){
		scanf("%d %c %d", &a, &op, &b);
		if (op == '+'){
			printf("%d\n", a + b);
		}
		else if (op == '-'){
			printf("%d\n", a - b);
		}
		else if (op == '*'){
			printf("%d\n", a*b);
		}
		else if (op == '/'){
			printf("%d\n", a / b);
		}
		else if(op == '?'){
			printf("\n"); break;
		}
		else{}
		
	}
	return 0;
}
