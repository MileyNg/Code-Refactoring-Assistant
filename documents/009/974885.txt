#include <iostream>
using namespace std;
int main(){
  int a=1,b=1,c=1,n,x,i=0;
  
  while(1){
    
    cin>>n>>x;
    i=0;
    
    if(n==0 && x==0)return 0;

    for(a=1; a<=n; a++){
      
      for(b=1; b<=n; b++){
       
	for(c=1; c<=n; c++){

	  if(a+b+c==x && a<b && b<c){
	    i++;
	    continue;
	    
	  }
	  
	}
	
      }
      
    }
    
    cout<<i<<"\n";
    
  }
  
}
