#include<iostream>
using namespace std;

int main(){
	int i, j, n, q, ans = 0;
	int S[100];
	int T[100];

	cin >> n;
	for(i = 0; i < n; i++){
		cin >> S[i];
	}

	cin >> q;
	for(i = 0; i < q; i++){
		cin >> T[i];
	}

	for(i = 0; i < q; i++){
		for(j = 0; j < n; j++){
			if(S[j] == T[i]){
				ans++;
				break;
			}
		}
	}

	cout << ans << endl;

	return 0;
}
