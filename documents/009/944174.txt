#include<iostream>
#include<iomanip>
#include<cmath>
using namespace std;

#define PAI 3.14159265359

int main()
{
	int a,b,c,C;
	double S,L,h;

	cin >> a >> b >> C;

	S = (a * b * sin(C * PAI / 180)) / 2;
	L = a + b + sqrt(a * a + b * b - 2 * a * b * cos(C * PAI / 180));
	h = S / (a / 2);
	
	cout << setprecision(8) << fixed << S << endl << L << endl << h << endl;

	return 0;
}
