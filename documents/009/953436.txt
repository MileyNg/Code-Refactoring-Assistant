#include <stdio.h>
int a[20],n,cnt=1;
int canmake(int x){
	int i,j,k,sum;
	for(i=0;i<cnt;i++){
		sum=0;
		for(j=0,k=1;j<n||k<=i;j++,k<<=1){
			if(i&k)sum+=a[j];
			if(sum>=x)break;
		}
		if(sum==x)return 1;
	}
	return 0;
}
int main(void){
	int q,mq,i;
	scanf("%d",&n);
	for(i=0;i<n;i++)scanf("%d",&a[i]);
	cnt<<=n;	
	scanf("%d",&q);
	for(i=0;i<q;i++){
		scanf("%d",&mq);
		if(canmake(mq))printf("yes\n");
		else printf("no\n");
	}
	return 0;
}
