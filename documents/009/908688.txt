#include<iostream>
using namespace std;
int main(void)
{
  int n;
  cin>>n;
  int num[4][13];
  for(int i=0;i<4;i++)
    {
      for(int j=0;j<13;j++)
	{
	  num[i][j]=0;
	}
    }
  for(int i=0;i<n;i++)
    {
      char x;
      cin>>x;
      int y;
      cin>>y;
      if(x=='S')num[0][y-1]++;
      else if(x=='H')num[1][y-1]++;
      else if(x=='C')num[2][y-1]++;
      else if(x=='D')num[3][y-1]++;
    }
  for(int i=0;i<4;i++)
    {
      for(int j=0;j<13;j++)
	{
	  if(num[i][j]==0 && i==0)cout<<'S'<<" "<<j+1<<endl;
	  else if(num[i][j]==0 && i==1)cout<<'H'<<" "<<j+1<<endl;
	  else if(num[i][j]==0 && i==2)cout<<'C'<<" "<<j+1<<endl;
	  else if(num[i][j]==0 && i==3)cout<<'D'<<" "<<j+1<<endl;
	}
    }
  return 0;
}
