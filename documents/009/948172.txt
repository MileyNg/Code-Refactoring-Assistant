bool field[8][8];
int dx[4] = {1, 0, -1, 0};

void bomb(int x, int y) {
  field[x][y] = false;
  REP(i, 4) {
    for(int j = 1; j <= 3; j++) {
      int tx = x+dx[i]*j;
      int ty = y+dy[i]*j;
      if(tx < 0 || ty < 0 || tx == 8 || ty == 8) continue; 
      if(field[tx][ty])bomb(tx, ty);
    }
  }
}
