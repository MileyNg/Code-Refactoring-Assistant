#include<stdio.h>

main(){
  double a, b, c;
  double d, e, f;
  int res;
  double s_x, s_y, tmp1, tmp2;

  while(1){
    res = scanf("%lf %lf %lf %lf %lf %lf", &a, &b, &c, &d, &e, &f);

    if(res == EOF) break;

    if(a != 1){
      b /= a;
      c /= a;
      a /= a;
    }
    
    if(d != 1){
      e /= d;
      f /= d;
      d /= d;
    }

    tmp1 = b - e;
    tmp2 = c - f;
    s_y = tmp2 / tmp1;

    s_x = c - b*s_y;

   printf("%f %f\n", s_x, s_y);
  }

  return 0;
}
