int main(){
    string n, m;
    while(cin >> n >> m){
        int ans = 0;
        if(n.size() > m.size()) swap(n, m);
        for(int i=0; i < n.size(); i++){
            for(int j=i+1; j < n.size(); j++){
                string s = n.substr(i, j-i);
                if(m.find(s) != string::npos)ans = max(ans, (int)s.size());
                else {i++; j = i;}
            }
        }
        cout << ans << endl;
    }
}
