#include <stdio.h>

int main(void)
{
	char c;
	int n[25];
	int i;
	
	for (i=0;i < 25; i++){
		n[i] = 0;
	}
	
	while (scanf("%c", &c) != EOF){
		
		if(c >= 'a' &&c <= 'z'){
		i = c - 'a';
		n[i]++;
		}
		else if(c >= 'A' &&c <= 'Z'){
		i = c - 'A';
		n[i]++;
		}
		
	}
	
	for (i=0;i < 25; i++){
		printf("%c : %d\n", i+'a', n[i]);
	}
	
	return (0);
}
