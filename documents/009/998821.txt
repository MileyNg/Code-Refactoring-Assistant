#include <stdio.h>

int main(){
	int i,j;
	int a,b,c,d;
	int date_a[200];
	int date_b[200];
	
	while(1){
		date_a[0]=0;
		date_b[0]=0;
		for(i=1;i<200;i++){
			date_a[i]=-1;
			date_b[i]=-1;
		}
		scanf("%d",&c);
		if(c==0){
			break;
		}
		d=1;
		while(c!=d){
				scanf("%d %d",&a,&b);
			if(b%2){
				date_b[d]=date_b[a];
				if(b==1){
					date_a[d]=date_a[d-1]+1;
				}else{
					date_a[d]=date_a[d-1]-1;
				}
			}else{
				date_a[d]=date_a[a];
				if(b==0){
					date_b[d]=date_b[d-1]+1;
				}else{
					date_b[d]=date_b[d-1]-1;
				}
			}
			d++;
		}
		int a_max=-1,b_max=-1;
		int a_min=1,b_min=1;
		for(i=0;i<c;i++){
			if(date_a[i]>a_max){
				a_max=date_a[i];
			}
			if(date_a[i]<a_min){
				a_min=date_a[i];
			}
			if(date_b[i]>b_max){
				b_max=date_b[i];
			}
			if(date_b[i]<b_min){
				b_min=date_b[i];
			}
		}
		printf("%d %d\n",b_max-b_min+1,a_max-a_min+1);
	}
	return 0;
}
