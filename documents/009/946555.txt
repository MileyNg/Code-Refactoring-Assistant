#include <stdio.h>

#define PRIME_MAX 1000000
int getprimes(int list[], int top){
    int i, j;
    int index = 0;
    char tmp[PRIME_MAX];

    for(i=0;i<top;i++) {
        tmp[i] = 1;
    }
   
    tmp[0] = tmp[1] = 0;
    for(i=2;i<top;i++) {
        if(tmp[i] == 1) {
            /* 素数 */
            list[index] = i;
            index ++;
            for(j=2;j*i<top;j++) {
                tmp[j*i] = 0;
            }
        }
    }

    list[index] = -1;
    return index;
}

int main(void) {
    int i;
    int n;
    int size;
    int primes[100000];
    size = getprimes(primes, PRIME_MAX);
   
    while(scanf("%d\n", &n) == 1) {
        for(i=0;i<size;i++) {
            if(primes[i] > n) {
            	printf("%d %d\n", primes[i-1], primes[i]);
            	break;
            }
        }
    }
   
    return 0;
}
