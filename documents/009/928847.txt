#include <stdio.h>
 
int main(void) {
    int a, b, xy, i, m;
    scanf("%d %d", &a, &b);
    /* 整数 x, y について、1 < y < x ならば
       x と y の最大公約数は y と x%y の
       最大公約数に等しい。
       ここで a%b は a を b で割った余りである。 */
        
    if(1 < a && 1 < b) {
        if(a == b) {
            printf("%d\n", a);
            return 0;
        }
        
        if(a < b) m = a; a = b;	b = m;
        }
        do{
        	m = b % a;
        	if(m == 0) {
        		printf("%d\n", a);
        		return 0;
        	}
        	b = a; a = m;
        }while(1);
    }
     
    return 0;
}
