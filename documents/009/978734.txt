#include <iostream>
using namespace std;

int main() {
	int i;
	char c;
	string s;
	getline(cin,s);
	for(i=0; i<s.size(); i++){
		if(65<=s[i] && s[i]<=90){
			c=s[i]+32;
		}else if(97<=s[i] && s[i]<=122){
			c=s[i]-32;
		}
		else{
			c=s[i];
		}
		cout <<c;
	}

	return 0;
}
