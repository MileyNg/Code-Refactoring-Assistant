#include <iostream>
#include <vector>

using namespace std;

int main() {
    int n;
    while (cin >> n && n) {
        int g[26] = {0};
        string s;
        for (int i=0; i<n; ++i) {
            cin >> s;
            g[s.back() - 'a'] ++;
            g[s.front() - 'a'] ++;
        }
        bool ok = true;
        for (int x : g) {
            if (x % 2 != 0) {
                ok = false;
                break;
            }
        }
        cout << (ok ? "OK" : "NG") << endl;
    }
    return 0;
}
