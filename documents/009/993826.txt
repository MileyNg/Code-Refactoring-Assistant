#include<iostream>
#include<algorithm>
#include<cmath>
#include<cctype>
#include<vector>
#include<numeric>
using namespace std;
int main(){
        int n;
        scanf("%d",&n);
        vector<int>ar(n);
        for(auto&e:ar){
                scanf("%d",&e);
        }
        sort(ar.begin(),ar.end());
        long long s=accumulate(ar.begin(),ar.end(),0);
        if(n!=0){
            printf("%d %d %lld\n",ar[0],ar[n-1],s);
        }else{
                printf("0 0 0\n");
        }
        return 0;
}
