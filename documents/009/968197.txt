#include<stdio.h>
#define MAX 1000

int A[MAX],B[MAX],C[MAX];
int main(){
	int n,i;

scanf("%d",&n);
for(i=1;i<=n;i++){
scanf("%d",&A[i]);
}
for(i = 1;i <= n;i++){
	C[i]=0;
}
for(i=1;i<=n;i++){
	C[A[i]]++;
}
for(i=1;i<=n;i++){
	C[i] = C[i] + C[i - 1];
}
for(i = n; i >= 1;n--){
	B[C[A[i]]] = A[i];
	C[A[i]]--;
}
for(i=1;i<=n;i++){
	printf("%d",B[i]);
	if(i != n){
	printf(" ");
	}
}
return 0;
}
