#include<stdio.h>
#define SENTINEL 100000
int Merge(int *,int,int,int);
int Merge_Sort(int*,int,int);
 int S[SENTINEL];
int A[SENTINEL];
int L[SENTINEL];
int R[SENTINEL];
int n,count = 0;

main(){
  int left,right,i;
  scanf("%d",&n);
  for(i=0;i<n;i++){
    scanf("%d",&S[i]);
  }
  left = 0;
  right = n;
 
  Merge_Sort(S,left,right);

for(i=0;i<n;i++){
  printf("%d",S[i]);
 
 if(i < n-1){
   printf(" ");
 } 
}
 printf("\n");
 printf("%d\n",count);
 return 0;
}
int Merge(int *A,int left,int mid,int right){
  int n1,n2,i,j,k;
  n1= mid - left;
  n2 = right - mid;
  for(i = 0;i < n1;i++){
    L[i] = S[left + i];
  }
  for(i = 0; i < n2;i++){
    R[i] = S[mid + i];
  }
  L[n1] = SENTINEL;
  R[n2] = SENTINEL;
  i = 0;
  j = 0;
 for(k = left;k < right;k++){
   if(L[i] <= R[j]){
     count++;
     A[k] = L[i];
     i++;
   }
   else{
     count++;
     A[k] = R[j];
     j++;
   }
 }
 return 0;
}
int Merge_Sort(int* A,int left,int right){
  int mid;
  if(left+1 < right){
    mid = (left + right)/2;
    Merge_Sort(A,left,mid);
    Merge_Sort(A,mid,right);
    Merge(A,left,mid,right);
  }
}
