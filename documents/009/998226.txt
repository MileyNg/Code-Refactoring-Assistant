#include<iostream>
using namespace std;
int main(){
  long n,m,l;
  long A[100][100],B[100][100],C[100][100];


  cin >> n >> m >> l;

 for(int i=0;i<n;i++){
   for(int j=0;j<l;j++){
   C[i][j]=0;
   }
 }
 

  for(int i=0;i<n;i++){
    for(int k=0;k<m;k++){
      cin >> A[i][k];
    }
  }

for(int k=0;k<m;k++){
    for(int j=0;j<l;j++){
      cin >> B[k][j];
    }
  }

 for(int i=0;i<n;i++){
   for(int k=0;k<m;k++){
     for(int j=0;j<l;j++){
       
       C[i][j]+=A[i][k]*B[k][j];
     }
   }
 }

 for(int i=0;i<n;i++){
   for(int j=0;j<l-1;j++){
     
     cout << C[i][j] << " " ;
   }
   cout << C[i][l-1] <<endl;
 }
   

    return 0;}
