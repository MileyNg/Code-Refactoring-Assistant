#include <iostream>
using namespace std;

int H,W;
int main() {
	while(true) {
		cin >> H >> W;
		if(H == 0 && W == 0) break;
		for (int i = 0; i < H; ++i) {
			for(int k=0; k<W; ++k) {
				if(i>0 && i < H-1 && k>0 && k < W-1) cout << ".";
				else cout << "#";
			}
			cout << endl;
		}
		cout << endl;
	}
}
