#include<stdio.h>
int main(void)
{
   int x,y;
   scanf("%d%d",&x,&y);
   for (;;){
   if(x <= y && 0 < x){
      printf("%d %d\n",x,y);
      scanf("%d%d",&x,&y);
   }
   else if(y < x){
      printf("%d %d\n",y,x);
      scanf("%d%d",&x,&y);
   }
   else if(x == 0 && y == 0) break;
   }
   return 0;
}
