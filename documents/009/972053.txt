#include<stdio.h>
#include<string.h>
main(){
  char W[100],T[10000];
  int d=0,i;
  scanf("%s",W);
  for(i=0;i<100;i++){
    W[i] = toupper(W[i]);
  }
  while(1){
    scanf("%s",T);
    for(i=0;i<10000;i++){
      T[i]=toupper(T[i]);
    }
    if(strcmp(T,"END_OF_TEXT") == 0){
      break;
    }
    if(strcmp(W,T) == 0){
      d=d+1;
    }
  }
  printf("%d\n",d);
  return 0;
}
