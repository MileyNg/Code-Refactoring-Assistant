#include<iostream>
#include<string>
using namespace std;
int main(){
  string s;
  int count[3]={0},a=0;
  int ans = 0;
  cin >> s;
  for(int i=0;i<s.size();i++){
    if(a==0&&s[i]=='J') count[0]++;
    else if(a==0&&count[0]>0&&s[i]=='O'){
      //cout << count[0] << endl;
      a = 1;
      count[1]=1;
    }
    else if(a==1&&s[i]=='O') count[1]++;
    else if(a==1&&s[i]=='I'&&count[0]>=count[1]){
      //cout << count[1] << endl; 
      a = 2;
      count[2]=1;
    }
    else if(a==2&&s[i]=='I') count[2]++;
    else if(a==2&&s[i]!='I'){
      if(count[2]>=count[1]&&count[1]>ans) ans = count[1];
      count[0] = 0;
      if(s[i]=='J'){
	//cout << 'K' << endl;
	a = 0;
	count[0] = 1;
      }
    }
    else{
      a = 0;
      count[0] = count[1] = count[2] = 0;
    }
  }
  if(count[2]>=count[1]&&count[1]>ans) ans = count[1];
  cout << ans << endl;
  return 0;
}
