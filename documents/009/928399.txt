#include<stdio.h>
#include<string.h>
#define LEN 100005

typedef struct pp{
  char name[100];
  int t;
}P;

P Q[LEN+1];
int head, tail, n;

void enqueue(P x){
  Q[tail] = x;
  tail++;
}

P dequeue(){
  P s;
  head++;
  s=Q[head];
  return s;
}

int main(){
  int elaps = 0, c;
  int i, q;
  P u;
  scanf("%d %d", &n, &q);

  for ( i = 1; i <= n; i++){
    scanf("%s", Q[i].name);
    scanf("%d", &Q[i].t);
  }
  
  tail = n+1;
  
  while(head < tail-1){
    u = dequeue();
    c = u.t - q;
    
    if(c > 0){
      elaps += q;
      u.t = c;
      enqueue(u);
    }
    
    if(c <= 0){
      elaps += u.t;
      printf("%s %d\n", u.name, elaps);
    }
  }
  return 0;
