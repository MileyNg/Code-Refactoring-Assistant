#include <stdio.h>
int main(){
	int a=0, b=0, c=0, d=0, e=0, f=0, aaa[4][3][10], ai=0, ue=0, oka=0;
	scanf("%d", &a);
	scanf("%d %d %d %d", &c, &d, &e, &f);
	aaa[c][d][e] = f;
	for(b=1;b<a;b++){
		scanf("%d %d %d %d", &c, &d, &e, &f);
		aaa[c][d][e] = aaa[c][d][e] + f;
	}
	for(ai=1;ai<=4;ai++){
		for(ue=1;ue<=3;ue++){
			for(oka=1;oka<=10;oka++){
				if(aaa[ai][ue][oka]>=0 && aaa[ai][ue][oka] <=100){
					printf(" %d", aaa[ai][ue][oka]);
				}else{printf(" 0");}

			}printf("\n");

		}printf("####################\n");
	}
	return 0;
}
