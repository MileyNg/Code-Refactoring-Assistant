#include <iostream>
#include <string>
#include <algorithm>

using namespace std;

int main(int argc, char** argv){
	string str, word;
	int cnt = 0;
	string::size_type index = (string::size_type)0;
	getline(cin, word);
	while (getline(cin, str), str != "END_OF_TEXT"){
		transform(str.begin(), str.end(), str.begin(), ::tolower);
		while(true){
			index = str.find(word, index);
			if (index == string::npos){
				break;
			}
			index++;
			cnt++;
		}
	}
	cout << cnt << endl;
}
