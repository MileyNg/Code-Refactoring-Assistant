#include <iostream>
using namespace std;

int main() {
	int m,f,r,s;
        while(1){
             cin>>m>>f>>r;
             s=m+f;
             if(m==-1　&& f==-1) break;
             if(m==-1 ||　f==-1) cout <<F<<endl;
             if(s>= 80) cout<<A<<endl;
             if(s<80 && s>=65) cout<<B<<endl;
             if(s<65 && s>=50) cout<<C<<endl;
             if(s<50 && s>=30 && r>=50) cout<<C<<endl;
             if(s<50 && s>=30 && r<50) cout <<D<<endl;
             if(s<30) cout<<F<<endl;
         }

	return 0;
}
