#include<stdio.h>

int A[20],M[200];
int flag=0,n;

void recursion(int p,int t){
  if(t==0)flag=1;
  if(p<n && t>0){
    p++;
    recursion(p,t); 
    t-=A[p];
    recursion(p,t);
  }
}



main(){
  int i,q;
  scanf("%d",&n);
  for(i=0;i<n;i++){
    scanf("%d",&A[i]);
  }

  scanf("%d",&q);
  for(i=0;i<q;i++){
    scanf("%d",&M[i]);
  }

  for(i=0;i<q;i++){
    recursion(-1,M[i]);
    if(flag==1) printf("yes\n");
    else printf("no\n");
    flag=0;
  } 
  return 0;
}
