#include<stdio.h>

int main(){
  int n[10000]={0};
  int m,f,r,i;
  
  for(i=0;i<10000;i++){
    scanf("%d %d %d",&m,&f,&r);
    if(m==-1&&f==-1&&r==-1)break;
    if(m==-1||f==-1){
      n[i]=6;
    }
    else if(m+f >=80){
      n[i]=1;
    }
    else if(m+f>=65){
      n[i]=2;
    }
    else if(m+f>=50){
      n[i]=3;
    }
    else if(m+f>=30){
      if(r>=50){
	n[i]=3;
      }
      else{
	n[i]=4;
      }
    }
    else{
      n[i]=6;
    }
  }
  for(i=0;n[i]!=0;i++){
    switch (n[i]){
    case 1:printf("A\n");	break;
    case 2:printf("B\n");break;
    case 3:printf("C\n");break;
    case 4:printf("D\n");break;
    case 6:printf("F\n");break;
    }
  }
  return 0;
}
