{
    int a[5],i,b,j;
    for(i=0;i<5;i++)

        scanf("%d",&a[i]);
    }
    for(i=0;i<20;i++)
    {
        for(j=0;j<5;j++)
        {
            if(a[j]<a[j+1])

            }

    }

}
