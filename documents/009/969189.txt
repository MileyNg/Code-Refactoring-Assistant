#include <stdio.h>
#define N 100000

int Partition(int, int);
int A[N];

main()
{

  int n,i,tmp,k;
 

  scanf("%d",&n);

  for(i = 0; i < n; i++){

    scanf("%d",&A[i]);

  }
  
 k = Partition(0, n-1);
  for(i = 0; i < k; i++){
    printf("%d ",A[i]);

  }
  printf("[%d] ",A[k]);
for(i = k+1; i < n; i++){
    printf("%d",A[i]);
    if(i < n){
      printf(" ");
    }
  }

  printf("\n");
  return 0;
}


int Partition(p, r)
{

  int i,j,x,tmp;
 
 


  x = A[r];
  i = p-1;

  for(j = p; j <= r-1; j++){
    if(A[j] <= x){
      i = i+1;
      tmp = A[i];
      A[i] = A[j];
      A[j] = tmp;
      
    }
  }
  
  tmp = A[i+1];
  A[i+1] = A[r];
  A[r] = tmp;

  return i+1;

}
