#include<stdio.h>
#define N 10000
#define Q 500

int Linearsearch(void);
int n,S[N],q,T[Q];

int main(){
  int i,j;

  //input
  scanf("%d",&n);

  for(i=0 ; i<n ; i++){
    scanf("%d",&S[i]);
  }
  
  scanf("%d",&q);

  for(i=0 ; i<q ;i++){
    scanf("%d",&T[i]);
  }

  //seach
  printf("%d\n", Linearsearch());
  
  return 0;
}

int Linearsearch(){
  int i,j,key;
 
  for(i=0 ; i<q ; i++){
    key=T[q];
    for(j=0 ; j<n ; j++){
      if(S[j]==key)   return S[j];
    }
  }
}
