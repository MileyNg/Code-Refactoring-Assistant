#include <stdio.h>

#define N 100

int main(){
  int i,j,n,A[N],tmp,count=0;
  scanf("%d",&n);
  for(i=0;i<n;i++){
    scanf("%d",&A[i]);
  }
  for(i=0;i<n;i++){
    for(j=n;j>0;j--){
      if(A[j]<A[j-1]){
	tmp=A[j];
	A[j]=A[j-1];
	A[j-1]=tmp;
	count++;
      }
    }
  }
 for(i=0;i<n;i++) printf("%d ",A[i]);
 printf("\n%d\n",count);
}
