#include <stdio.h>
main(){
  int a[100],tmp=0,n,j,i;
  scanf("%d",&n);
  for(i=0;i<n;i++){
    scanf("%d",&a[i]);
  }
  for(i=0;i<n-1;i++){
    for(j=n-1;j>i;j--){
      tmp=a[j];
      a[j]=a[j-1];
      a[j-1]=tmp;
    }
  }
  for(i=0;i<n;i++){
    printf("%d",a[i]);
    if(i<n-1){
      printf(" ");
    }
  }
  printf("\n");
  return 0;
}
