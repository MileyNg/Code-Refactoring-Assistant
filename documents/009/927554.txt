#include <iostream>
using namespace std;
int main (){

  int a,b,c,d,e,f;
  char op;
  
  while(0<=a && b<=20000){

    cin >> a >> op >> b;

    if( op == '+' ){
    
      c=a+b;
      cout <<c<<"\n";
 
    } else if( op == '-' ){

      d=a-b;
      cout <<d<<"\n";

    } else if( op == '*'){

      e=a*b;
      cout <<e<<"\n";

    } else if( op == '/'){

      f=a/b;
      cout <<f<<"\n";
 
    } else if( op =='?'){

      break;

    }
  }

  return 0;

}
