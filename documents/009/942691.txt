bool isPalindrome(int k) {
	stringstream ss;
	ss << k;
	string s(ss.str());
	int n = s.length();
	for (int i = 0; i < n/2; i++) {
		if (s[i] != s[n-i-1])
			return false;
	}
	return true;
}

int main() {
	int n;
	cin >> n;
	int min = 10000000, res = 0;
	for (int i = 0; i < 100000; i++) {
		if (isPalindrome(i) and abs(n-i) < min) {
			min = abs(n-i);
			res = i;
		}
	}
	cout << res;
}
