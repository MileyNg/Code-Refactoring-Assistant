#include <iostream>
#include <string>

using namespace std;

int main(int argc, char** argv){
	string str, shuffle;
	int time, num;

	while (cin >> str, str != "-"){
		cin >> time;
		for (int i = 0; i < time; i++){
			cin >> shuffle;
			num = std::atoi(shuffle.c_str());
			str += str.substr(0, num);
			str.erase((string::size_type)0, (string::size_type)num);
		}
		cout << str << endl;
	}
}
