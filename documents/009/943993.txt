#include<stdio.h>

int main(void){
	int sum,card,one,i;
	char code;

	while(0==0){
		sum=0; one=0;
		scanf("%d%c",&card,&code);
		if(card>10) card=10;
		else if(card==1) one++;
		sum+=card;

		if(code=='\n'){
			if(sum==0) break;
			else if(sum>21) sum=0;

			else if(one>0)
				for(i=0;i<one;i++)
					if(sum+10<=21) sum+=10;
			printf("%d\n",sum);
		}
	}
	return 0;
}
