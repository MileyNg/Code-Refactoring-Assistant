#include<cstdio>
#include<iostream>
#include<string>
#include<vector>
#include<algorithm>

using namespace std;

#define reps(i,f,n) for(int i=f;i<int(n);i++)
#define rep(i,n) reps(i,0,n)

typedef long long ll;

const int N = 100010;

class BIT{
	static const int N = 100010;
	public:
	ll t[N];
	void init(){rep(i,N)t[i]=0;}
	ll calc(ll a,ll b){return max(a,b);}
	
	void setval(int pos, ll val){
		pos++;
		for(int i=0;;i++){
			if(pos>=N)break;
			if(pos&(1<<i)){
				t[pos] = calc(t[pos],val);
				pos += (1<<i);
			}
		}
	}
	
	ll getval(int pos){
		pos++;
		ll ret = 0;
		for(int i=0;;i++){
			if(pos==0)break;
			if(pos&(1<<i)){
				ret = calc(ret, t[pos]);
				pos -= (1<<i);
			}
		}
		return ret;
	}
	
	void print(){
		printf("t : ");
		rep(i,10)printf("%d ",t[i]);puts("");
	}
};

int main(){
	int n;
	cin>>n;
	
	int p[N];
	ll sum = 0;
	rep(i,n){
		cin>>p[i];
		sum += p[i];
	}
	
	BIT bit;
	bit.init();
	
	rep(i,n){
		bit.setval( p[i], bit.getval(p[i])+p[i] );
		//bit.print();
	}
	
	cout<<sum - bit.getval(n-1)<<endl;
}
