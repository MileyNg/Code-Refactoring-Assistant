#include<stdio.h>
int main(){

  int i,j;
  int n,q;
  int s[100]={0};
  int t;
  int cnt=0;

  scanf("%d",&n);

  for(i=0;i<n;i++){
    scanf("%d",&s[i]);
  }
  
  scanf("%d",&q);

  for(i=0;i<q;i++){
    scanf("%d",&t);

    for(j=0;j<n;j++){
      if(t==s[j]){
	cnt++;
      }
    }
  }

  printf("%d\n",cnt);

  return 0;
}
