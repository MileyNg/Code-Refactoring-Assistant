#include <iostream>
#include <vector>
#include <string>
#include <map>
#include <set>
#include <algorithm>

#define REP(i,n) for(int i=0;i<(int)(n);i++)

using namespace std;

int main() {
  while(1){
    string s;
    string t;
    cin >> s;
    cin >> t;
    if(!cin) break;
    set<string> st;
    for (int i = 0; i < s.size(); ++i) {
      for (int j = 0; i + j < s.size(); ++j) {
        st.insert(s.substr(i,j));
      }
    }
    size_t ml = 0;
    for (const string& str : st) {
      if(t.find(str) != string::npos)
        ml = max(ml, str.size());
    }
    cout << ml << endl;
  }
  return 0;
}
