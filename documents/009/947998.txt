using namespace std;

string IntToString(int number) {
    stringstream ss;
    ss << number;
    return ss.str();
}

int main(){
    int n;
    string s;
    cin >> n >> s;
    for(int i=0; i < n; i++){
        string t;
        for(int j=0; j < s.size(); j++){
            string p = s.substr(j, 1);
            int count = 1;
            for(int k=j+1; k < s.size(); k++){
                j = k;
                if(s.substr(k, 1) == p)count++;
                else {j--; break;}
            }
            t += IntToString(count) + p;
        }
        s = t;
    }
    cout << s << endl;
}
