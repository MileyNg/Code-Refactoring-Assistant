#include<iostream>
#include<string>
using namespace std;
int main() {
string str;
int n;
while(1){
	cin >> str;
	if(str=="-")break;
	cin >> n;
	int h[n];
	for(int i=0;i<n;i++){
		cin >> h[i];
      	string left =str.substr(0,h[i]);
      	string right =str.substr(h[i]);
        str = right + left;
	}
	cout <<  str << endl;
}
return 0;
}
