using namespace std;

int main() {
    int a, b;
    char op;
    while (cin >> a >> op >> b) {
        int ans = 0;
        switch (op) {
        case '+':
            ans = a + b;
            break;
        case '-':
            ans = a - b;
            break;
        case '*':
            ans = a * b;
            break;
        case '/':
            ans = a / b;
            break;
        default:
            return 0;
        }
        cout << ans << endl;
    }
}
