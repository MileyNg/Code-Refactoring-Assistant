#include <stdio.h>

int func(double a ,double b,double c,double d,double e,double f){
	int y;
	y=(c*d-a*f)/(b*d-a*e)+0.00001;
	return y;
}

int main(void){
	double a,b,c,d,e,f,x,y;
	while(scanf("%lf %lf %lf %lf %lf %lf",&a,&b,&c,&d,&e,&f) != EOF){
		y=func(a,b,c,d,e,f);
		x=(c-b*y)/a+0.00001;
		printf("%.3lf %.3lf\n",x,y);
	}
	return 0;
}
