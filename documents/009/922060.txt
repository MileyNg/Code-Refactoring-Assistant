#include <cstdlib>
#include <iostream>
#include <string>
std::string B="4141212141412121";
int main(){
	int i,j,C[8]={},x[8]={},s,m,p,tmp;
	for(i=8;i--;)C[i]=atoi(B.substr(i,8).c_str());
	while(std::cin>>x[0]){
		m=x[0];
		for(i=1;i<8;i++){
			std::cin>>x[i];
			m+=x[i];
		}
		for(i=8;i--;){
			s=0;
			for(j=8;j--;){
				tmp=x[j]-B[i+j]+'0';
				s+=(tmp>0?tmp:0);
			}
			if(s<m){
				m=s;
				p=i;
			}
			else if(m==s&&C[i]<C[p])p=i;
		}
		for(i=0;i<8;i++)std::cout<<B[p+i]<<" \n"[i==7];
	}
}
