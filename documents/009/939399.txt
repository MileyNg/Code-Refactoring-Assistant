#include <sstream>
#include <algorithm>
#include <iomanip>
#include <iostream>

#define rep(i,n) for(int i=0;i<(int)(n);i++)

using namespace std;
/** Problem0169 : Blackjack **/
int main()
{
	int ans, one, t;
	string s;
	
	while (!cin.eof()) {
		getline(cin, s);
		if (s == "0") break;
		stringstream ss(s);
		ans=0; one=0;
		while (!ss.eof()) {
			ss >> t;
			ans += t;
			if (ans > 21 && one>0) {
				ans -= 10;
				one--;
			}
		}
		if (ans > 21) ans=0;
		cout << ans << endl;
	}
	return 0;
}
