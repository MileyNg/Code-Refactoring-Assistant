#include<stdio.h>
#include<stdlib.h>

int main(){
  int n,i,r;
  int h[10];
  
  while(scanf("%d",&n) == 1){

    r=-1;
    if(n>=512){
      n-=512;
      ++r;
      h[r]=512;
    }
    if(n>=256){
      n-=256;
      ++r;
      h[r]=256;
    }
    if(n>=128){
      n-=128;
      ++r;
      h[r]=128;
    }
    if(n>=64){
      n-=64;
      ++r;
      h[r]=64;
    }
    if(n>=32){
      n-=32;
      ++r;
      h[r]=32;
    }
    if(n>=16){
      n-=16;
      ++r;
      h[r]=16;
    }
    if(n>=8){
      n-=8;
      ++r;
      h[r]=8;
    }
    if(n>=4){
      n-=4;
      ++r;
      h[r]=4;
    }
    if(n>=2){
      n-=2;
      ++r;
      h[r]=2;
    }
    if(n>=1){
      n-=1;
      ++r;
      h[r]=1;
    }
    for(i=r;i>=0;--i){
      printf("%d ",h[i]);
      // h[i] <=> *(h + i)
    }
    printf("\b\n");
  }
}
