#include<iostream>

using namespace std;

int main(){

	int N,M;
	int a;
	int amida[10][10];
	int h;

	cin>>N>>M;

	for(int i=0;i<10;i++){
		for(int j=0;j<10;j++){
			amida[i][j]=0;
		}
	}

	for(int i=1;i<=M;i++){
		cin>>a;
		amida[i][a]=1;
	}

	for(int i=1;i<=M;i++){
		for(int j=1;j<=N;j++){
			if(amida[i][j]==1){
				h=i;
				while(1){
					if(amida[h-1][j]==0&&amida[h-1][j+1]==0&&amida[h-1][j-1]==0&&h!=1){
						h--;
					}
					else break;
				}
				amida[i][j]=0;
				amida[h][j]=1;
			}
		}
	}

	for(int i=9;;i--){
		for(int j=0;j<=9;j++){
			if(amida[i][j]==1){
				cout<<i<<endl;
				return 0;
			}
		}
	}

	return 0;
}
