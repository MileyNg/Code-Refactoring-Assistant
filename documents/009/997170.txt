#include <bits/stdc++.h>
using namespace std;
#define rep(i,n) for(int i=0;i<(n);i++)

int dx[] = {0,1,1,0,-1,-1};
int dy[] = {1,1,0,-1,-1,0};
int t,n;
set<pair<int,int> > point;
typedef pair<int,int> P;
const int C = 40;
const int INF = 100000;

void solve() {
    int dist[103][103];
    rep(i,103) rep(j,103) dist[i][j]=INF;
    int x,y;
    rep(i,n) {
        cin>>x>>y;
        x+=C;
        y+=C;
        
        point.insert(make_pair(y,x));
    }
    queue<P> que;
    cin>>x>>y;
    x+=C;
    y+=C;
    que.push(P(y,x));
    dist[y][x]=0;

    while(que.size()) {
        P p = que.front(); que.pop();
        int y = p.first;
        int x = p.second;
        rep(i,6) {
            int nx = x + dx[i];
            int ny = y + dy[i];
            if(0<=nx && 0<=ny && dist[ny][nx] == INF && point.find(P(ny,nx)) == point.end()) {
                dist[ny][nx] = dist[y][x]+1;
                que.push(P(ny,nx));
            }
        }
    }
    int ans = 0;
    rep(i,102) rep(j,102) if(dist[i][j]<=t) ans++;
    cout<<ans<<endl;

    point.clear();
    

}

int main() {
    while(cin>>t>>n) {
        if(t+n==0) return 0;
        solve();
    }
    return 0;
}
