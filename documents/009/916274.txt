#include<stdio.h>
main()
{
	int x = 0;
	int y = 0;
	
	scanf("%d %d", &x, &y);
	
	do
	{
		if(x > y)
		{
			printf("%d %d\n", y, x);
		}
		else if(x < y)
		{
			printf("%d %d\n", x, y);
		}
		else
		{
			printf("%d %d\n", x, y);
		}
		
		scanf("%d %d", &x, &y);
	}while((x != 0) || (y != 0));
	
	return 0;
}
