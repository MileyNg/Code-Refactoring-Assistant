#include <iostream>

bool u_c[17][17];

void shokika(){
	for(int i = 0; i < 17; ++i){
		u_c[0][i] = true;
		u_c[i][0] = true;
	}
}

//階乗を求める関数
int f(int n){
	if(n == 0){
		return 1;
	}
	else{
		return f(n - 1) * n;
	}
}

int calc_road(int x, int y){
	if(x == 1 && y == 1){
		return 1;
	}
	else if(x <= 0 || y <= 0){
		return 0;
	}
	else{
		int num;
		if(u_c[x - 1][y] && u_c[x][y - 1]){
			return 0;
		}
		else if(u_c[x - 1][y]){
			num = calc_road(x, y - 1);
			return num;
		}
		else if(u_c[x][y - 1]){
			num = calc_road(x - 1, y);
			return num;
		}
		else{
			num = calc_road(x - 1, y) + calc_road(x, y - 1);
			return num;
		}
	}

}

int main(){
	//1行目の入力
	while(true){
		int a, b;
		std::cin >> a >> b;
		if(a == 0 && b == 0){
			break;
		}
		//2行目の入力
		int n;
		std::cin >> n;
		//3行目以降の入力と処理（工事中の箇所）
		for(int i = 0; i < n; ++i){
			int x, y;
			std::cin >> x >> y;
			u_c[x][y] = true;
		}
		shokika();
		std::cout << calc_road(a, b) << std::endl;
	}
}
