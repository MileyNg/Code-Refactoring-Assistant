#include<iostream>
using namespace std;

static int cnt = 0;

template<class T>
void insertionSort(T* A, int n, int g)
{
        T v;
        int j;
        for(int i=g; i<n; i++){
                v = A[i];
                j = i-g;
                while(j>=0 && A[j]>v){
                        A[j+g] = A[j];
                        j = j - g;
                        cnt++;
                }
                A[j+g] = v;
        }
}

// ソートする配列、配列の要素数
template<class T>
void shellSort(T* array, int size)
{
        int m=1, n;
        n = (size-1)/3 + 1;
        if( n>100 ){ n=100; }
        int G[n];
        for(n=0; m<size && n<=100; n++){
                G[n] = m;
                m = 3*m+1;
        }

        cout << n << endl;
        cout << G[n-1];
        for(int i=n-2; i>=0; i--){
                cout << " " << G[i];
        }
        cout << endl;

        for(int i=0; i<n; i++){
                insertionSort(array, size, G[i]);
        }
}



int main()
{
        int n;
        cin >> n;
        int a[n];
        for(int i=0; i<n; i++){
                cin >> a[i];
        }

        shellSort(a, n);
        cout << cnt << endl;
        for(int i=0; i<n; i++){
                cout << a[i] << endl;
        }

        return 0;
}
