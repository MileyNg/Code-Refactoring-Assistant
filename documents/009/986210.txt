#include <stdio.h>
#include <stdlib.h>
int m;
struct seiseki_st
{
  int gakuseki_no;
  int tokuten;
} seiseki[40];

int compare(const void *a, const void *b)
{
  return ((struct seiseki_st *)b)->tokuten - ((struct seiseki_st *)a)->tokuten;
}
void display_result()
{
  int i;
  for(i=0;i<m;i++)
    printf("%d %d\n",seiseki[i].gakuseki_no,seiseki[i].tokuten);
} 
void clear()
{
  int i;

  for(i=0;i<40;i++)
    seiseki[i].gakuseki_no=seiseki[i].tokuten=0;
}
main()
{
  int i,current,next,next2,pin,prev,toume,score,frame,strike_f;

  while(scanf("%d",&m) && m)
    {
      clear();
      for(i=0;i<m;i++)
	{
	  scanf("%d",&seiseki[i].gakuseki_no);
	  current=next=next2=1;
	  frame=1;
	  score=0;
	  toume=1;
	  while(frame<=10)
	    {
	      scanf("%d",&pin);
	      score += current*pin;
#ifdef DEBUG
	      printf("fr=%d %d score=%d\n",frame,toume,score);
#endif
	      if(frame<10)
		{
		  strike_f=0;
		  if(pin==10 && toume==1)
		    {
		      next++;
		      next2++;
		      strike_f=-1;
		      //printf("**");
		    }
		  else if(pin+prev==10 && toume==2)
		    {
		      //printf("*@");
		      next++;
		    }
		  if(toume==2||strike_f)
		    {
		      toume=1;
		      frame++;
		    }
		  else
		    toume=2;

		}
	      else // 10th frame
		{
		  if(toume==3||(toume==2 && pin+prev<10))
		    goto END;
		  toume++;
		}
	      prev=pin;
	      current=next;
	      next=next2;
	      next2=1;
	    }
	END:
	  //printf("score=%d\n",score);
	  seiseki[i].tokuten=score;
	}
      qsort(seiseki,m,sizeof(struct seiseki_st),compare);
      display_result();
    }
  return(0);
}
