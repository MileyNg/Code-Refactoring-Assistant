#include<stdio.h>

int main(void)
{
    int H, W;
    scanf("%d %d", &H, &W);
    while( !( H == 0 && W == 0)){
        for(int i = 0; i < W; ++i){
            printf("#");
        }
        printf("\n");
        for(int i = 0; i < H - 2; ++i){
             printf("#");
             for(int j = 0; j < W -2; ++j){
                 printf(".");
             }
             printf("#");
             printf("\n");
        }
        for(int i = 0; i < W; ++i){
            printf("#");
        }
        scanf("%d %d", &H, &W);
    }
    return 0;
}
