#include <stdio.h>
#include <ctype.h>
#include <string.h>

int main(void){
	
	char w[10], t;
    char e[] = {"END_OF_TEXT"};
    int p, ep, sum;
    
	sum = 0;
    p = 0;
    ep = 0;
    
    scanf("%s", &w);
    
    
    while (1) {
        
        scanf("%c", &t);
        
        if (w[p] == toupper(t) || w[p] == tolower(t)) {
            p++;
            ep = 0;
        }
        else if (e[ep] == t){
            ep++;
            p = 0;
        }
        else if (t != ' ' || t ==' ' && (p < strlen(w) || ep < strlen(e))) {
            p = 0;
            ep = 0;
        }
        
        if(p == strlen(w)){
            sum++;
            p = 0;
            
        }
        else if(ep == strlen(e)){
            break;
        }
        
        
        
        
    }
    
    printf("%d\n", sum);
	
	return 0;
    
}
