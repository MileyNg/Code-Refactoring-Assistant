#include <iostream>
#include <ctime>
#include <cstdlib>
#include <vector>
#include <algorithm>
#include <cstdio>

using namespace std;
#define loop(i,a,b) for(int i=(a); i<int(b); i++)
#define rep(i,b) loop(i,0,b)

typedef vector<int> vi;

ostream & operator<<(ostream & os, vi const& v){
    rep(i,v.size()){
        os << v[i];
        if(i==v.size()-1) continue;
        os << ' ';
    }
    return os;
}

void solve(vi & v, int begin, int end){
    int const n = end-begin;
    if(n<=1) return;
    else if(n==2){
        if(v[begin]>v[begin+1]) swap(v[begin],v[begin+1]);
        return;
    }

    int pt[3];
    rep(i,3) pt[i] = rand()%(n-2);
    if(pt[0]>pt[1]) swap(pt[0],pt[1]);
    if(pt[0]>pt[2]) swap(pt[0],pt[2]);
    if(pt[1]>pt[2]) swap(pt[2],pt[1]);
    rep(i,3) pt[i]+=i;

    int piv=v[begin+pt[1]];

    int l=begin, r=end-1;
    while(1){
        // cout << l << " " << r << endl;
        while(v[l]<piv) l++;
        while(v[r]>piv) r--;
        if(l<r) swap(v[l++],v[r--]);
        else break;
    }
    // cout << v << endl;
    solve(v,begin,l);
    solve(v,l,end);
}

int main(){
    srand(time(0u));
    int n;
    while(cin>>n){
        vi v(n);
        rep(i,n) scanf("%d",&v[i]);
        solve(v,0,n);
        cout << v << endl;
    }
}
