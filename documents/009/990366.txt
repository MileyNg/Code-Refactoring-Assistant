#include <stdio.h>
#include <stdlib.h>
 
int main(void){
    int n, q, i, j;
    int *S, *T;
    int cnt;
 
    scanf("%d", &n);
    S = (int*)malloc(sizeof(int)*n);
    for(i = 0; i < n; i++){
        scanf("%d", S+i);
    }
     
    scanf("%d", &q);
    T = (int*)malloc(sizeof(int)*q);
    for(i = 0; i < q; i++){
        scanf("%d", T+i);
    }
 
    for(i = 0; i < n; i++){
        for(j = 0; j < q; j++){
            if(S[i] == T[j]){
                cnt++;
                break;
            }
        }
    }
 
    printf("%d\n", cnt);
 
    free(S);
    free(T);
 
    return 0;
}
