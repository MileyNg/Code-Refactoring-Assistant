#include<iostream>
using namespace std;

int main(){

  int i=0,j,k ,l;
  int exam[1000][4];
  
  for(i=0;i<1000;i++)
    {
      cin >>exam[i][0] >>exam[i][1] >>exam[i][2];
      
      if(exam[i][0] == -1 && exam[i][1] == -1 && exam[i][2] == -1)
	break;
      
    }
  
  
  
  for(k=0;k < i+1;k++)
    {
      
      exam[k][3] = exam[k][0] + exam[k][1];
      
      if(exam[k][0] == -1 || exam[k][1] == -1 ||exam[k][3] < 30){
	exam[k][3] = -1;
      }else{
	
	if(exam[k][3]< exam[k][2] && exam[k][2] != -1 && exam[k][2] > 50)
	  exam[k][3] = 50;
	
	
      }
    }
  
  for(j=0;j < i;j++){
    
    if(exam[j][3] >= 80)
      cout <<"A\n";
    else if(exam[j][3]<80 && exam[j][3]>=65)
      cout <<"B\n";
    else if(exam[j][3] <65 && exam[j][3]>=50)
      cout<<"C\n";
    else if(exam[j][3]<50 &&exam[j][3]>=30)
      cout<<"D\n";
    else 
      cout<<"F\n";
    
  }
  
  return 0;
}
