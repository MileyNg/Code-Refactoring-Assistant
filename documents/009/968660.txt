#include<stdio.h>
#define N 2000000
 
int main(){
    int n,i,A[N],B[N],C[N];
    scanf("%d",&n);
    for(i=0;i<n;i++) scanf("%d",&A[i]);
    for(i=0;i<N;i++) C[i] = 0;
    for(i=0;i<n;i++) C[A[i]]++;
    for(i=1;i<N;i++) C[i] = C[i] + C[i-1];
    for(i=n;i>=0;i--){
        B[C[A[i]]] = A[i];
        C[A[i]]--;
        }
    for(i=1;i<n;i++) printf("%d ",B[i]);
    printf("%d\n",B[n]);
    return 0;
    }
