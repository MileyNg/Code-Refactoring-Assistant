#include <iostream>
#include <cctype>
#include <string>
#include <cstdio>
using namespace std;
int main () {

  string  ch;

  getline(cin,ch);


  for(int i=0;i <ch.size() ;i++){
    if( islower(ch[i]) ){
      printf("%c",toupper(ch[i]));
    }else if( isupper(ch[i]) ){
      printf("%c",tolower(ch[i]));
    }else {
      printf("%c",ch[i]);
    }

 
  }
  cout <<"\n";
return 0;
}
