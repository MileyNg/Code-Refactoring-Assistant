#include <stdio.h>
#include <stdlib.h>

int main()
{
	int i, j, n, max;
	int *a;

	scanf("%d", &n);
	a = (int *)malloc(n * sizeof(int));
	for(i = 0; i < n; i++) scanf("%d", &a[i]);
	max = 0;
	for(i = 0; i < n; i++){
		for(j = i + 1; j < n; j++){
			if(a[j] - a[i] > max) max = a[j] - a[i];
		}
	}
	printf("%d\n", max);
	return 0;
}
