#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;

#define INF 1000000007
#define min(a,b) (a>b?b:a)

int lst[6][2] = {
	{ 2, 380},
	{ 3, 550},
	{ 5, 850},
	{10, 1520},
	{12, 1870},
	{15, 2244}
};
#define M 50
int dp[M+1];

int main() {
	int n;
	for(int i=0; i<=M; ++i) dp[i] = INF;
	dp[0] = 0;
	for(int i=1; i<=M; ++i) {
		for(int j=0; j<6; ++j) {
			if(i>=lst[j][0]) dp[i] = min(dp[i], dp[i-lst[j][0]]+lst[j][1]);
		}
		cout << dp[i] << endl;
	}
	while(cin >> n && n) cout << dp[n/100] << endl;
	return 0;
}
