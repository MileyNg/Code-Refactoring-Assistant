#include<iostream>
#include<algorithm>
using namespace std;
bool mas(int x,int y){
  if(x<0||x>1000000) return false;
  if(y<0||y>1000000) return false;
  return true;
}
int main(){
  int m,n;
  while(cin>>m,m){
    int x[m],y[m];
    for(int i=0;i<m;i++) cin >> x[i] >> y[i];
    cin >> n;
    int rx[n],ry[n];
    for(int i=0;i<n;i++) cin >> rx[i] >> ry[i];
    int a;
    for(int i=m-1;i>=0;i--){
      for(int j=0;j<i;j++){
	if(x[j]<x[j+1]){
	  a = x[j];
	  x[j] = x[j+1];
	  x[j+1] = a;
	  a = y[j];
	  y[j] = y[j+1];
	  y[j+1] = a;
	}
      }
    }
    for(int i=m-1;i>=0;i--){
      for(int j=0;j<i;j++){
	if(y[j]<y[j+1]&&x[j]==x[j+1]){
	  a = x[j];
	  x[j] = x[j+1];
	  x[j+1] = a;
	  a = y[j];
	  y[j] = y[j+1];
	  y[j+1] = a;
	}
      }
    }
    for(int i=n-1;i>=0;i--){
      for(int j=0;j<i;j++){
	if(rx[j]<rx[j+1]){
	  a = rx[j];
	  rx[j] = rx[j+1];
	  rx[j+1] = a;
	  a = ry[j];
	  ry[j] = ry[j+1];
	  ry[j+1] = a;
	}
      }
    }
    //for(int i=0;i<m;i++) cout << x[i] << ' ' << y[i] << endl;
    for(int i=n-1;i>=0;i--){
      for(int j=0;j<i;j++){
	if(ry[j]<ry[j+1]&&rx[j]==rx[j+1]){
	  a = rx[j];
	  rx[j] = rx[j+1];
	  rx[j+1] = a;
	  a = ry[j];
	  ry[j] = ry[j+1];
	  ry[j+1] = a;
	}
      }
    }
    for(int i=0;i<n;i++){
      int dx = rx[i] - x[0];
      int dy = ry[i] - y[0];
      int j = 1;
      int k = i+1;
      bool ok = false;
      while(k<n){
	if(!mas(dx+x[j],dy+y[j])) break;
	if(dx+x[j]!=rx[k]||dy+y[j]!=ry[k]) k++;
	if(dx+x[j]==rx[k]||dy+y[j]==ry[k]){
	  j++;
	  k++;
	}
	if(j>=m){
	  ok = true;
	  break;
	}
      }
      if(ok){
	cout << dx << ' ' << dy << endl;
	break;
      }
    }
  }
  return 0;
}
