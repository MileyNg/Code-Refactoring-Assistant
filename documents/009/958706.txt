#include <stdio.h>
int main(){
	int tate=0, yoko=0,a=0,b=0,d=0;
	int c[100][100]={0};
	int f[100]={0};
	scanf("%d %d", &tate, &yoko);
	if(tate<1 || yoko>100 )return 1;
	for(a=0;a<tate;a++){
		for(b=0;b<yoko;b++){
			
			scanf("%d", &c[a][b]);
		}
	}
	for(a=0;a<tate;a++){
		d=0;
		for(b=0;b<yoko;b++){
			printf("%d ", c[a][b]);
			d+=c[a][b];
			f[b]+=c[a][b];
		}printf("%d\n", d);
	}
	d=0;
	for(a=0;a<yoko;a++){
		d+=f[a];
		printf("%d ",f[a]);
	}
	printf("%d\n",d);

	
	scanf("%d", &a);

	return 0;
}
