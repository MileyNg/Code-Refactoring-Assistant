#include <stdio.h>
int n,A[20],M[200];
 
int res(int i,int m)
{
  if(A[i]==m) return 1;
  if(i==n) return 0;
  else
    {
      if(res(i+1,m)||res(i+1,m-A[i]))
    return 1;
    }
  return 0;
}
 
int main(){
  int i,k;
   
  scanf("%d",&n);
  for(i=0;i<n;i++)
    {
      scanf("%d",&A[i]);
    }
  scanf("%d",&k);
  for(i=0;i<k;i++)
    {
      scanf("%d",&M[i]);
    }
 
  for(i=0;i<k;i++)
    {
      if(res(0,M[i]))
    printf("yes\n");
       
      else
    printf("no\n");
    }
return 0;
}
