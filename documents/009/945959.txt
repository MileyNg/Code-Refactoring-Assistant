//global-----------------------------------------------------------------------------------------------------------------------
int n;
int a, b, c;

//function---------------------------------------------------------------------------------------------------------------------
int main() {
	scanf( "%d", &n );
	rep( i, n ) {
		scanf( "%d %d %d", &a, &b, &c);
		int ans = 0;
		while ( true ) {
			if ( a >= 1 && b >= 1 && c >= 1 ) {
				ans++;
				a--;
				b--;
				c--;
				continue;
			}
			if ( a >= 2 && b >= 1 ) {
				ans++;
				a -= 2;
				b--;
				continue;
			}
			if ( a >= 3 ) {
				ans++;
				a -= 3;
				continue;
			}
			break;
		}
		printf( "%d\n", ans );
	}
	return 0;
}
