#include<iostream>
#include<stdio.h>
using namespace std;
int n;

int dy[]={-1,1};
int dx[]={1,-1};
int t[10][10];

bool inField(int y,int x){
  if (0<=y&&y<n&&0<=x&&x<n)return true;
  return false;
}

int main(){
  int Tc=1;
  while(cin>>n&&n){
    int y=0,x=0,cnt=1;
    
    for(int i=0;i<n*2-1;i++){  
      t[y][x]=cnt++;
      while(inField(y+dy[i%2],x+dx[i%2])){
	y+=dy[i%2];
	x+=dx[i%2];
	t[y][x]=cnt++;
      }
      if(i<n-1){
	if(i%2==0)x++;
	else y++;
      }else{
	if(i%2==1)x++;
	else y++;
      }
    }
  
    t[n-1][n-1]=n*n;
    
    cout<<"Case "<<Tc++<<":"<<endl;
    for(int i=0;i<n;i++){
      for(int j=0;j<n;j++){
	printf("%3d",t[i][j]);
      }
      cout<<endl;
    }
  }
  return 0;
}
