#include <bits/stdc++.h>

using namespace std;

#define MAX 101
#define INF INT_MAX/4

int G[MAX][MAX], N;

void make(){
  for(int k = 0 ; k < N ; k++){
    for(int i = 0 ; i < N ; i++){
      for(int j = 0 ; j < N ; j++){
	G[i][j] = min(G[i][j],G[i][k]+G[k][j]);
      }
    }
  }
}

int main(){
  int K, num, a, b, c;

  while(scanf("%d%d",&N,&K), (N | K)){
    fill(G[0], G[N], INF);
    while(K--){
      scanf("%d",&num);
      if(num){
	cin >> a >> b >> c;
	a--, b--;
	G[a][b] = G[b][a] = c;
	make();
      }else{
	cin >> a >> b;
	if(G[a-1][b-1] != INF){
	  printf("%d\n",G[a-1][b-1]);
	}else{
	  puts("-1");
	}
      }
    }
  }

  return 0;
}
