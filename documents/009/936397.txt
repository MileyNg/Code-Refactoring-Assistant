#include <stdio.h>
#include <string.h>
#include <stdlib.h>

int main(){
  char input[1010];
  int b=0;

  while(1){
     scanf("%s", input);
     if(input[0] != '0' && strlen(input) == 1) break;
     int i=0;
     for(i=0 ; i < strlen(input); i++){
       int x = input[i] - '0';
       b = b + x;
     }
       
    printf("%d\n",b); 
    b=0;
    } 
      return 0;
  }
