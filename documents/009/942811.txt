#include <stdio.h>

int main(int argc, const char * argv[])
{

    int a[1000],b[1000],n;
    char op[1000];
    
    n=0;
    
    scanf("%d %c %d",&a[n] ,&op[n] ,&b[n]);
    
  //  printf("%d %c %d\n",a[n],op[n],b[n]);
    
    while(op[n]!='?'){
        n++;
        
        scanf("%d %c %d",&a[n] ,&op[n] ,&b[n]);
        
    //    printf("%d %c %d\n",a[n],op[n],b[n]);

    }
    for(int i=0;i<n;i++){
        
        if(op[i] == '+'){
            printf("%d\n",a[i]+b[i]);
        }else if(op[i]=='-'){
            printf("%d\n",a[i]-b[i]);
        }else if(op[i]=='*'){
            printf("%d\n",a[i]*b[i]);
        }else if(op[i]=='/'){
            printf("%d\n",a[i]/b[i]);
        }else{
            break;
        }
        
        
    }
    
    
    
    return 0;
}
