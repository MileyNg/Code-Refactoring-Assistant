{
	int n, m;
	vector<int> log;
	while(true){
		cin >> n >> m;
		if(!n && !m){break;}
		vector<int> log(n+m);
		rep(i, n+m){
			cin >> log[i];
		}
		SORT(log);

		int ans=0;
		int tmp=0;
		if(n+m == 1){ans=log[0];}

			if(tmp > ans){ ans=tmp; }
			//DUMP(ans);
		}
		cout << ans << endl;
