#include <stdio.h>

int main(void)
{
	int num, cnt = 0;
	int i, j, k, l;

	while( scanf("%d", &num) ){
		for(i = 0; i < 10; i++){
			for(j = 0; j < 10; j++){
				for(k = 0; k < 10; k++){
					for(l = 0; l < 10; l++){
						if(i + j + k + l == num){
							cnt++;
						}
					}
				}
			}
		}
		printf("%d\n", cnt);
		cnt = 0;
	}

	return 0;
}
