#include<stdio.h>

int main(void) {
	int h[10];
	int i,j;
	
	for(i=0; i<10; i++) 
		scanf("%d", &h[i]);
	
	for(i=0;i<10;i++) {
		for(j=i; j<10; j++) {
			if(h[i] < h[j]) {
				int swap =h[i];
				h[i] = h[j];
				h[j] = swap;
			}
		}
	}
	
	for(i=0; i<3; i++)
		printf("%d\n", h[i]);	
	
	
	return 0;
}
