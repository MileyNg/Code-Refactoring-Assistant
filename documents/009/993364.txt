#include<iostream>
#include<string>
using namespace std;

int key[1000];
char alpha[100];
int main(){
  int n,j;
  string s;
  int k=0;
  for(int i=0;i<52;i++){
    if(i>=26)
      alpha[i] = 'A'+i+6;
    else  alpha[i] = 'A'+i;
  }
  //cout<<(int)'A'<<(int)'a'<<endl;
  while(cin>>n,n){
    string ans;
    k = 0;
    j = 0;
    for(int i = 0;i<n;i++)cin>>key[i];
    cin>>s;
    // cout<<s<<endl;
    for(int i = 0;i<s.length();i++){
      if(s[i]-key[j]>'Z')
	s[i] -= 6;
      if(s[i]-key[j]-'A'>=0)
	ans[i] = alpha[s[i] - key[j]-'A'];
      else{
	ans[i] = alpha[s[i]-key[j]-'A'+'z'-'A'-5];
      }
      if(j == n-1) j = 0;
      else j++;
    }
    //    cout<<alpha[51];
    for(int i=0;i<s.length();i++)
      cout<<(char)ans[i];
    cout<<endl;
  }
  return 0;
}
