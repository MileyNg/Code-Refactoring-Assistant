#include <iostream>
using namespace std;

/** Problem0048 : Class **/
int main()
{
	double input;
	string output;
	while (cin >> input) {
		if (input <= 48.0)
			output = "light fly";
		else if (input <= 51.0)
			output = "fly";
		else if (input <= 54.0)
			output = "bantam";
		else if (input <= 57.0)
			output = "feather";
		else if (input <= 60.0)
			output = "light";
		else if (input <= 64.0)
			output = "light welter";
		else if (input <= 69.0)
			output = "welter";
		else if (input <= 75.0)
			output = "light middle";
		else if (input <= 81.0)
			output = "middle";
		else if (input <= 91.0)
			output = "light heavy";
		else
			output = "heavy";
		
		cout << output << endl;
	}
	return 0;
}
