#include <stdio.h>
#define N 200005


int main(void){
  int n,r[N],i,j,d,c,max,min;
  scanf("%d",&n);
  for(i=0;i<n;i++){
    scanf("%d",&r[i]);
  }
  max=1;
  min=1000000000;
  d=r[1]-r[0];
  for(i=1;i<n;i++){
    if(r[i]>max){
      max=r[i];
      for(j=0;j<i;j++){
	if(r[j]<min){
	  min=r[j];
	}
      }
      c=max-min;
      if(c>d){
	d=c;
      }
      max=1;
      min=1000000000;
    }
  }
  printf("%d\n",d);

  return 0;
}
