#include <iostream>
#include <stdio.h>
#include <cstdlib>
#include <vector>
#include <string>
#include <queue>
#include <map>
#include <algorithm>
#define INF 10000000
using namespace std;
typedef pair<int,int> P;


struct node{
  bool marked;
  int parent;
  int num;
};

vector<node> T;
int main(){
  while(1){
    int n,q;
    scanf("%d%d",&n,&q);
    if(n==0&&q==0){
      return 0;
    }

    node root;
    root.marked = true;
    root.parent = 0;
    root.num = 1;
    T.push_back(root);
    for(int i=2;i<n+1;i++){
      int parent;
      cin >> parent;
      node child;
      T.push_back(child);
      T[i-1].marked = false;
      T[i-1].num = i;
      T[i-1].parent = parent-1;
    }

    int sum = 0;
    for(int i=0;i<q;i++){
      char q;
      int v;
      cin >> q >> v;
      if(q=='M'){
        T[v-1].marked = true;
      }
      if(q=='Q'){
        node parent = T[T[v-1].parent];
        while(!parent.marked){
          parent = T[parent.parent];
        }
        sum += parent.num;
      }
    }
    cout << sum << endl;

  }
}
