#include <stdio.h>
#include <string.h>
main(){
  int i,j=0;
  char W[100],T[10000];
  scanf("%s",W);
  for(i=0;i<100;i++){
    W[i]=toupper(W[i]);
  }
  while(1){
    scanf("%s",T);
    if(strcmp(T,"END_OF_TEXT")==0){
      break;
    }
    for(i=0;i<10000;i++){
      T[i]=toupper(T[i]);
    }
    if(strcmp(W,T)==0){
      j=j+1;
    }
  }
  printf("%d\n",j);
  return 0;
}
