#include<stdio.h>
int main(){
    int a;
    while(1){
            a=0;
            scanf("%d",&a);
            if(a>10 && a<100){
                printf("%d \n",a%10+(a/10)%10);
            }
            else if(a<10 && a>0){
                printf("%d",a);
            }
            else if(a>100 && a<1000){
                printf("%d \n",(a%10)+(a/10)%10+((a/10)/10)%10);
            }
            else if(a==1000){
                printf("%d",1);
            }
            else if(a==0){
                break;
            }
        }
        return 0;
        }
