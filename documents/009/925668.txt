#include<iostream>
using namespace std ;

int map[100][100] = {} ;
int dx[] = { -1,1 } ;
int dy[] = { 1,-1 } ;
int n , m , p=0 ;

void move( int count , int ny , int  nx ){
	
	map[ny][nx] = count ;
	count++;
	
	if( ny == n-1 && nx == n-1 ){}
	else {
		if( ny+dy[m%2] >= 0 && ny+dy[m%2] <= n-1 &&
			nx+dx[m%2] >= 0 && nx+dx[m%2] <= n-1 )
				move( count , ny+dy[m%2] , nx+dx[m%2] ) ;
		else {
			m++;
			if( ny == 0 || ny == n-1 ){
				if( nx == n-1 ) move( count , ny+1 , nx ) ;
				else move( count , ny , nx+1 ) ;
			}
			else if( nx == 0 || nx == n-1 )
				move( count , ny+1 , nx ) ;
		}
	}
}

int main(){
	
	while(1){
		cin >> n ;
		if( n == 0 ) break ;
		
		m = 1 ;
		p++ ;
		
		move( 1 , 0 , 0 ) ;
		
		cout << "Case " << p << ':' << endl ;
		for( int i=0 ; i<n ; i++ ){
			for( int j=0 ; j<n ; j++ ){
				if( map[i][j] < 10 ) cout << ' ' ;
				if( map[i][j] < 100 ) cout << ' ' ;
				cout << map[i][j] ;
			}
			cout << endl ;
		}
	}
	
return 0 ;
}
