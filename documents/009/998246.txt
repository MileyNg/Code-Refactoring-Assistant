#include<iostream>
using namespace std;

int main(){
  int m,f,r,sum;

  while(1){
    sum = 0;
    cin >> m >> f >> r ;
    sum = m+f;
    if ( m == -1 && f == -1 && r == -1)break;
    else {
      if ( m <= 0 || f <= 0 || sum < 30){
	cout << "F" << endl;
      }
      else if ( 30 <= sum && sum < 50){
	if ( r >= 50){
	  cout << "C" <<endl;
	}
	else {
	  cout << "D" << endl;
	}
      }
      else if ( 50 <= sum && sum < 65){
	cout << "C" <<endl;
      }
      else if ( 65 <= sum && sum < 80){
	cout << "B" <<endl;
      }
      else if ( 80 <= sum ){
	cout << "A" <<endl;
      }
    }
  }

  return 0;
}
