#include <stdio.h>
#define N 100000000
void Merge(int,int,int);
void MergeSort(int,int);
int n,l=0,S[N];
int main()
{
  int i,n;
  
  scanf("%d",&n);
  for(i=0;i<n;i++){
    scanf("%d",&S[i]);
  }
  MergeSort(0,n);  
  
  for(i=0;i<n;i++){
    printf("%d",S[i]);
   if(i!= n-1)printf(" ");
  };
  printf("n%d\n",l);
  return 0;
}

void Merge(int left, int mid, int right){
  int n1,n2,i,k,j;
  n1 = mid - left;
  n2 = right - mid;
  int L[n1+1],R[n2+1];
  for (i = 0 ;i<n1;i++){
    L[i] = S[left + i];
  }
  for (i = 0 ;i<n2 ;i++){
    R[i] = S[mid + i];
  }
  L[n1]=N+1;
  R[n2]=N+1;
  i = 0;
  j = 0;
  
  for (k = left;k< right;k++){
    if (L[i] <= R[j])
      {
	S[k] = L[i];
	i = i + 1;
      }
    else {
      S[k] = R[j];
      j = j + 1;
    }
    l++;
  }
}

void MergeSort(int left, int right){
  int mid;
  if (left+1<right){
    mid = (left+right)/2;
    MergeSort(left, mid);
    MergeSort(mid, right);
    Merge(left,mid,right);
  }  
}
