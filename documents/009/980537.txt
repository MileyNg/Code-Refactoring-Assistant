#include <stdio.h>
main(){
  int S[1000],T[1000],C[1111],n,q,i,j,ans=0,ccc;
  scanf("%d",&n);
  for(i=0;i<n;i++){
      scanf("%d",&S[i]);
  }
  scanf("%d",&q);
  for(i=0;i<q;i++){
    scanf("%d",&T[i]);
  }
  for(i=0;i<q;i++){
    ccc=0;
    for(j=0;j<n;j++){
      if(S[i]==T[j]){
	if(ccc!=0)break;
	  ccc=1;
	ans=ans+1;
      }
    }
  }
  printf("%d\n",ans);
  return 0;
}
