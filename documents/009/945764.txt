#define M 1000000
#define NTL (-1)
#define L 14

char H[M][L]; /* Hash Table */
int count=0,k=0;
long long search[M];

int getChar(char ch){
  if ( ch == 'A') return 1;
  else if ( ch == 'C') return 2;
  else if ( ch == 'G') return 3;
  else if ( ch == 'T') return 4;
}

/* convert a string into an integer value */
long long getKey(char str[]){
  long long sum = 0, p = 1, i;
  for ( i = 0; i < strlen(str); i++ ){
    sum += p*(getChar(str[i]));
    p *= 5;
  }
  return sum;
}

    //   if( strcmp(H[j],str)==0) return 1;
    if(search[j]==getKey(str) ) return 1;     
     else i++;
    
  }
  return NTL;
  
}

int insert(char str[]){
  while(k!=M){

      strcpy(H[k],str);

       search[k]=getKey(str);
       return k;
}


    else k++;
  }
  return NTL;
}
