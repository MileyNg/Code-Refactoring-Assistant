#include <stdio.h>

int main(){

  int n,m,i,a[10000],b[10000],k,j;
  m=0;
  
  scanf("%d",&n);

  for(i=0;i<n;i++){
    scanf("%d",&a[i]);
  }
  scanf("%d",&k);
  for(i=0;i<k;i++){
    scanf("%d",&b[k]);
  }


  for(i=0;i<n;i++){
    for(j=0;j<k;j++){
      if(a[i]==b[j]){
	m++;
      }
    }
  }
  printf("%d\n",m);
  return 0;
}
