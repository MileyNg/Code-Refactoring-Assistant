#include <stdio.h>

int main(){

int office[4][3][10];
int n,b,f,r,v;

for(b=0;b<4;b++){
 for(f=0;f<3;f++){
  for(r=0;r<10;r++){
   office[b-1][f-1][r-1]=0;
  }
 }
}


scanf("%d",&n);

while(n--){

scanf("%d%d%d%d",&b,&f,&r,&v);
office[b-1][f-1][r-1] += v;

}

for(b=0;b<4;b++){
 for(f=0;f<3;f++){
  for(r=0;r<10;r++){
   printf(" %d",office[b-1][f-1][r-1]);
  }
 }
 printf("\n");
 printf("####################\n");
}


return 0;
}
