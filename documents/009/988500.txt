int dfs(int n){
  if(n == 9) return judge();
  if(a[n]) return dfs(n+1);
  int res = 0;
  REP(i, 1, 11){
    if(used[i]) continue;
    a[n] = i;
    used[i] = 1;
    res += dfs(n+1);
    a[n] = 0;
    used[i] = 0;
  }
  return res;
}
