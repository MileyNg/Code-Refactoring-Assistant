#include <iostream>
#include <algorithm>
using namespace std;
int
main ()
{
  int h, w, w2;
  while (true)
    {
      cin >> h >> w;
      if (h == 0 && w == 0)
	exit (0);
      w2 = w;
      while (h-- > 0)
	{
	  w = w2;
	  while (w-- > 0)
	    cout << "#";
	  cout << endl;
	}
          cout << endl;
    }
}
