#include <iostream>
#include <algorithm>
using namespace std;

int n,x,com;
int main() {
	while(true) {
		cin >> n >> x;
		if(n == 0 && x == 0) break;
		int m;
		m = min(n,x-3);
		for(int i = m; i > x/3; --i) {
			int sum = x-i;
			int num = i - sum/2 -1;
			com += num;
		}
		cout << com << endl;
	}
}
