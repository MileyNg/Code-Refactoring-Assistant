int main(){
  int z, x, y;

  while(scanf("%d", &z) && z){
    z = pow((double)z, 3);
    int res = 0, tx, ty;
    for(x = 1; x < 1100; x++){
      for(y = 1; y < 1100; y++){
	tx = pow((double)x, 3); ty = pow((double)y, 3);
