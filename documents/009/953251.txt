#define _CRT_SECURE_NO_WARNINGS
#include<stdio.h>
#define MAX 2000
int A[MAX], M[MAX];
int a,b,i,j,k,f=0;

int solve(int p, int t){
	if (t - A[p] == 0){
		return 1;
	}
	if (p+1>a){
		return 0;
	}
	if (solve(p + 1, t - A[p]) || solve(p + 1, t) ){
		return 1;
	}
	return 0;
}


int main(){
	

	scanf("%d",&a);
	for (i = 0; i < a;i++){
		scanf("%d",&A[i]);
	}
	scanf("%d",&b);

	for (i = 0; i < b;i++){
		scanf("%d",&M[i]);
	}

	for (i = 0; i < b; i++){
		if (solve(0, M[i])){
			printf("yes\n");
		}else{
			printf("no\n");
		}
	}

	return 0;

}
