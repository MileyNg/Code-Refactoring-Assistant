
#include<iostream>
using namespace std;

int main(){
int a,b;
char op;
int rst;
cin >> a >> op >> b ;

if((op !='+') && (op !='-') && (op != '*') && (op != '/') && (op != '?')){
     cout << "error!!" << "\n" ;
}

else{
     switch(op){
          case '+' :
          rst = a+b ;
          cout << rst << "\n" ;
          break ;

          case '-' :
          rst = a-b ;
          cout << rst << "\n" ;
          break ;


          case '*' :
          rst = a*b ;
          cout << rst << "\n" ;
          break ;

          case '/':
          if(b!=0){
          rst = a/b ;
          cout << rst << "\n" ;
          break ;
          }
          else{
          cout << "error!!"<< "\n" ;
          break ;
          }

          case '?' :
          break ;

     }
}
return 0 ;
}
