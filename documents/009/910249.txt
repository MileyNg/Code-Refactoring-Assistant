#include<stdio.h>
#define n 1000000000

int main(){
  int N;
  int R[n];
  int i,j;
  int a,ans;
  ans=0;

  scanf("%d",&N);
  for(i=0;i<N;i++){
    scanf("%d",&R[i]);
  }

  for(i=0;i<N-1;i++){
    for(j=i+1;j<N;j++){
      a=R[j]-R[i];
      if(a>ans)ans=a;
    }
  }
  printf("%d\n",ans);

  return 0;
}
