#include <stdio.h>
#include <string.h>

int main(void)
{
	char word[501][33];
	int i, j;
	int c;
	int flag;
	char str[32];
	int  count[501];
	int maxC = 500;;
	char maxL[33] = "\0";
	int n = 8;

	count[500] = 0;
	c = 0;
	while (n--){
		scanf("%s", str);
		flag = 0;
		for (i = 0; i < c; i++){
			if (strcmp(word[i], str) == 0){
				count[i]++;
				if (count[maxC] < count[i]){
					maxC = i;
				}
				flag++;
			}
		}

		if (!flag){
			strcpy(word[i], str);
			c++;
		}

		if (strlen(maxL) < strlen(str)){
			strcpy(maxL, str);
		}
	}

	printf("%s %s\n", word[maxC], maxL);

	return (0);
}
