      cin>>str[i];
      for(int j=0;j<i;j++){
	t[i][j]=count(str[i],str[j],i,j);
	t[j][i]=count(str[j],str[i],j,i);
	//cout<<str[i]<<"==="<<str[j]<<endl;
	//cout<<u[i][j]<<" --- "<<u[j][i]<<endl;
      }
    }
    ans="";
    for(int i=0;i<n;i++)v[i]=i;
    do{
      tmp=str[v[0]];
      for(int i=1;i<n;i++){
	tmp+=u[v[i-1]][v[i]];
      }
      if(ans.size()==0||tmp.size()<ans.size())ans=tmp;
      else if(tmp.size()==ans.size()&&tmp<ans)ans=tmp;
    }while(next_permutation(v,v+n));
    cout<<ans<<endl;
  }
  return 0;
}

int count(string& a,string& b,int I,int J){
  int i,j,c=a.size(),d=b.size();
  //cout<<a<<' '<<b<<endl;
  bool flg;
  for(i=0;i<c;i++){
    flg=true;
    for(j=0;(j<d&&i+j<c);j++){
      if(a[i+j]!=b[j])flg=false;
    }
    if(flg){
      if(j>0)u[I][J]=b.substr(j);
      else u[I][J]="";
      return u[I][J].size();
    }
  }
  u[I][J]=b;
  return d;
}
