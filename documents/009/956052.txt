#include<stdio.h>
#include<math.h>
int main(){
    int n[1000],s,i,b;
    double a=0,c,d=0,e;
    scanf("%d",&b);
    if(b==0)return 0;
    for(i=0;i<b;i++){
        scanf("%d",&n[i]);
        a+=n[i];
    }
    c=a/(double)b;
    for(i=0;i<b;i++){
        d+=(n[i]-c)*(n[i]-c);
    }
    e=d/(double)b;
    printf("%.8f\n",sqrt(e));
    return 0;
}
