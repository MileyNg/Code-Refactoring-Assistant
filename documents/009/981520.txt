#include <stdio.h>

int main(){

	char n[1001] = {};
	int m[1001] = {}, x = 0, k;
	while (1){
		scanf("%s", n);
		if (n[0] == '0'){ break; }
		for (k = 0; k < 1000; k++)	{
			m[k] = n[k] - '0';
		}
		for (k = 0; k < 1000; k++){
			if (m[k] > 0){
				x = x + m[k];
			}
		}
		printf("%d\n", x); x = 0;
		for (k = 0; k < 1000; k++){ n[k] = 0, m[k] = 0;}
	}
	return 0;
}
