#include <stdio.h>

#define N 100002

int Partition(int A[],int p,int r);

int i,j;

main(){
  int n,A[N],c;
  scanf("%d",&n);
  for(i = 0; i < n; i++){
    scanf("%d",&A[i]);
  }
  c = Partition(A,0,n-1);
  for(i = 0; i < n-1; i++){
    if(i == c){
      printf("[%d] ",A[i]);
    }else{
      printf("%d ",A[i]);
    }
  }
  printf("%d\n",A[n-1]);
  return 0;
}
  
int Partition(int A[],int p,int r){
  int x,y,z;
  x = A[r];
  i = p-1;
  for(j = p; j < r; j++){
    if(A[j] <= x){
      i++;
      y = A[i];
      A[i] = A[j];
      A[j] = y;
    }
  }
  z = A[i+1];
  A[i+1] = A[r];
  A[r] = z;
  return i+1;
}
