#include <iostream>
#include <cmath>
#include <algorithm>

using namespace std;

int main() {
	int n, m;
	int taro[100];
	int hanako[100];
	while (cin >> n >> m) {
		if (n == 0 && m == 0) {
			break;
		}
		int taros = 0;
		int hanakos = 0;
		int sa;
		for (int i = 0; i < n; i++) {
			cin >> taro[i];
			taros += taro[i];
		}
		for (int i = 0; i < m; i++) {
			cin >> hanako[i];
			hanakos += hanako[i];
		}
		sa = abs(taros-hanakos);
		if (sa % 2 != 0) {
			cout << -1 << endl;
		} else {
			for (int i = n-1; i > 0; i--) {
				for (int j = 0; j < i; j++) {
					if (taro[j] > taro[j+1]) {
						swap(taro[j], taro[j+1]);
					}
				}
			}

			for (int i = m-1; i > 0; i--) {
				for (int j = 0; j < i; j++) {
					if (hanako[j] > hanako[j+1]) {
						swap(hanako[j], hanako[j+1]);
					}
				}
			}
			bool hantei = false;
			int p, q;
			for (p = 0; p < n; p++) {
				for (q = 0; q < m; q++) {
					if (abs(taro[p]-hanako[q])*2 == sa) {
						hantei = true;
						break;
					} //else if (abs(taro[p]-hanako[q])*2 > sa) {
						//break;
					//}
				}
				if (hantei) {
					break;
				}
			}
			if (hantei) {
				cout << taro[p] << ' ' << hanako[q] << endl;
			} else {
				cout << -1 << endl;
			}
		}
	}
	return 0;
}
