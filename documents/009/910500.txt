#include<stdio.h>
#include<math.h>
#define MAX 10000
int main()
{
  int h,li,be,b=0; /*li,beはカウンタ変数*/
  int he[MAX];
  scanf("%d",&h);
  for(li=0;li<h;li++)
    {
      scanf("%d",&he[li]);
    }
  for(li=0;li<h;li++)
    {
      for(be=2;be<=sqrt(he[li]);be++)
	{
	  if(he[li] % be == 0)
	    {
	      b++;
	      break;
	    }
	}
    }
  printf("%d\n",h-b);
  return 0;
}
