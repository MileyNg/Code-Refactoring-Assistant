int main(int argc, const char * argv[])
{
    int n;
    int i;
    scanf("%d",&n);
    
    int a[n];
    
    for(i=0; i< n; i++) {
        scanf("%d",&a[n-i-1]);
    }
    
    for (i=0; i < n; i++) {
        if (i == n) {
            printf("%d\n",a[i]);
        }else {
            printf("%d ",a[i]);
        }
    }
    return 0;
}
