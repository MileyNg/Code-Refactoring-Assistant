typedef long long ll;

const int iINF = 2147483647;
const ll lINF = 9223372036854775807;

template <class T> inline void dbg(T t) { cout << t << endl; }

int a[5050];

int main() {
	int n;
	
	while (cin>>n) {
		if (n==0) break;
		for (int i=0; i<n; i++) cin >> a[i];
		
		int ans = -iINF;
		for (int i=0; i<n; i++) {
			int sum = 0;
			for (int j=i; j<n; j++) {
				sum += a[j];
				ans = max(ans, sum);
			}
		}
		cout << ans << endl;
	}
	
	return 0;
}
