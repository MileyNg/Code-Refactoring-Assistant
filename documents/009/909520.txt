#include <iostream>
using namespace std;

int main() {
        int card[4][13]={0};
        int n,x,y;
        char c;

        cin >> n;
        for(int i=0;i<n;i++){
             cin >> c >> y;
             if(c=='S') x=0;
             if(c=='H') x=1;
             if(c=='C') x=2;
             if(c=='D') x=3;
             card[x][y]=1;
             }
             for(int j=0;j<4;j++){
                  for(int k=0;k<13;k++){
                      if(card[j][k]==0){
                         if(j==0)
                         cout<<"S "<<k+1<<endl;
                         if(j==1)
                         cout<<"H "<<k+1<<endl;
                         if(j==2)
                         cout<<"C "<<k+1<<endl;
                         if(j==3)
                         cout<<"D "<<k+1<<endl;
                      }
                    }
               }
	return 0;
}
