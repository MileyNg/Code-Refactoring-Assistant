#include<iostream>
using namespace std;
double  q1,b,q2,ans;
int A,B;
double c1,c2;
int L,R,M;

void calcB(){
  int money = b-A*c1;
  if(money>=0)B=money/c2;
  else B=-99999;
}

  
bool check(int x){
  A=x;
  calcB();
  if(B>=0&&A+B>=q1)return true;
  else return false;
}

int main(){
  while(cin>>q1){
    ans=-1;
    if(q1==0)break;
    cin>>b>>c1>>c2>>q2;

    L=1;
    R=q2;
    M=(L+R)/2;

    while(L<R){
      if(check(M)&&!check(M+1)){
	break;
      }else if(check(M)){
	L=M+1;
	M=(L+R)/2;
      }else{
	R=M;
	M=(L+R)/2;
      }
	
    }


    if(check(M)){
      A=M;
      calcB();
      cout<<M<<' '<<B<<endl;
    } else cout<<"NA"<<endl;
    
  }
  return 0;
}
