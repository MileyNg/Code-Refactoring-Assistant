#include <stdio.h>

int main(){
	char op;
	int a, b;
	for (op = 1; op > 0;op++){
		scanf("%d %c %d", &a, &op, &b);
		if (op == '?'){ printf("\n"); break; }
		if (op == '+'){
			printf("%d\n", a + b);
		}
		else if (op == '-'){
			printf("%d\n", a - b);
		}
		else if (op == '*'){
			printf("%d\n", a * b);
		}
		else if (op == '/'){
			printf("%d\n", a / b);
		}
	}
	return 0;
}
