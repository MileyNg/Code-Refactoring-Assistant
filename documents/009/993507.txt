#include<iostream>
#include<queue>
#include<map>
#include<algorithm>
#include<vector>
 
#define fs first
#define sc second
using namespace std;
typedef pair<int,int> pii;
typedef pair<int,pii> tri;

const int INF = 1e8;
int num[110][110];

int main(){
  int n,m,l;
  while(cin >> n >> m >> l, n){
    //テ」ツつーテ」ツδゥテ」ツδ陛」ツ?ョテゥツ堋」テヲツ篠・テ」ツδェテ」ツつケテ」ツδ暗ィツ。ツィテァツ渉セ
    //g[i].fs: テヲツ閉オテ」ツ?ョテヲツ閉ー, g[i].sc.fs: ティツ。ツ古」ツ?催・ツ?暗」ツ?ョテ・ツョツソテ・ツ?エ, g[i].sc.sc: テゥツ閉キテ」ツ??ティツュツキティツ。ツ崚」ツつ津ゥツ崢?」ツ??ヲツ鳴凖ゥツ??
    vector<tri> g[110];
 
    for(int i=0;i<m;i++){
      int a,b,d,e;
      cin >> a >> b >> d >> e;
      g[a].push_back(tri(e,pii(b,d)));
      g[b].push_back(tri(e,pii(a,d)));
    }
 
    //テ・ツ按敕ヲツ慊淌・ツ個?
    for(int i=1;i<=n;i++){
      for(int j=0;j<=l;j++)num[i][j] = INF;
    }    
    num[1][l] = 0;

    //テヲツ伉?ゥツ??」ツ?ョテ・ツ?ェテ・ツ?暗・ツコツヲテ、ツサツ佚」ツ?催」ツつュテ」ツδ・テ」ツδシ
    priority_queue<tri ,vector<tri> ,greater<tri> > q;
    q.push(tri(0,pii(1,l)));
 
    //テ」ツδ?」ツつ、テ」ツつッテ」ツつケテ」ツδ暗」ツδゥ
    while(q.size()){
      tri p = q.top();q.pop();
      //p.fs: テヲツ閉オテ」ツ?ョテァツキツ湘ヲツ閉ー, p.sc.fs: テァツ渉セテ・ツ慊ィテ・ツ慊ー, p.sc.sc: テヲツョツ凝ゥツ??
      int enemy = p.fs, pos = p.sc.fs, money = p.sc.sc;
 
      for(int i=0;i<(int)g[pos].size();i++){
	int nxt = g[pos][i].sc.fs, cost = g[pos][i].sc.sc;
	if(cost <= money){
	  int rem = money - cost;
	  if(num[nxt][rem] > enemy){
	    num[nxt][rem] = enemy;
	    q.push(tri(enemy,pii(nxt,rem)));
	  }
	}
	if(num[nxt][money] > enemy + g[pos][i].fs){
	  num[nxt][money] = enemy + g[pos][i].fs;
	  q.push(tri(enemy+g[pos][i].fs, pii(nxt,money)));
	}
      }
    }
   
    //テ」ツつエテ」ツδシテ」ツδォテ」ツ?ァテ」ツ?ョテヲツョツ凝ゥツ?妥」ツ?ッテァツクツ崚」ツつ甘」ツ?古」ツ?ェテ」ツ??」ツ?ョテ」ツ?ァテ」ツ??」ツ?凖」ツ?ケテ」ツ?ヲテ」ツ?ョテ、ツクツュテ」ツ?凝」ツつ嘉ヲツ慊?・ツーツ湘・ツ?、テ」ツつ津ヲツアツづ」ツつ?」ツつ?
    int ans = INF;
    for(int i=0;i<=l;i++)ans = min(ans,num[n][i]);
    cout << ans << endl;
  }
}
