#include<iostream>

using namespace std;

#define MMAX 1050
#define NMAX 1050

int main(){

	int n,m,a;
	int h,p,q;
	int amida[MMAX+2][NMAX+2];

	while(1){
		cin>>n>>m>>a;
		if(n==0&&m==0&&a==0) break;

		for(int i=0;i<=m+5;i++){
			for(int j=0;j<=n+5;j++){
				amida[i][j]=0;
			}
		}

		for(int i=0;i<m;i++){
			cin>>h>>q>>p;
			amida[h][q]=1;
		}

		for(int i=n-1;i>=1;i--){
			if(amida[i][a]==1) a+=1;
			else if(amida[i][a-1]==1) a-=1;
		}

		cout<<a<<endl;
	}

}
