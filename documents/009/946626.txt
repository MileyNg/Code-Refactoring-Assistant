#include<stdio.h>

int main(void)
{
	int a,n,sum=0,i=0;
	int counter[26] = {0};
	char number = 0;
	char c = 97;
	while(scanf("%c",&number) != EOF)
	{
		if(number - 'a' < 0)
		{
			sum = number - 'A';
		}
		else
		{
			sum = number - 'a';
		}
		counter[sum]++;
		i++;
	}
	for(a = 0;a < 26;a++)
	{
		printf("%c :%2d\n",c,counter[a]);
		c++;
	}
		
	return 0;
}
