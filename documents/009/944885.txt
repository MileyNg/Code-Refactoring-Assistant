#include<stdio.h>
int main(void){
  char x,y;
  int a=1,b=0,c=0;
  int w=0;

  while(scanf("%c,%c",&x,&y) != EOF){
    if(x == 'A' && y =='B' || y == 'A' && x =='B'){
      w=a;a=b;b=w;}
    else if(x == 'A' && y == 'C' || y == 'A' && x == 'C'){
      w=a;a=c;c=w;}
    else if(x == 'B' && y == 'C' || y == 'B' && x == 'C'){
      w=b;b=c;c=w;}
    
  }
  if(a==1)printf("A\n");
  else if(b == 1)printf("B\n");
  else printf("C\n");
  return 0;
}
