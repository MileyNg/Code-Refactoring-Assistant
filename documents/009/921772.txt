#include<iostream>
using namespace std;

int main()
{
        int n;
        double ret, min, tmp;

        cin >> n >> min;
        ret = 0;

        for(int i=0; i<n-1; i++){
                cin >> tmp;
                if( ret < tmp-min ){ ret = tmp-min; }
                if( min > tmp ){ min = tmp; }

        }
        cout << ret << endl;

        return 0;
}
