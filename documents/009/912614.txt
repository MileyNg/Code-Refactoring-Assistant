#include <iostream>
using namespace std;

int main() {
char ch;
int num,count[26];
       while(cin >> ch){
        ch= tolower(ch);
        num = ch;
        if(num>=0 && num<26){
           count[num]++;
           }
          }
        for(int i=0;i<26;i++){
         cout << i +'a' <<" : " << count[i]<<endl;
}

        
	return 0;
}
