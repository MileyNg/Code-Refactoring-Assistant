while True:
    try:
        path = [0]*100
        while True:
            a,b = map(int, raw_input().split())
            if a == 0: break
            path[a-1] += 1; path[b-1] += 1
        if (path[0]%2 == path[1]%2 == 1 ):
            print "OK"
        else:
            print "NG"
    except:
        break
