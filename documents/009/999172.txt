#include <stdio.h>
#define N 250

int parent(int i){
  return i/2;
}

int Left(int i){
  return 2*i;
}

int Right(int i){
  return 2*i+1;
}   

main(){
  int i, n, h[N];
  scanf("%d",&n);
  for(i=1;i<n+1;i++){
    scanf("%d",&h[i]);
  }
  for(i=1;i<n+1;i++){
    if(i==1) {
      printf("node %d: key = %d, ", i, h[i]);
    }
    else if(i>1) {
      printf("node %d: key = %d, parent key = %d, ", i, h[i], h[parent(i)]);
      }
    if(Left(i) <= n){
      printf("left key = %d, ", h[Left(i)]);
    }
    if(Right(i) <= n){
      printf("right key = %d, ", h[Right(i)]);
    }
    printf("\n");
  }
   
  return 0;   
}
