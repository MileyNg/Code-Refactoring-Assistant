#include<stdio.h>
#define N 2000001
int A[N];

void QuickSort(int p, int r){
  int q;
  if (p<r){
    q = Partition(p,r);
    QuickSort(p,q-1);
    QuickSort(q+1,r);
  }
}

int Partition(int p, int r){
  int x,i,j,temp;
  x = A[r];
  i = p-1;
  for (j=p;j<=r-1;j++){
    if (A[j]<=x){
      i++;
      temp=A[i];
      A[i]=A[j];
      A[j]=temp;
    }
  }
  temp=A[i+1];
  A[i+1]=A[j];
  A[j]=temp;
  return i+1;
}

int main(){
  int n,i,key;
  scanf("%d",&n);
  for (i=0;i<n;i++){
    scanf("%d",&A[i]);
  }
  key=A[n-1];
  //QuickSort(0,n-1);
  Partition(0,n-1);

  for (i=0;i<n;i++){
    if (i!=0){
      printf(" ");
    }
    if (A[i]==key){
      printf("[");
    }
    printf("%d",A[i]);
    if (A[i]==key){
      printf("]");
    }
  }
  printf("\n");
  return 0;
}
