import std.stdio;
import std.conv;
import std.algorithm;
import std.range;
import std.string;

void main() {
  string s = readln.chomp;

  foreach (c; s) {
    if ('a' <= c && c <= 'z') {
      c.toUpper.write;
    } else if ('A' <= c && c <= 'Z') {
      c.toLower.write;
    } else {
      c.write;
    }
  }

  writeln;
}
