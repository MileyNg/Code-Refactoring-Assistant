#include<stdio.h>
int main(){
  int i,j;
  int n,m;

  while(1){
    scanf("%d%d", &n, &m);

    if(n==0 &&  m==0){
     break;}

    for(i=1;i<=n;i++){   
      if(i%2!=0){
	if(m%2==0){
	  for(j=1;j<=m/2;j++){
	    printf("#.");
	  }
	  printf("\n");
	}

	if(m%2!=0){
	  for(j=1;j<=m/2;j++){
	    printf("#.");
	  }
	  printf("#\n");}
      }
      
      if(i%2==0){
	if(m%2==0){
	  for(j=1;j<=m/2;j++){
	    printf(".#");
	  }
	  printf("\n");
	}
	
	if(m%2!=0){
	  for(j=1;j<=m/2;j++){
	    printf(".#");
	  }
	  printf(".\n");}
      }
      
    }
    printf("\n");
  }
  return 0;
}
