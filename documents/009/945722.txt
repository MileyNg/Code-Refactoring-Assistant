using namespace std;

bool field[25][25];

int dx[4] = {1, 0, -1, 0};
int dy[4] = {0, 1, 0, -1};
string compass = "NESW";

int main(){
    int n, m;
    while(cin >> n, n){
        int ans = 0;
        int mx=10, my=10;
        for(int i=0; i < 25; i++)for(int j=0; j < 25; j++) field[i][j] = false;

        for(int i=0; i < n; i++){
            int p, q;
            cin >> p >> q;
            field[p][q] = true;
        }
        cin >> m;
        for(int i=0; i < m; i++){
            string p; int q;
            cin >> p >> q;
            int position;
            for(int j=0; j < 4; j++) if(compass.substr(j, 1) == p) position = j;

            for(int j=1; j <= q; j++){
                if(field[my+dy[position]*j][mx+dx[position]*j]){
                    ans++;
                    field[my+dy[position]*j][mx+dx[position]*j] = false;
                }
            }
            mx += dx[position]*q; my += dy[position]*q;
        }
        if(n == ans) cout << "Yes" << endl;
        else cout << "No" << endl;
    }
}
