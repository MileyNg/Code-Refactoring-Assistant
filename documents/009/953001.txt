using namespace std;

#define reps(i,f,n) for(int i=f;i<int(n);i++)
#define rep(i,n) reps(i,0,n)

string z;
string y;

bool isaz(int p,string str){
	
	const int C = 333;
	int sum[C]={0};
	int sum2[C]={0};
	
	rep(i,z.size())sum[z[i]]++;
	reps(i,p,p+z.size())sum2[str[i]]++;
	
	
	rep(i,C){
		if(sum[i]!=sum2[i])return false;
	}
	return true;
}

string setaz(int p,string str){
	rep(i,z.size()){
		str[i+p] = y[i];
	}
	return str;
}
