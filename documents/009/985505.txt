#include<stdio.h>
int main(void){
	int t[3][3][3], i, j, k;
	for (i = 0; i < 3; i++){
		for (j = 0; j < 2; j++){
			for (k = 0; k < 3; k++){
				scanf("%d", &t[i][j][k]);
			}
		}
	}
	for (i = 0; i < 3; i++){
		for (j = 0; j < 2; j++){
			t[i][j][2] += t[i][j][1] * 60 + t[i][j][0] * 3600;
		}
		t[i][0][2] = t[i][1][2] - t[i][0][2];
		t[i][2][0] = t[i][0][2] / 3600;
		t[i][0][2] %= 3600;
		t[i][2][1] = t[i][0][2] / 60;
		t[i][2][2] = t[i][0][2] % 60;
	}
	for (i = 0; i < 3; i++){
		for (j = 0; j < 2; j++){
			printf("%d ", t[i][2][j]);
		}
		printf("%d\n",t[i][2][2]);
	}
	return 0;
}
