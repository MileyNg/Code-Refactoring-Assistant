#include<stdio.h>
#include<iostream>
#include<algorithm>
void solve(void);

int a,b,c,sum;
int n;
int date[1000][4]={0};
int at[1002];

using namespace std;

int main(){
  
  int i,j;
      
  while(1){
    
    scanf("%d %d %d",&a,&b,&c);
    
    if(a==0 && b==0 && c==0){
      break;
    }
    
    sum=a+b+c;

    for(i=0;i<sum;i++){
      at[i]=2;
    }
    
    scanf("%d",&n);
    
    for(i=0;i<n;i++){
      scanf("%d %d %d %d",&date[i][0],&date[i][1],&date[i][2],&date[i][3]);
    }
    
    solve();
	
    for(i=1;i<=sum;i++){
      printf("%d\n",at[i]);
    }
    
  }
  return 0;
}

void solve(){

  int i,j,k,cnt;

  for(i=0;i<n;i++){
    if(date[i][3]==1){
      for(j=0;j<3;j++){
	at[date[i][j]]=1;
      }
    }
  }

  for(i=0;i<n;i++){
    cnt=0;
    if(date[i][3]==0){
      for(k=0;k<3;k++){
	if(at[date[i][k]]==1){
	  cnt++;
	}
      }
    }
    if(cnt==2){
      for(j=0;j<3;j++){
	if(at[date[i][j]]!=1){
	  at[date[i][j]]=0;
	}
      }
    }
    if(cnt==1){
      for(j=0;j<3;j++){
      if(at[date[i][j]]!=1 && at[date[i][j]]!=0){
	at[date[i][j]]=2;
      }
      }
    }
  }
  
}
