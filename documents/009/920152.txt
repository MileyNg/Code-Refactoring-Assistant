#include <stdio.h>
#include <math.h>
#include <algorithm>
using namespace std;

int main() {
  int n;
  scanf("%d", &n);
  long long minv = 1e+9;
  long long ans = -1e+9;
  for (int i = 0; i < n; i++) {
    long long v;
    scanf("%lld", &v);
    ans = max(ans, v - minv);
    minv = min(minv, v);
  }
  printf("%lld\n", ans);
}
