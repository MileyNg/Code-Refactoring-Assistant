include<iostream>
#include<stdio.h>
using namespace std;

main() {

  int i, j, t, y;

  while(1){

    cin >> t >> y;
  
    if(t == 0 && y == 0){
      break;
    }else{
      
      for(i = 0; i < t; i++){
	for(j = 0; j < y; j++){
	  if(i != 0 && i != t-1 && j != 0 && j != y-1 ){
	    printf(".");
	  }else {
	    printf("#");
	  }
	}
	printf("\n");
      }
      printf("\n");
    }
  }
}
