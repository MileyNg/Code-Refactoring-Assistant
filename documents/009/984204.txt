#include<iostream>
#include<algorithm>
#include<bitset>

using namespace std;

void input(int *a,int n){
  for(int i=0;i<n;i++){
    for(int j=0;j<16;j++){
      int ai;
      cin>>ai;
      a[i]=a[i]*2+ai;
    }
  }
}


int main(){
  int pop[1<<16];
  for(int i=0;i<1<<16;i++){
    pop[i]=bitset<32>(i).count();
  }
  for(int n,c;cin>>n>>c,n;){
    int a[30]={},b[30]={};
    input(a,n);
    input(b,c);
    static int dp[31][1<<16];
    fill(dp[0],dp[31],-(1<<29));
    dp[0][0]=0;
    for(int i=0;i<n;i++){
      for(int j=0;j<1<<16;j++){
	for(int k=0;k<c;k++){
	  int l=j|a[i];
	  int &t=dp[i+1][l&~b[k]];
	  t=max<int>(t,dp[i][j]+pop[l&b[k]]);
	}
      }
    }
    cout<<*max_element(dp[0],dp[31])<<endl;
  }
}
