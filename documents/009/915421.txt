#include <stdio.h>

main(){
  int count;
  int i,j;
  int R[200000];
  int max;

  scanf("%d",&count);
  if(count<2)return 0;

  for(i=0;i<count;i++){
    scanf("%d",&R[i]);
  }

  max=R[1]-R[0];

  for(i=1;i<count;i++){
    for(j=2;j<count;j++){
      if(j<i)continue;
    if(max<R[j]-R[i])max=R[j]-R[i];
    }
  }

  printf("%d\n",max);
  return 0;

}
