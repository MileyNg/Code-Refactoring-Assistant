#include<iostream>
using namespace std;
int main(void)
{
  char a[10000];
  cin>>a;
  int countj=0,counti=0;
  for(int i=0;i<10000-3;i++)
    {
      if(a[i]=='J' && a[i+1]=='O' && a[i+2]=='I')countj++;
      if(a[i]=='I' && a[i+1]=='O' && a[i+2]=='I')counti++;
    }
  cout<<countj<<endl;
  cout<<counti<<endl;
  return 0;
}
