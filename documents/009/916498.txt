#include<iostream>
#include<string>
#include<vector>

using namespace std;

//Nは100種類以下

int dp[101][10001],w[101],v[101];
int main(){
	int N,W;
	cin >> N >> W;
	for(int i=0;i<N;i++){
		cin >> v[i];
		cin >> w[i];
	}

	for(int i=0;i<N;i++){
		for(int j=0;j<=W;j++){
			dp[i+1][j+w[i]] = max(dp[i][j+w[i]],dp[i][j]+v[i]);
		}
	}

	int ans=0;
	for(int i=0;i<=W;i++){
		ans = max(ans,dp[N][i]);
	}
	cout << ans << endl;
	return 0;
}
