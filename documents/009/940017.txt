module main;

import std.conv;
import std.math;
import std.stdio;
import std.string;

void main() {
  string[] strs = readln().split();
  float p1x = strs[0].to!float();
  float p1y = strs[1].to!float();
  float p2x = strs[2].to!float();
  float p2y = strs[3].to!float();
  writefln("%f", sqrt(pow(p2x - p1x, 2) + pow(p2y - p1y, 2)));
}
