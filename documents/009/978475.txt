#include<stdio.h>
int main(void)
{
	int h,w,i,j;
	scanf("%d %d",&h,&w);
	while(h!=0 || w!=0){
		for(i=0;i<h;i++){
			if(i%2==0){
				printf("#");
			}
			else if(i%2==1){
				printf(".");
			}
			for(j=1;j<w;j++){
				if(i%2==1){
					if(j%2==1){
						printf("#");
					}
					else if(j%2==0){
						printf(".");
					}
				}
				else if(i%2==0){
					if(j%2==1){
						printf(".");
					}
					else if(j%2==0){
						printf("#");
					}
				}
			}
			printf("\n");
		}
		scanf("%d %d",&h,&w);
	}
	printf("\n");
	return 0;
}
