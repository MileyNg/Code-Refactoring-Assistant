#include<iostream>
using namespace std;
int main(){
  int n,m;
  while(cin>>n>>m,n||m){
    int p[501][501]={0,0};
    int count = 0;
    int h[n+1];
    for(int i=0;i<=n;i++) h[i] = 0;
    h[1] = 1;
    int a,b;
    for(int i=0;i<m;i++){
      cin >> a >> b;
      p[a][b] = p[b][a] = 1;
    }
    for(int i=1;i<=n;i++){
      if(p[1][i]!=0){
	h[i] = 1;
	count++;
      }
    }
    for(int i=1;i<=n;i++){
      for(int j=1;j<=n;j++){
	if(h[i]==1){
	  if(p[i][j]!=0&&h[j]==0){
	    count++;
	    h[j] = 2;
	  }
	}
      }
    }
    cout << count << endl;
  }
  return 0;
}
