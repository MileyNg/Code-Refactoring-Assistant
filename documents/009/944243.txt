#include <stdio.h>

int main(int argc,const char * argv[]){
    int i,j,H,W;     //Hは高さ、Wは幅
    
    for(;;){
        scanf("%d %d",&H,&W);
        for(i=0;i<H;i++){     //縦のくりかえし
            for(j=0;j<W;j++){  //幅の繰り返し
                if(0<j && j<W-1 && 0<i && i<H-1){
                    printf(".");
                }
                else{
                    printf("#");
                }
            }
            
            printf("\n");
        }
        printf("\n");
    }
    
    return 0;
}
