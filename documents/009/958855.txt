#include<iostream>
#include<algorithm>
#include<cstring>
#include<cmath>
using namespace std;


int main(){
  int n,q;
  for(;cin>>n>>q;)
    {
      pair<int,int> tree[n*2-1];
      for(int i=0;i<n*2-1;i++)
	tree[i]=make_pair(0,0);
      for(int i=0;i<q;i++)
	{
	  int a,v;
	  cin>>a>>v;
	  a--;
	  int num=a+(n-1);
	  // cout<<num<<endl;
	  tree[num].first+=v;
	  tree[num].second=a+1;
	  pair<int,int> base;
	  while(true)
	    {
	      num=(num-1)/2;
	      if(num*2+2<n*2-1 &&( tree[num*2+1].first<tree[num*2+2].first|| (tree[num*2+1].first==tree[num*2+2].first &&tree[num*2+1].second>tree[num*2+2].second)))
		base=tree[num*2+2];
	      else
		base=tree[num*2+1];
	      tree[num]=base;
	      if(num==0)
		break;
	    }
	  cout<<tree[0].second<<" "<<tree[0].first<<endl;
	  //	  output();
	}
    }
}
