#include <stdio.h>
#define N 100000
int A[N];
 
int Partition( int p, int r ){
  int x,change;
  int i,j;
  x=A[r];
  i=p-1;

  for(j=p;j<r;j++){
    if(A[j]<=x)
      {
	j++;
	change=A[i];
	A[i]=A[j];
	A[j]=change;
      }
  }

  change=A[i+1];
  A[i+1]=A[r];
  A[r]=change;
  return i+1;
}



int main()
{
  int n,x,i;
  scanf("%d",&n);
  for(i=1;i<=n;i++)
    {
      scanf("%d",&A[i]);
      //配列
    }
  x=Partition(1,n);
  for(i=1;i<=n;i++)
    {
      if(i==x)
	{
	  printf("[%d]",A[i]);  
	}
      else
	{
	  printf("%d",A[i]);
	}
      if(i<n)
	{
	  printf(" ");
	}
    }
  printf("\n");
  return 0;
}
