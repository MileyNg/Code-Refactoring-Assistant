#include<stdio.h>
main(){
int n,i,j,tmp,a[100];
 scanf("%d",&n);
 for(i=0;i<n;i++){
   scanf("%d",&a[i]);
 }
 for(i=0;i<n-1;i++){
  for(j=n-1;j>i;j--){
    if(a[j]<a[j-1]){
      tmp=a[j];
      a[j]=a[j-1];
      a[j-1]=tmp;
    }
  }
}
for(i=0;i<n;i++){
  printf("%d",a[i]);
 }
 printf("\n");
 return 0;
}
