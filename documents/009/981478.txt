bool equation(string inp,int& i){
  bool right,left;
  left=formula(inp,i);
  i++;
  // cout<<"i:"<<i<<endl;
  right=formula(inp,i);
  //  cout<<alphaset<<endl<<left<<'='<<right<<endl;
  return right==left;
}
bool formula(string inp,int& i){
  bool left=alphabet(inp,i);
  bool right;
  while(true){
    if(inp[i]=='+'){
      right=formula(inp,++i);
      left=right||left;
    }
    else    if(inp[i]=='*'){
      right=formula(inp,++i);
      left=right&&left;
    }
    else    if(inp[i]=='-'){
      if(inp[++i]=='>'){
	right=formula(inp,++i);
	left=(!left)||right;
      }
    }
    else return left;
  }
}
bool alphabet(string inp,int& i){
  if(inp[i]=='(')
    {
      int out=formula(inp,++i);
      i++;return out;
    }
  else if(inp[i]=='-')
    return !alphabet(inp,++i);
  else if(inp[i]=='T'){i++;
    return true;}
  else if(inp[i]=='f'){i++;
    return false;}

  else{
    return alphaset&(1<<(inp[i++]-'a'))==true;
  }

}
