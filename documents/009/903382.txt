int TakeCustomNum(string s){
	int val = 0, digit = 10;
	FOR(i, s.size()){
		if (s[i] == ',') break;
		else{
			val *= digit;
			val += s[i] - '0';
		}
	}
	return val;
}

void InputLastData(map<int, int> &ans){
	string s;
	while (getline(cin, s) && s != "\0"){
		ans[TakeCustomNum(s)]++;
	}
}

void InputCurrentData(map<int, int> &prev, map<int, int> &now){
	string s;
	while (getline(cin, s) && s != "\0"){
		int key = TakeCustomNum(s);

		if (now.count(key)) now[key]++;
		else if (prev.count(key)) now[key] += prev[key] + 1;
	}
}

void OutputData(map<int, int> &ans){
	for(auto i: ans){
		if (i.second > 1){
			cout << i.first << " " << i.second << endl;
		}
	}
}

int main(){
	map<int, int> prev, now;
	
	InputLastData(prev); 
	InputCurrentData(prev, now);

	OutputData(now);

	return 0;
}
