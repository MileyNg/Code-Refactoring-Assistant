#include <stdio.h>

int main(void)
{
	float x1, y1, x2, y2, x3, y3, x4, y4;
	float abx, aby, cdx, cdy;
	float ans, ans2;
	int n;
	int a;
	
	scanf("%d", &n);
	
	for (a = 0; a < n; a++) {
		scanf("%f %f %f %f %f %f %f %f", &x1, &y1, &x2, &y2, &x3, &y3, &x4, &y4);
		abx = x2 - x1;
		aby = y2 - y1;
		cdx = x4 - x3;
		cdy = y4 - y3;
		
		ans = aby / abx;
		ans2 = cdy / cdx;
		
		if (ans == ans2) {
			printf("YES\n");
		}
		else {
			printf("NO\n");
		}
	}
	return (0);
}
