char change[100];
int main(){
	vector<string> ans;
	while(1){
		REP(i,100)change[i]='?';
		int n;
		cin>>n;
		if(n==0)break;
		REP(i,n){
			char t1,t2;
			cin>>t1>>t2;
			change[(unsigned int)(t1-'0')]=t2;
		}
		int m;
		cin>>m;
		string temp;
		REP(i,m){
			char tt,ss;
			cin>>tt;
			ss=(change[(unsigned int)(tt-'0')]!='?')?change[(unsigned int)(tt-'0')]:tt;
			temp+=ss;
		}
		ans.push_back(temp);
	}
	REP(i,ans.size())cout<<ans[i]<<endl;
	return 0;
}
