int main() {
  int n, max;
  bool b[10];
  int w[10] = {512, 256, 128, 64, 32, 16, 8, 4, 2, 1};

  while(cin >> n) {
    fill(b, b+10, false);
    max = 0;    
    for(int i = 0; i < 10; i++) {
      if(n/w[i] > 0) {
	b[i] = true;
	n -= w[i];
	if(max < w[i]) max = w[i];
      }
    }

    for(int i = 9; i >= 0; i--) {
      if(b[i]) {
	cout << w[i];
	if(w[i] == max) cout << endl;
	else cout << ' ';
      }
    }
  }
}
