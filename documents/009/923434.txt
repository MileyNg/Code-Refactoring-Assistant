#include<iostream>
#include<algorithm>

using namespace std;
int W,H;
void solve(int,int);
char dfs[20][20];
int d_x[]={1,0,-1,0},d_y[]={0,1,0,-1};

int main(){
	
	while(1){
		int ans=0;
		cin>>W>>H;
		if(W==0&&H==0)	break;

		int dx,dy;
		for(int i=0;i<H;i++){
			for(int j=0;j<W;j++){
				cin>>dfs[j][i];
				if(dfs[j][i]=='@'){
					dx=j;
					dy=i;
				}
			}
		}
		solve(dx,dy);
		for(int i=0;i<H;i++)
			for(int j=0;j<W;j++)
				if(dfs[j][i]=='T')
					ans++;
		cout<<ans<<endl;
		memset(dfs,0,sizeof(dfs));
	}
}

void solve(int x,int y){
	dfs[x][y]='T';
	
	for(int j=0;j<4;j++){
		int mx=x+d_x[j],my=y+d_y[j];
		if(mx<0||my<0||mx>W||my>H)	continue;
		if(dfs[mx][my]=='.')
			solve(mx,my);
	}
	return ;
}
