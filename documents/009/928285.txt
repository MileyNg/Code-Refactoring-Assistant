  while( scanf("%s", s) != EOF ){
    if ( s[0] == '+' ){

      i--;
      sgym[i-1]+=sgym[i];
     } else if ( s[0] == '-' ){
      i--;
      sgym[i-1] -= sgym[i];

 i--;
