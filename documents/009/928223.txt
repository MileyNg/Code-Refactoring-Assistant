#include<stdio.h>
#include<stdlib.h>
#include<string.h>

int main(){
  int x,i=0,store[100];
  char s[100];

  for(i=0;scanf("%s", s) != EOF;i++){
    if ( s[0] == '+' ){
      store[i-2]=store[i-2]+store[i-1];
      printf("i=%d %d\n",i-2,store[i-2]);
     i=i-2;
 
    } else if ( s[0] == '-' ){
      store[i-2]=store[i-2]-store[i-1];
      printf("i=%d %d\n",i-2,store[i-2]);
      i=i-2;

    } else if ( s[0] == '*' ){
      store[i-2]=store[i-2]*store[i-1];
      printf("i=%d %d\n",i-2,store[i-2]);
    i=i-2;
 
    } else {
      x = atoi(s);
      store[i]=x;
    
    }
   
  }
  printf("%d\n",store[0]);
 
  return 0;
}
