#include <iostream>
#include <vector>
using namespace std;
int main() {
    ios::sync_with_stdio(false);
    vector<int> v(1000, 0);
    int a, b;
    while ( ! cin.eof()) {
        while (1) {
            cin >> a >> b;
            if ( ! a && ! b) break;
            v[a] ++;
            v[b] ++;
        }
        bool ok = true;
        if (v[1] % 2 == 0 || v[2] % 2 == 0) {
            ok = false;
        }
        for (int i=3; i<1000; ++i) {
            if (v[i] % 2 != 0) {
                ok = false;
            }
        }
        cout << (ok ? "OK" : "NG") << endl;
    }
    return 0;
}
