#include <stdio.h>

int main(){

  int n,m,i,a[10000],b[10000],k,j,x,y,z;
  m=0;
  
  scanf("%d",&n);

  for(i=0;i<n;i++){
    scanf("%d",&a[i]);
  }
  scanf("%d",&k);
  for(i=0;i<k;i++){
    scanf("%d",&b[i]);
  }
 

  for(j=0;j<k;j++){
    x=n;
    y=0;
    while(x>=y){
      z=(x+y)/2;    
      if(a[z]>b[j]){
	x=z;
      }
      else if(a[z]<b[j]){
	y=z+1;
      }
      else if(a[z]==b[j]){
	m++;
	break;
      }
    }

  }
  printf("%d\n",m);
  return 0;
}
