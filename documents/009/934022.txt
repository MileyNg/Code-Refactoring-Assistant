#include<iostream>
using namespace std;

int main()
{
	int trnp[4][13] = {0}; //[4]:S>H>C>D [13]:tranp1ã€œ13;
	int n,i,j,num;
	char mark;

	cin >> n;

	for(i = 0;i < n;i++){
		cin >> mark >> num;
		
		switch(mark){
			case 'S':
				trnp[0][num] = 1;
				break;
			case 'H':
				trnp[1][num] = 1;
				break;
			case 'C':
				trnp[2][num] = 1;
				break;
			case 'D':
				trnp[3][num] = 1;
				break;
		}
	}

	for(i = 0;i < 4;i++){
		for(j = 1;j <= 13;j++){
			if(trnp[i][j] != 1){
				switch(i){
					case 0:
						cout << 'S' << " " << j << endl;
						break;
					case 1:
						cout << 'H' << " " << j << endl;
						break;
					case 2:
						cout << 'C' << " " << j << endl;
						break;
					case 3:
						cout << 'D' << " " << j << endl;
						break;
				}
			}
		}
	}

	return 0;
}
