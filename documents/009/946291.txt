#include<stdio.h>
#include<string.h>
int main(void){
    int a[3]={1,0,0},work;
    char c1,c2,g;
    while(scanf("%c,%c",&c1,&c2)!=EOF){
        work=a[c1-65];
        a[c1-65]=a[c2-65];
        a[c2-65]=work;
    }
    while(1){
        int i=0;
        if(a[i]==1){
            g=i+'A';
            printf("%c\n",g);
            break;
        }
        i++;
    }
    return 0;
}
