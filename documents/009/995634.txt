#include <stdio.h>
 
int main(void){
    int n,input[1024];

    scanf("%d", &n);
    
    for(int i=0;i<n;i++)
    {
        scanf("%d", &input[i]);
    }
    for(int i=n-1;i>-1;i--)
    {
        if(n==0)
       {
            printf("%d\n",input[i]);
        }else
        {
            printf("%d ",input[i]);
        }

    }
    return 0;
}
