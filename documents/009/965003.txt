#include <stdio.h>

short person[1000000];

int main(void)
{
	int n, q;
	short max = 0;
	int max_num = -1;
	
	scanf("%d %d", &n, &q);
	while (q-- > 0){
		int a;
		short v;
		int i;
		
		scanf("%d %hd", &a, &v);
		person[a - 1] += v;
		
		if (max < person[a - 1]){
			max = person[a - 1];
			max_num = a;
		}
		else if (max_num == a && v < 0){
			max_num = -1;
			for (i = 0; i < n; i++){
				if (max_num == -1 || max < person[i]){
					max = person[i];
					max_num = i + 1;
				}
			}
		}
		printf("%d %hd\n", max_num, max);
	}
	
	return 0;
}
