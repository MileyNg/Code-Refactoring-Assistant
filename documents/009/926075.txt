#include <stdio.h>

int main(void) {
	// your code goes here
	int z, a,b,c,d,ans=0;
	while(scanf("%d",&z)!=EOF) {
		for(a = 0;a < 10; a++) {
			for(b=0;b < 10; b++) {
				for(c=0;c < 10; c++) {
					for (d=0;d < 10; d++) {
						if((a+b+c+d) == z) {
							ans += 1;
						}
					}
				}
			}
		}
				
		printf("%d",ans);
		ans = 0;
	}
	return 0;
}
