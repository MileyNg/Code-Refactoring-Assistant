#include <stdio.h>

short person[1000000];

int main(void)
{
	int n, q;
	
	scanf("%d %d", &n, &q);
	while (q-- > 0){
		int a;
		short v;
		int i;
		int max;
		int max_num = -1;
		
		scanf("%d %hd", &a, &v);
		person[a - 1] += v;
		
		if (max_num != a - 1 || v < 0){
			for (i = 0; i < n; i++){
				if (max_num == -1 || max < person[i]){
					max = person[i];
					max_num = i + 1;
				}
			}
		}
		printf("%d %d\n", max_num, max);
	}
	
	return 0;
}
