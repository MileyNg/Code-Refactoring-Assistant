#include <stdio.h>
#include <math.h>

int main(void)
{
	int n, i;
	int ans;
	float xa, ya, ra,
		  xb, yb, rb;
	float x, y, len, tmp;
	
	scanf("%d", &n);
	
	for (i = 0; i < n; i++){
		scanf("%f %f %f %f %f %f", &xa, &ya, &ra, &xb, &yb, &rb);
		
		x = xb - xa;
		y = yb - ya;
		
		len = sqrt((x * x) + (y * y));
		
		/*
		printf("%f %f\n", x, y);
		printf("%f\n", len);
		*/
		
		if (len <= ra + rb){
			if (len + rb < ra){
				ans = 2;
			}
			else if (len + ra < rb){
				ans = -2;
			}
			else {
				ans = 1;
			}
		}
		else if (len > ra + rb){
			ans = 0;
		}
		
		printf("%d\n", ans);
	}
	
	return (0);
}
