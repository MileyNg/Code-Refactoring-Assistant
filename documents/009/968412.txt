#include<iostream>
#include<cstring>

int res[50];

void calc(int a, int b){
	if (a == 0){
		for (int i = 1; i < b; i++)std::cout << res[i] << " ";
		std::cout << std::endl;
		return;
	}
	for (int i = a; i >= 1; i--){
		if (res[b - 1] >= i){
			res[b] = i;
			calc(a - i, b + 1);
		}
	}
	return;
}


int main(){

	int n;
	while (std::cin >> n, n){
		memset(res, 0, sizeof(res));
		res[0] = 10000000;
		calc(n, 1);
	}

	return 0;
}
