#include <iostream>
#include <cstdio>
#include <string>

using namespace std;

int main(){
	string s;
	char ch;
	int flag[26];

	for (int i = 0; i < 26; i++){
		flag[i] = 0;
	}

	while (getline(cin, s)){
		for (int i = 0; i < s.length(); i++){
			ch = s[i];
			flag[ch - 'a']++;
		}
		for (int i = 0; i < 26; i++){
			printf("%c : %d", 'a' + i, flag[i]);
		}
	}
}
