include <stdio.h>

int main(void)
{
  int x;
  int y;
  scanf("%d %d", &x, &y);
  if(x > y ){
    printf("%d %d\n", y , x);
  }
  else if(x < y){
    printf("%d %d\n", x , y);
  }
  else if(x == 0 && y == 0){//例外処理って面倒だな                              
  }
  else {
    printf("%d %d\n", x, y);
  }
  return 0;
}//綱野雄太
